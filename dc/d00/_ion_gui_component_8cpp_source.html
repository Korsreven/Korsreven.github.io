26bool is_descendant_of(const GuiContainer &owner, const GuiComponent &component) noexcept
27{
28 if (auto component_owner = component.Owner(); component_owner == &owner)
29 return true;
30 else if (component_owner)
31 return is_descendant_of(owner, *component_owner); //Recursive
32 else
33 return false;
34}

157void GuiComponent::Parent(GuiComponent &parent) noexcept
158{
159 if (owner_ == &parent || owner_ == parent.Owner())
160 {
161 if (node_ && node_->ParentNode())
162 parent.Node()->Adopt(node_->ParentNode()->Orphan(*node_));
163
164 parent_ = &parent;
165 }

168void GuiComponent::Owner(GuiContainer &owner) noexcept
169{
170 managed::ManagedObject<GuiContainer>::Owner(owner);
171 Detach();
172
173 parent_ = &owner;
174 node_ = parent_->Node() ?
175 parent_->Node()->CreateChildNode({}, visible_ && parent_->Node()->Visible()) :
176 nullptr;
177}

194void GuiComponent::GlobalZOrder(real z) noexcept
195{
196 if (node_)
197 {
198 auto [x, y, _] = node_->Position().XYZ();
199 auto parent_z = node_->ParentNode() ?
200 node_->ParentNode()->DerivedPosition().Z() : 0.0_r;
201 node_->Position({x, y, z - parent_z});
202 //Faster than calling DerivedPosition when changing z
203 }
204}
