33void NodeAnimationTimeline::ResetCycle() noexcept
34{
35 current_time_ = 0.0_sec;
36 reverse_ = false;
37
38 for (auto &animation : AttachedAnimations())
39 animation.Reset();
40
41 for (auto &animation_group : AttachedAnimationGroups())
42 animation_group.Reset();
43}

45duration NodeAnimationTimeline::RetrieveTotalDuration() const noexcept
46{
47 auto total_duration = 0.0_sec;
48
49 if (!std::empty(attached_animations_))
50 {
51 if (auto animation_group = attached_animations_.back().group_ptr; animation_group)
52 total_duration = std::max(total_duration_, animation_group->StartTime() + animation_group->TotalDuration());
53
54 else if (auto animation = attached_animations_.back().ptr; animation)
55 total_duration = std::max(total_duration_, animation->StartTime() + animation->TotalDuration());
56 }
57
58 return total_duration;
59}

180void NodeAnimationTimeline::Revert(duration total_duration)
181{
182 //Something to revert
183 if (current_time_ > 0.0_sec)
184 {
185 reverse_ = true;
186
187 //Over time
188 if (total_duration > 0.0_sec)
189 reverse_playback_rate_ = current_time_ / total_duration;
190 else //Instantaneously
191 {
192 reverse_playback_rate_ = 1.0_r;
193 Elapse(current_time_);
194 }

313void NodeAnimationTimeline::Elapse(duration time) noexcept
314{
315 if (running_)
316 {
317 if (reverse_)
318 time = -time;
319
320 time *= (reverse_ ? reverse_playback_rate_ : playback_rate_);
321 current_time_ += time;
322
323 //Reverse
324 if (reverse_)
325 {
326 for (auto &a : adaptors::ranges::ReverseIterable<detail::attached_animations&>(attached_animations_))
327 a.ptr->Elapse(time, current_time_, a.group_ptr ? a.group_ptr->StartTime() : 0.0_sec);
328 }
329 else //Forward
330 {
331 for (auto &a : attached_animations_)
332 a.ptr->Elapse(time, current_time_, a.group_ptr ? a.group_ptr->StartTime() : 0.0_sec);
333 }
334
335
336 //A timeline cycle has been completed
337 if (reverse_ ?
338 current_time_ <= 0.0_sec :
339 current_time_ >= total_duration_)
340 {
341 //Loop (next cycle)
342 if (!reverse_ && //Forwarding
343 (!repeat_count_ || //Indefinitely
344 repeat_count_->first < repeat_count_->second))
345 {
346 if (repeat_count_)
347 ++repeat_count_->first;
348
349 //Finish cycle
350 if (on_finish_cycle_ &&
351 current_time_ >= total_duration_ && current_time_ - time < total_duration_)
352 (*on_finish_cycle_)(*this);
353
354 ResetCycle();
355 }
356 //Timeline is done
357 else
358 {
359 if (reverse_)
360 {
361 //Finish revert
362 if (on_finish_revert_ &&
363 current_time_ <= 0.0_sec && current_time_ - time > 0.0_sec)
364 (*on_finish_revert_)(*this);
365 }
366 else
367 {
368 //Finish
369 if (on_finish_ &&
370 current_time_ >= total_duration_ && current_time_ - time < total_duration_)
371 (*on_finish_)(*this);
372 }
373
374 Reset();
375 }
376 }
377 }
