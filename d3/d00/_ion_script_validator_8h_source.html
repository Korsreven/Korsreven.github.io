120 inline auto HasDefinition() const noexcept
121 {
122 return std::holds_alternative<T>(entity_);
123 }
124
127 inline auto& Name() const noexcept
128 {
129 return HasDefinition() ?
130 std::get<T>(entity_).Name() :
131 std::get<std::string>(entity_);
132 }
133
135 inline auto& Definition() const noexcept
136 {
137 assert(HasDefinition());
138 return std::get<T>(entity_);

142 inline auto Required() const noexcept
143 {
144 return ordinality_ == Ordinality::Mandatory;
145 }

155 inline auto operator<(const declaration &rhs) const noexcept
156 {
157 return Name() < rhs.Name();
158 }

163 inline auto operator<(const T &rhs) const noexcept
