276 [[nodiscard]] inline auto operator*(const Vector2 &vector) const noexcept
277 {
278 auto [x, y] = vector.XY();
279
280 #ifdef ION_ROW_MAJOR
281 //Row-major layout (Direct3D)
282 auto inv_w = 1.0_r / (m_[0][2] * x + m_[1][2] * y + m_[2][2]);
283 return Vector2{(m_[0][0] * x + m_[1][0] * y + m_[2][0]) * inv_w,
284 (m_[0][1] * x + m_[1][1] * y + m_[2][1]) * inv_w};
285 #else
286 //Column-major layout (OpenGL)
287 auto inv_w = 1.0_r / (m_[2][0] * x + m_[2][1] * y + m_[2][2]);
288 return Vector2{(m_[0][0] * x + m_[0][1] * y + m_[0][2]) * inv_w,
289 (m_[1][0] * x + m_[1][1] * y + m_[1][2]) * inv_w};
290 #endif
291 }
