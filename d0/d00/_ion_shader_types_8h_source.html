281 template <typename T>
282 class ValueAccessorBase
283 {
284 protected:
285
286 basic_type_t<T> *values_ = nullptr;
287 int off_ = 0;
288
289 public:
290
291 ValueAccessorBase() = default;
292
294 explicit ValueAccessorBase(basic_type_t<T> *values, int off = 0) noexcept :
295 values_{values},
296 off_{off}
297 {
298 //Empty
299 }
300
301
308 inline auto& operator=(basic_type_t<T> value) noexcept
309 {
310 for (auto i = 0; i < type_components_v<T>; ++i)
311 values_[type_components_v<T> * off_ + i] = value;
312
313 return *this;
314 }
315
317
324 [[nodiscard]] inline auto Values() noexcept
325 {
326 return &values_[type_components_v<T> * off_];
327 }
328
330 [[nodiscard]] inline auto Values() const noexcept
331 {
332 return &values_[type_components_v<T> * off_];
333 }

767 template <typename T, int N, int M>
768 struct ValueAccessor<Mat<N, M, T>> : ValueAccessorBase<Mat<N, M, T>>
769 {
770 using ValueAccessorBase<Mat<N, M, T>>::ValueAccessorBase;
771
772
779 inline auto& operator=(T value) noexcept
780 {
781 ValueAccessorBase<Mat<N, M, T>>::operator=(value);
782 return *this;
783 }
784
786 template <typename Matrix,
787 typename = std::enable_if_t<(N == 2 && M == 2 && std::is_same_v<Matrix, Matrix2>) ||
788 (N == 3 && M == 3 && std::is_same_v<Matrix, Matrix3>) ||
789 (N == 4 && M == 4 && std::is_same_v<Matrix, Matrix4>)>>
790 inline auto& operator=(const Matrix &matrix) noexcept
791 {
792 for (auto i = 0; i < N; ++i)
793 {
794 for (auto j = 0; j < M; ++j)
795 this->Values()[M * i + j] = static_cast<T>(matrix.M()[i][j]);
796 }
797
798 return *this;
799 }
800
802 [[nodiscard]] inline auto operator[](int column_off) noexcept
803 {
804 assert(column_off >= 0 && column_off < N);
805 return ValueAccessor<Vec<M, T>>{this->Values() + M * column_off};
806 }

928 template <typename T>
929 class UniformValue : public ShaderValue<T>, public ValueAccessor<T>
930 {
931 private:
932
933 const int size_ = 1;
934
935 detail::sbo_container_type<basic_type_t<T>, type_components_v<T>> stack_values_{}; //SBO (small buffer optimization)
936 detail::container_type<basic_type_t<T>> heap_values_; //Heap allocated if size is greater than 1
937 basic_type_t<T> *external_values_ = nullptr;
938
939
940 inline auto Data() noexcept
941 {
942 return external_values_ ? external_values_ :
943 size_ > 1 ? std::data(heap_values_) : std::data(stack_values_);
944 }
945
946 public:
947
949 explicit UniformValue(int size = 1) :
950
951 size_{size > 0 ? size : 1},
952 heap_values_{size_ > 1 ?
953 decltype(heap_values_)(type_components_v<T> * size_, basic_type_t<T>{}) :
954 decltype(heap_values_){}}
955 {
956 ValueAccessor<T>::values_ = Data();
957 }
958
960 UniformValue(const UniformValue &rhs) :
961
962 size_{rhs.size_},
963 stack_values_{rhs.stack_values_},
964 heap_values_{rhs.heap_values_},
965 external_values_{rhs.external_values_}
966 {
967 ValueAccessor<T>::values_ = Data();
968 }
969
971 UniformValue(UniformValue &&rhs) noexcept :
972
973 size_{rhs.size_},
974 stack_values_{std::move(rhs.stack_values_)},
975 heap_values_{std::move(rhs.heap_values_)},
976 external_values_{rhs.external_values_}
977 {
978 ValueAccessor<T>::values_ = Data();
979 }
980
981
988 inline auto& operator=(const UniformValue &rhs) noexcept
989 {
990 if (size_ == rhs.size_)
991 {
992 stack_values_ = rhs.stack_values_;
993 heap_values_ = rhs.heap_values_;
994 external_values_ = rhs.external_values_;
995 }
996
997 return *this;
998 }
999
1001 inline auto& operator=(basic_type_t<T> value) noexcept
1002 {
1003 ValueAccessor<T>::operator=(value);
1004 return *this;
1005 }
1006
1008 template <typename U>
1009 inline auto& operator=(const U &value) noexcept
1010 {
1011 ValueAccessor<T>::operator=(value);
1012 return *this;
1013 }
1014
1016 [[nodiscard]] inline auto At(int off) noexcept
1017 {
1018 assert(off < size_);
1019 return ValueAccessor<T>{Data(), off};
1020 }
1021
1023
1030 [[nodiscard]] inline auto Size() const noexcept
1031 {
1032 return size_;
1033 }
