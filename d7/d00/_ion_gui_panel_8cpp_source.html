28void GridCell::Adopt(SceneNode &node)
29{
30 if (auto grid = Owner(); grid)
31 {
32 if (auto panel = grid->Owner(); panel)
33 {
34 if (!node_)
35 node_ = panel->Node()->CreateChildNode({}, Position(), vector2::UnitY);
36
37 node_->Adopt(panel->Node()->Orphan(node));
38 }
39 }
40}

42void GridCell::Orphan(SceneNode &node)
43{
44 if (auto grid = Owner(); grid)
45 {
46 if (auto panel = grid->Owner(); panel)
47 {
48 panel->Node()->Adopt(node_->Orphan(node));
49
50 if (std::empty(node_->ChildNodes()))
51 {
52 node_->ParentNode()->RemoveChildNode(*node_);
53 node_ = nullptr;
54 }
55 }
56 }

60void GridCell::Align(controls::GuiControl &control) noexcept
61{
62 if (auto size = control.Size(); size && node_)
63 {
64 auto position = vector2::Zero;
65
66 switch (alignment_)
67 {
68 case GridCellAlignment::Left:
69 position.X(size->X() * 0.5_r);
70 break;
71
72 case GridCellAlignment::Right:
73 position.X(-size->X() * 0.5_r);
74 break;
75 }
76
77 switch (vertical_alignment_)
78 {
79 case GridCellVerticalAlignment::Top:
80 position.Y(-size->Y() * 0.5_r);
81 break;
82
83 case GridCellVerticalAlignment::Bottom:
84 position.Y(size->Y() * 0.5_r);
85 break;
86 }
87
88 control.Node()->Position(position);
89 }
90}

137Vector2 GridCell::Position() const noexcept
138{
139 if (auto grid = Owner(); grid)
140 {
141 if (auto panel = grid->Owner(); panel)
142 {
143 auto [grid_width, grid_height] = grid->Size().XY();
144 auto [cell_width, cell_height] = Size().XY();
145 auto [row, column] = Offset();
146
147 auto position = panel->Node()->Position() +
148 Vector2{-grid_width * 0.5_r + cell_width * 0.5_r, grid_height * 0.5_r - cell_height * 0.5_r} +
149 Vector2{cell_width, -cell_height} * Vector2{static_cast<real>(column), static_cast<real>(row)};
150
151 switch (alignment_)
152 {
153 case GridCellAlignment::Left:
154 position.X(position.X() - cell_width * 0.5_r);
155 break;
156
157 case GridCellAlignment::Right:
158 position.X(position.X() + cell_width * 0.5_r);
159 break;
160 }
161
162 switch (vertical_alignment_)
163 {
164 case GridCellVerticalAlignment::Top:
165 position.Y(position.Y() + cell_height * 0.5_r);
166 break;
167
168 case GridCellVerticalAlignment::Bottom:
169 position.Y(position.Y() - cell_height * 0.5_r);
170 break;
171 }
172
173 return position;
174 }
175 }
176
177 return vector2::Zero;
178}

208bool GridCell::AttachControl(NonOwningPtr<controls::GuiControl> control)
209{
210 if (auto grid = Owner(); grid && control && control->Node() &&
211 control->Owner() == grid->Owner() && //Control and grid has same owner
212 control->Parent() == grid->Owner() && //Parent of the control is the owner of the grid
213 control->Node()->ParentNode() == grid->Owner()->Node().get())
214 //Parent node of the control is the node of the owner of the grid
215 {
216 Adopt(*control->Node()); //Adopt
217 Align(*control);
218 controls_.push_back(control);
219 return true;
220 }
221 else
222 return false;
223}
224
225bool GridCell::DetachControl(controls::GuiControl &control) noexcept
226{
227 auto iter =
228 std::find_if(std::begin(controls_), std::end(controls_),
229 [&](auto &x) noexcept
230 {
231 return x.get() == &control;
232 });
233
234 //Control found
235 if (iter != std::end(controls_))
236 {
237 Orphan(*control.Node()); //Orphan
238 controls_.erase(iter);
239 return true;
240 }
241 else
242 return false;
243}

245void GridCell::DetachAllControls() noexcept
246{
247 for (auto &control : controls_)
248 {
249 if (control)

251 }
252
253 controls_.clear();
254 controls_.shrink_to_fit();
