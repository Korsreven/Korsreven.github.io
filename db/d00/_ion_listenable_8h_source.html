188 auto Subscribe(T &listener)
189 {
190 auto iter =
191 std::find_if(std::begin(listeners_), std::end(listeners_),
192 [&](auto &x) noexcept
193 {
194 return x == &listener;
195 });
196
197 //Listener not already subscribed
198 if (iter == std::end(listeners_) &&
199 NotifySubscribable(listener))
200 {
201 listeners_.push_back(&listener);
202 NotifySubscribed(listener);
203 return true;
204 }
205 else
206 return false;
207 }

217 auto Unsubscribe(T &listener) noexcept
218 {
219 auto iter =
220 std::find_if(std::begin(listeners_), std::end(listeners_),
221 [&](auto &x) noexcept
222 {
223 return x == &listener;
224 });
225
226 //Listener found
227 if (iter != std::end(listeners_) &&
228 NotifyUnsubscribable(listener))
229 {
230 listeners_.erase(iter);
231 NotifyUnsubscribed(listener);
232 return true;
233 }
234 else
235 return false;
236 }

239 void UnsubscribeAll() noexcept
240 {
241 //[listeners to keep, listeners to unsubscribe]
242 auto iter =
243 std::stable_partition(std::begin(listeners_), std::end(listeners_),
244 [&](auto &listener) noexcept
245 {
246 return !NotifyUnsubscribable(*listener);
247 });
248
249 decltype(listeners_) listeners;
250 std::move(iter, std::end(listeners_), std::back_inserter(listeners));
251 listeners_.erase(iter, std::end(listeners_)); //Erase the moved range
252 listeners_.shrink_to_fit();
253 NotifyUnsubscribedAll(listeners);
