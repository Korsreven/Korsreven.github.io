30std::string current_date_time()
31{
32 std::array<char, 32> data;
33 auto now = std::chrono::system_clock::to_time_t(std::chrono::system_clock::now());
34 return std::strftime(std::data(data), std::size(data), "%F %T", std::localtime(&now)) != 0 ?
35 std::data(data) : "";

38std::string print_output(std::filesystem::path file_path, duration built_time, ScriptError error, OutputOptions output_options)
39{
40 std::string output;
41
42 if (output_options == OutputOptions::Header ||
43 output_options == OutputOptions::HeaderAndSummary)
44 {
45 if (!std::empty(output))
46 output += "\n\n";
47
48 output += ion::utilities::string::Concat(
49 file_path.filename().string(), "\n",
50 current_date_time());
51 }
52
53 if (output_options == OutputOptions::HeaderAndSummary)
54 {
55 if (!std::empty(output))
56 output += "\n\n";
57
58 output += ion::utilities::string::Concat(
59 "[Build summary]\n"
60 "Message - ",
61 !error ?
62 "Build succeeded!" :
63 ion::utilities::string::Concat("Build failed. ", error.Condition.message()),
64 "\n"
65 "Built time - ", ion::utilities::string::Format(built_time.count(), "0.0000"sv), " seconds");
66 }
67
68 return output;
69}

153std::string ScriptBuilder::PrintOutput(OutputOptions output_options) const
154{
155 auto error = compile_error_ ?
156 ScriptError{compile_error_} :
157 ScriptError{validate_error_};
158
159 auto output = detail::print_output(compile_error_.FilePath,
160 compiler_.CompileTime() + (validator_ ? validator_->ValidateTime() : duration{}),
161 error, output_options);
162
163 if (compiler_output_options_)
164 {
165 if (!std::empty(output))
166 output += "\n\n";
167
168 output += compiler_.PrintOutput(*compiler_output_options_);
169 }
170
171 if (!compile_error_ && validator_ && validator_output_options_)
172 {
173 if (!std::empty(output))
174 output += "\n\n";
175
176 output += validator_->PrintOutput(*validator_output_options_);
177 }
178
179 if (tree_ && tree_print_options_)
180 {
181 if (!std::empty(output))
182 output += "\n\n[Tree view]";
183
184 output += tree_->Print(*tree_print_options_);
185 }
186
187 return output;
188}
