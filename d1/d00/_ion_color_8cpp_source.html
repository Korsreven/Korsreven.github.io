25std::tuple<real, real, real> hue_to_rgb(real hue, real c, real x) noexcept
26{
27 if (hue < 60.0_r)
28 return std::tuple{c, x, 0.0_r};
29 else if (hue < 120.0_r)
30 return std::tuple{x, c, 0.0_r};
31
32 else if (hue < 180.0_r)
33 return std::tuple{0.0_r, c, x};
34 else if (hue < 240.0_r)
35 return std::tuple{0.0_r, x, c};
36
37 else if (hue < 300.0_r)
38 return std::tuple{x, 0.0_r, c};
39 else //less than 360.0
40 return std::tuple{c, 0.0_r, x};
41}

43real rgb_to_hue(real red, real green, real blue, real c_max, real delta) noexcept
44{
45 if (delta == 0.0_r)
46 return 0.0_r;
47
48 real hue = [=]() noexcept
49 {
50 if (c_max == red)
51 return (green - blue) / delta;
52 else if (c_max == green)
53 return (blue - red) / delta + 2.0_r;
54 else //blue is max
55 return (red - green) / delta + 4.0_r;
56 }() * 60.0_r;
57
58 return hue < 0.0_r ?
59 360.0_r + hue :
60 hue;
61}

63std::tuple<real, real, real> cmyk_to_rgb(real cyan, real magenta, real yellow, real black) noexcept
64{
65 auto k_inv = 1.0_r - black;
66 return std::tuple{(1.0_r - cyan) * k_inv,
67 (1.0_r - magenta) * k_inv,
68 (1.0_r - yellow) * k_inv};
69}

71std::tuple<real, real, real> hsl_to_rgb(real hue, real saturation, real lightness) noexcept
72{
73 auto c = (1.0_r - math::Abs(2.0_r * lightness - 1.0_r)) * saturation;
74 auto x = c * (1.0_r - math::Abs(std::fmod(hue / 60.0_r, 2.0_r) - 1.0_r));
75 auto m = lightness - c / 2.0_r;
76
77 auto [r, g, b] = hue_to_rgb(hue, c, x);
78 return std::tuple{r + m, g + m, b + m};
79}

81std::tuple<real, real, real> hsv_to_rgb(real hue, real saturation, real value) noexcept
82{
83 auto c = value * saturation;
84 auto x = c * (1.0_r - math::Abs(std::fmod(hue / 60.0_r, 2.0_r) - 1.0_r));
85 auto m = value - c;
86
87 auto [r, g, b] = hue_to_rgb(hue, c, x);
88 return std::tuple{r + m, g + m, b + m};
89}

96std::tuple<real, real, real> ycbcr_to_rgb(real luma, real blue_difference, real red_difference) noexcept
97{
98 return std::tuple{luma + 1.402_r * (red_difference - 128.0_r),
99 luma - 0.344136_r * (blue_difference - 128.0_r) - 0.714136_r * (red_difference - 128.0_r),
100 luma + 1.772_r * (blue_difference - 128.0_r)};

104std::tuple<real, real, real, real> rgb_to_cmyk(real red, real green, real blue) noexcept
105{
106 auto k = 1.0_r - std::max({red, green, blue});
107 auto k_inv = 1.0_r - k;
108 return std::tuple{(1.0_r - red - k) / k_inv,
109 (1.0_r - green - k) / k_inv,
110 (1.0_r - blue - k) / k_inv,
111 k};
112}

114std::tuple<real, real, real> rgb_to_hsl(real red, real green, real blue) noexcept
115{
116 auto [c_min, c_max] = std::minmax({red, green, blue});
117 auto delta = c_max - c_min;
118
119 auto h = rgb_to_hue(red, green, blue, c_max, delta);
120 auto l = (c_min + c_max) / 2.0_r;
121 auto s = (delta == 0.0_r) ?
122 0.0_r :
123 delta / (1.0_r - math::Abs(2.0_r * l - 1.0_r));
124
125 return std::tuple{h, s, l};
126}

128std::tuple<real, real, real> rgb_to_hsv(real red, real green, real blue) noexcept
129{
130 auto [c_min, c_max] = std::minmax({red, green, blue});
131 auto delta = c_max - c_min;
132
133 auto h = rgb_to_hue(red, green, blue, c_max, delta);
134 auto s = (c_max == 0.0_r) ?
135 0.0_r :
136 delta / c_max;
137
138 return std::tuple{h, s, c_max};
139}

141std::tuple<real, real, real> rgb_to_hwb(real red, real green, real blue) noexcept
142{
143 auto [h, s, v] = rgb_to_hsv(red, green, blue);
144 return std::tuple{h, (h - s) * v, 1.0_r - v};
145}

147std::tuple<real, real, real> rgb_to_ycbcr(real red, real green, real blue) noexcept
148{
149 return std::tuple{0.299_r *red + 0.587_r * green + 0.114_r * blue,
150 -0.168736_r * red - 0.331264_r * green + 0.5_r * blue + 128.0_r,
151 0.5_r * red - 0.418688_r * green - 0.081312_r * blue + 128.0_r};
152}

200Color Color::HSL(real hue, real saturation, real lightness, real alpha) noexcept
201{
202 //Make sure hue is inside range [0.0, 360.0)
203 hue = std::fmod(hue, 360.0_r);
204 if (hue < 0.0_r)
205 hue = 360.0_r - hue;
206
207 saturation = math::Saturate(saturation);
208 lightness = math::Saturate(lightness);
209
210 auto [r, g, b] = detail::hsl_to_rgb(hue, saturation, lightness);
211 return {r, g, b, alpha};
212}

228Color Color::HWB(real hue, real whiteness, real blackness, real alpha) noexcept
229{
230 //Make sure hue is inside range [0.0, 360.0)
231 hue = std::fmod(hue, 360.0_r);
232 if (hue < 0.0_r)
233 hue = 360.0_r - hue;
234
235 //If sum is greater than 1.0, scale back whiteness and blackness proportionally
236 if (auto sum = whiteness + blackness;
237 sum > 1.0_r)
238 {
239 whiteness /= sum;
240 blackness /= sum;
241 }
242
243 auto [r, g, b] = detail::hwb_to_rgb(hue, whiteness, blackness);
244 return {r, g, b, alpha};
245}
