43 template <typename T, typename U>
44 inline auto subscribe_to_publisher(T &publisher, U &listener) -> T*
45 {
46 //Try to subscribe to publisher
47 if (publisher.Subscribe(listener))
48 return &publisher; //Subscribed
49
50 return nullptr;
51 }

53 template <typename T, typename U>
54 inline auto unsubscribe_from_publisher(T &publisher, U &listener) noexcept -> T*
55 {
56 //Try to unsubscribe from publisher
57 if (publisher.Unsubscribe(listener))
58 return nullptr; //Unsubscribed
59
60 return &publisher;
61 }
