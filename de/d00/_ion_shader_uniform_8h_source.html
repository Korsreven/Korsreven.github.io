123 template <typename T>
124 inline auto is_value_different(const glsl::uniform<T> &lhs, const glsl::uniform<T> &rhs) noexcept
125 {
126 auto count = lhs.Components() * lhs.Size();
127
128 for (auto i = 0; i < count; ++i)
129 {
130 if (lhs.Values()[i] != rhs.Values()[i])
131 return true;
132 }
133
134 return false;
135 }
