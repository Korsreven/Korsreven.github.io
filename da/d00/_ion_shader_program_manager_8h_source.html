89 template <typename T>
90 inline void operator()(const glsl::attribute<T> &value) const noexcept
91 {
92 if constexpr (std::is_same_v<typename glsl::attribute<T>::basic_type, int32>)
93 set_vertex_ipointer(location_, value.Components(), value.Normalized(), value.Stride(), value.Pointer(), int32{});
94 else if constexpr (std::is_same_v<typename glsl::attribute<T>::basic_type, uint32>)
95 set_vertex_ipointer(location_, value.Components(), value.Normalized(), value.Stride(), value.Pointer(), uint32{});
96 else if constexpr (std::is_same_v<typename glsl::attribute<T>::basic_type, float32>)
97 set_vertex_pointer(location_, value.Components(), value.Normalized(), value.Stride(), value.Pointer(), real{});
98 else if constexpr (std::is_same_v<typename glsl::attribute<T>::basic_type, float64>)
99 set_vertex_lpointer(location_, value.Components(), value.Normalized(), value.Stride(), value.Pointer());
100 }

103 template <typename T, int N, int M>
104 inline void operator()(const glsl::attribute<glsl::Mat<N, M, T>> &value) const noexcept
105 {
106 if constexpr (std::is_same_v<T, int32>)
107 {
108 set_vertex_ipointer(location_, M, value.Normalized(), value.Stride(), value.Pointer(), int32{});
109
110 if constexpr (N >= 1)
111 set_vertex_ipointer(location_ + 1, M, value.Normalized(), value.Stride(), value.Pointer(), int32{});
112 if constexpr (N >= 2)
113 set_vertex_ipointer(location_ + 2, M, value.Normalized(), value.Stride(), value.Pointer(), int32{});
114 if constexpr (N >= 3)
115 set_vertex_ipointer(location_ + 3, M, value.Normalized(), value.Stride(), value.Pointer(), int32{});
116 }
117 else if constexpr (std::is_same_v<T, uint32>)
118 {
119 set_vertex_ipointer(location_, M, value.Normalized(), value.Stride(), value.Pointer(), uint32{});
120
121 if constexpr (N >= 1)
122 set_vertex_ipointer(location_ + 1, M, value.Normalized(), value.Stride(), value.Pointer(), uint32{});
123 if constexpr (N >= 2)
124 set_vertex_ipointer(location_ + 2, M, value.Normalized(), value.Stride(), value.Pointer(), uint32{});
125 if constexpr (N >= 3)
126 set_vertex_ipointer(location_ + 3, M, value.Normalized(), value.Stride(), value.Pointer(), uint32{});
127 }
128 else if constexpr (std::is_same_v<T, float32>)
129 {
130 set_vertex_pointer(location_, M, value.Normalized(), value.Stride(), value.Pointer(), real{});
131
132 if constexpr (N >= 1)
133 set_vertex_pointer(location_ + 1, M, value.Normalized(), value.Stride(), value.Pointer(), real{});
134 if constexpr (N >= 2)
135 set_vertex_pointer(location_ + 2, M, value.Normalized(), value.Stride(), value.Pointer(), real{});
136 if constexpr (N >= 3)
137 set_vertex_pointer(location_ + 3, M, value.Normalized(), value.Stride(), value.Pointer(), real{});
138 }
139 else if constexpr (std::is_same_v<T, float64>)
140 {
141 set_vertex_lpointer(location_, M, value.Normalized(), value.Stride(), value.Pointer());
142
143 if constexpr (N >= 1)
144 set_vertex_lpointer(location_ + 1, M, value.Normalized(), value.Stride(), value.Pointer());
145 if constexpr (N >= 2)
146 set_vertex_lpointer(location_ + 2, M, value.Normalized(), value.Stride(), value.Pointer());
147 if constexpr (N >= 3)
148 set_vertex_lpointer(location_ + 3, M, value.Normalized(), value.Stride(), value.Pointer());
149 }
