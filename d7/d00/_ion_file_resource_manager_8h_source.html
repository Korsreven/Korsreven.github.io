33 template <typename RepositoriesT>
34 inline auto file_data_from_resource_name(const RepositoriesT &repositories, std::string_view resource_name) noexcept ->
35 std::optional<std::pair<std::string, std::filesystem::path>>
36 {
37 for (auto &repository : repositories)
38 {
39 if (auto file_data = repository.FileData(resource_name); file_data)
40 return std::pair{std::move(*file_data), std::move(*repository.FilePath(resource_name))};
41 }
42
43 return {};
44 }
