298 [[nodiscard]] inline auto operator*(const Vector3 &vector) const noexcept
299 {
300 auto [x, y, z] = vector.XYZ();
301
302 #ifdef ION_ROW_MAJOR
303 //Row-major layout (Direct3D)
304 auto inv_w = 1.0_r / (m_[0][3] * x + m_[1][3] * y + m_[2][3], + m_[3][3]);
305 return Vector3{(m_[0][0] * x + m_[1][0] * y + m_[2][0] * z + m_[3][0]) * inv_w,
306 (m_[0][1] * x + m_[1][1] * y + m_[2][1] * z + m_[3][1]) * inv_w,
307 (m_[0][2] * x + m_[1][2] * y + m_[2][2] * z + m_[3][2]) * inv_w};
308 #else
309 //Column-major layout (OpenGL)
310 auto inv_w = 1.0_r / (m_[3][0] * x + m_[3][1] * y + m_[3][2], + m_[3][3]);
311 return Vector3{(m_[0][0] * x + m_[0][1] * y + m_[0][2] * z + m_[0][3]) * inv_w,
312 (m_[1][0] * x + m_[1][1] * y + m_[1][2] * z + m_[1][3]) * inv_w,
313 (m_[2][0] * x + m_[2][1] * y + m_[2][2] * z + m_[2][3]) * inv_w};
314 #endif
315 }
