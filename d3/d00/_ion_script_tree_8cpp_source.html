37void serialize_argument(const ArgumentNode &argument, std::vector<std::byte> &bytes)
38{
39 argument.Visit(
40 [&](auto &&value)
41 {
42 return serialize_argument(value, bytes);
43 });
44 serialize_value(argument.Unit(), bytes);

47void serialize_property(const PropertyNode &property, std::vector<std::byte> &bytes)
48{
49 for (auto &argument : property.Arguments())
50 serialize_argument(argument, bytes);
51
52 bytes.push_back(static_cast<std::byte>(serialization_section_token::Property));
53 serialize_value(property.Name(), bytes);
54}

56void serialize_object(const TreeNode &node, std::vector<std::byte> &bytes)
57{
58 for (auto &property : node.Object.Properties())
59 serialize_property(property, bytes);
60
61 bytes.push_back(static_cast<std::byte>(serialization_section_token::Object));
62 serialize_value(node.Depth, bytes);
63 serialize_value(node.Object.Name(), bytes);
64 serialize_value(node.Object.Classes(), bytes);
65}

78int deserialize_argument(std::string_view bytes, ArgumentNodes &arguments)
79{
80 auto bytes_deserialized = 0;
81
82 if (!std::empty(bytes))
83 {
84 switch (static_cast<size_t>(bytes.front()))
85 {
86 case variant_index<ArgumentType, ScriptType::Boolean>():
87 {
88 bytes_deserialized = deserialize_argument<ScriptType::Boolean>(bytes.substr(1), arguments);
89 break;
90 }
91
92 case variant_index<ArgumentType, ScriptType::Color>():
93 {
94 bytes_deserialized = deserialize_argument<ScriptType::Color>(bytes.substr(1), arguments);
95 break;
96 }
97
98 case variant_index<ArgumentType, ScriptType::Enumerable>():
99 {
100 bytes_deserialized = deserialize_argument<ScriptType::Enumerable>(bytes.substr(1), arguments);
101 break;
102 }
103
104 case variant_index<ArgumentType, ScriptType::FloatingPoint>():
105 {
106 bytes_deserialized = deserialize_argument<ScriptType::FloatingPoint>(bytes.substr(1), arguments);
107 break;
108 }
109
110 case variant_index<ArgumentType, ScriptType::Integer>():
111 {
112 bytes_deserialized = deserialize_argument<ScriptType::Integer>(bytes.substr(1), arguments);
113 break;
114 }
115
116 case variant_index<ArgumentType, ScriptType::String>():
117 {
118 bytes_deserialized = deserialize_argument<ScriptType::String>(bytes.substr(1), arguments);
119 break;
120 }
121
122 case variant_index<ArgumentType, ScriptType::Vector2>():
123 {
124 bytes_deserialized = deserialize_argument<ScriptType::Vector2>(bytes.substr(1), arguments);
125 break;
126 }
127
128 case variant_index<ArgumentType, ScriptType::Vector3>():
129 {
130 bytes_deserialized = deserialize_argument<ScriptType::Vector3>(bytes.substr(1), arguments);
131 break;
132 }
133 }
134 }
135
136 if (bytes_deserialized > 0)
137 ++bytes_deserialized;
138
139 return bytes_deserialized;
140}

153int deserialize_object(std::string_view bytes, std::vector<ObjectNodes> &object_stack, PropertyNodes &properties)
154{
155 int depth;
156 auto depth_bytes_deserialized = deserialize_value(bytes, depth);
157
158 std::string name;
159 auto name_bytes_deserialized = depth_bytes_deserialized > 0 ?
160 deserialize_value(bytes.substr(depth_bytes_deserialized), name) : 0;
161
162 std::string classes;
163 auto classes_bytes_deserialized = name_bytes_deserialized > 0 ?
164 deserialize_value(bytes.substr(depth_bytes_deserialized + name_bytes_deserialized), classes) : 0;
165
166 //Node depth, object name and classes serialized
167 if (classes_bytes_deserialized > 0)
168 {
169 if (depth >= std::ssize(object_stack))
170 object_stack.insert(std::end(object_stack), depth - std::size(object_stack) + 1, {});
171
172 //Leaf object, no childrens
173 if (auto &object = object_stack[depth];
174 &object == &object_stack.back())
175 object.emplace_back(std::move(name), std::move(classes), std::move(properties));
176 //Take childrens (if any) from one depth deeper
177 else
178 object.emplace_back(std::move(name), std::move(classes), std::move(properties),
179 std::move(object_stack[depth + 1]));
180
181 return depth_bytes_deserialized + name_bytes_deserialized + classes_bytes_deserialized;
182 }
183
184 return 0;
185}

187std::optional<ObjectNodes> deserialize(std::string_view bytes)
188{
189 std::vector<ObjectNodes> object_stack;
190 PropertyNodes properties;
191 ArgumentNodes arguments;
192
193 for (auto iter = std::begin(bytes), end = std::end(bytes); iter != end;)
194 {
195 auto c = *iter;
196 auto off = iter - std::begin(bytes);
197
198 auto bytes_deserialized = 0;
199
200 switch (static_cast<serialization_section_token>(c))
201 {
202 case serialization_section_token::Object:
203 {
204 //An object requires that any previous arguments, should have been processed by a property first
205 if (std::empty(arguments))
206 bytes_deserialized = deserialize_object(bytes.substr(off + 1), object_stack, properties);
207
208 break;
209 }
210
211 case serialization_section_token::Property:
212 {
213 //A property requires one or more previous arguments
214 if (!std::empty(arguments))
215 bytes_deserialized = deserialize_property(bytes.substr(off + 1), properties, arguments);
216
217 break;
218 }
219
220 case serialization_section_token::Argument:
221 {
222 bytes_deserialized = deserialize_argument(bytes.substr(off + 1), arguments);
223 break;
224 }
225 }
226
227 if (bytes_deserialized > 0)
228 iter += bytes_deserialized + 1;
229 else
230 {
231 object_stack.clear();
232 break;
233 }
234 }
235
236 return !std::empty(object_stack) ?
237 //Top-level objects
238 std::make_optional<ObjectNodes>(std::move(object_stack.front())) :
239 std::nullopt;
240}

247std::string print(const ObjectNodes &objects, PrintOptions print_options)
248{
249 std::string output;
250
251 //Print objects
252 for (auto &[object, parent, depth] : depth_first_search(objects, DepthFirstTraversal::PreOrder))
253 {
254 output += "\n" + std::string(depth * 4, ' ') + "+-- " + object.Name();
255
256 if (auto &classes = object.Classes(); !std::empty(classes))
257 output += " \"" + classes + "\"";
258
259 if (print_options == PrintOptions::Properties ||
260 print_options == PrintOptions::Arguments)
261 {
262 //Print properties
263 for (auto &property : object.Properties())
264 {
265 output += "\n" + std::string(depth * 4 + 4, ' ') + "|-- " + property.Name();
266
267 if (print_options == PrintOptions::Arguments)
268 {
269 if (auto remaining_args = std::ssize(property.Arguments());
270 remaining_args > 0)
271 {
272 output += ": ";
273
274 //Print arguments
275 for (auto &argument : property.Arguments())
276 {
277 output += argument.Visit(
278 [](const ScriptType::Boolean &value)
279 {
280 return value.Get() ? "true"s : "false"s;
281 },
282 [](const ScriptType::Color &value)
283 {
284 auto name = std::string{utilities::parse::AsString(value.Get()).value_or("")};
285 auto [r, g, b] = value.Get().ToRGB();
286 auto a = value.Get().A();
287
288 if (a < 1.0_r)
289 return ion::utilities::string::Format(name + "({0}, {1}, {2}, {3:0.##})", r, g, b, a);
290 else
291 return ion::utilities::string::Format(name + "({0}, {1}, {2})", r, g, b);
292 },
293 [](const ScriptType::Enumerable &value)
294 {
295 return value.Get();
296 },
297 [](const ScriptType::String &value)
298 {
299 return ion::utilities::string::Concat('"', value.Get(), '"');
300 },
301 [](const ScriptType::Vector2 &value)
302 {
303 auto [x, y] = value.Get().XY();
304 return ion::utilities::string::Concat('{', x, ", ", y, '}');
305 },
306 [](const ScriptType::Vector3 &value)
307 {
308 auto [x, y, z] = value.Get().XYZ();
309 return ion::utilities::string::Concat('{', x, ", ", y, ", ", z, '}');
310 },
311 //Default
312 [&](auto &&value)
313 {
314 return ion::utilities::convert::ToString(value.Get()) + argument.Unit();
315 });
316
317 if (--remaining_args > 0)
318 output += " ";
319 }
320 }
321 }
322 }
323 }
324 }
325
326 return output;
327}

334void breadth_first_search_impl(search_result &result, size_t off)
335{
336 auto last = std::size(result);
337
338 if (last - off > 0)
339 {
340 for (; off < last; ++off)
341 {
342 for (auto &object : result[off].Object.Objects())
343 result.emplace_back(object, result[off].Object, result[off].Depth + 1);
344 }
345
346 breadth_first_search_impl(result, last);
347 }
348}

350void depth_first_search_post_order_impl(search_result &result, const TreeNode &node)
351{
352 for (auto &object : node.Object.Objects())
353 depth_first_search_post_order_impl(result, {object, node.Object, node.Depth + 1});
354
355 result.push_back(node);
356}

358void depth_first_search_pre_order_impl(search_result &result, const TreeNode &node)
359{
360 result.push_back(node);
361
362 for (auto &object : node.Object.Objects())
363 depth_first_search_pre_order_impl(result, {object, node.Object, node.Depth + 1});
364}

366void fully_qualified_name_impl(std::string &name, const ObjectNode &parent, const ObjectNode &what_object)
367{
368 if (&parent == &what_object)
369 {
370 name = what_object.Name();
371 return;
372 }
373
374 for (auto &object : parent.Objects())
375 {
376 fully_qualified_name_impl(name, object, what_object);
377
378 if (!std::empty(name))
379 {
380 name = parent.Name() + "." + name;
381 break;
382 }
383 }

387search_result breadth_first_search(const ObjectNodes &objects)
388{
389 search_result result;
390
391 for (auto &object : objects)
392 result.emplace_back(object);
393
394 breadth_first_search_impl(result, 0);
395 return result;

398search_result depth_first_search(const ObjectNodes &objects, DepthFirstTraversal traversal)
399{
400 search_result result;
401
402 //Post-order
403 if (traversal == DepthFirstTraversal::PostOrder)
404 {
405 for (auto &object : objects)
406 depth_first_search_post_order_impl(result, {object});
407 }
408 //Pre-order
409 else
410 {
411 for (auto &object : objects)
412 depth_first_search_pre_order_impl(result, {object});
413 }
414
415 return result;
416}

434void lineage_depth_first_search_impl(lineage_search_result &result, generations &descendants, ObjectNode &object)
435{
436 result.push_back(descendants);
437
438 if (!std::empty(object.Objects()))
439 {
440 descendants.emplace_back(); //Add next generation
441
442 for (auto &sibling : object.Objects())
443 {
444 descendants.back().push_back(&sibling);
445 lineage_depth_first_search_impl(result, descendants, sibling);
446 }
447
448 descendants.pop_back();
449 }
450}
451
452lineage_search_result lineage_depth_first_search(ObjectNodes &objects)
453{
454 lineage_search_result result;
455 generations descendants;
456 descendants.emplace_back(); //Add next generation
457
458 for (auto &sibling : objects)
459 {
460 descendants.back().push_back(&sibling);
461 lineage_depth_first_search_impl(result, descendants, sibling);
462 }
463
464 return result;
465}

807{

811 {

824 std::nullopt;
