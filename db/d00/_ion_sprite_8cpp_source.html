30mesh::Vertices sprite_vertices(const Vector3 &position, real rotation, const Vector2 &size, const Color &color,
31 const Vector2 &lower_left_tex_coord, const Vector2 &upper_right_tex_coord)
32{
33 auto [half_width, half_height] = (size * 0.5_r).XY();
34
35 auto v1 = (position + Vector2{-half_width, half_height}).RotateCopy(rotation, position);
36 auto v2 = (position + Vector2{-half_width, -half_height}).RotateCopy(rotation, position);
37 auto v3 = (position + Vector2{half_width, -half_height}).RotateCopy(rotation, position);
38 auto v4 = (position + Vector2{half_width, half_height}).RotateCopy(rotation, position);
39
40 auto [ll_s, ll_t] = lower_left_tex_coord.XY();
41 auto [ur_s, ur_t] = upper_right_tex_coord.XY();
42
43 return {{v1, vector3::UnitZ, {ll_s, ur_t}, color},
44 {v2, vector3::UnitZ, {ll_s, ll_t}, color},
45 {v3, vector3::UnitZ, {ur_s, ll_t}, color},
46 {v3, vector3::UnitZ, {ur_s, ll_t}, color},
47 {v4, vector3::UnitZ, {ur_s, ur_t}, color},
48 {v1, vector3::UnitZ, {ll_s, ur_t}, color}};
49}

52std::optional<Vector2> get_texture_size(materials::Material &material) noexcept
53{
54 auto texture =
55 [&]() noexcept
56 {
57 if (auto [animation, texture] = material.DiffuseMap(); animation)
58 return animation->UnderlyingFrameSequence() ?
59 animation->UnderlyingFrameSequence()->FirstFrame() : nullptr;
60 else
61 return texture;
62 }();
63
64 if (texture)
65 {
66 if (texture->IsLoaded() || (texture->Owner() && texture->Owner()->Load(*texture)))
67 return Vector2{static_cast<real>(texture->Extents()->Width), static_cast<real>(texture->Extents()->Height)};
68 }
69
70 return {};
71}

85std::optional<Vector2> Sprite::GetTextureSize() const noexcept
86{
87 if (auto material = SurfaceMaterial(); material)
88 {
89 if (auto texture_size = detail::get_texture_size(*material); texture_size)
90 {
91 auto ppu = Engine::PixelsPerUnit();
92 return *texture_size /= ppu;
93 }
94 }
95
96 return {};
97}

105void Sprite::RecalculateTexCoords() noexcept
106{
107 if (auto texture_size = GetTextureSize(); texture_size)
108 {
109 auto [lower_left, upper_right] =
110 materials::material::detail::get_unflipped_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_);
111
112 lower_left = vector2::Zero;
113 upper_right = size_ / *texture_size;
114
115 auto [new_lower_left, new_upper_right] =
116 materials::material::detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_,
117 lower_left, upper_right);
118 lower_left_tex_coord_ = new_lower_left;
119 upper_right_tex_coord_ = new_upper_right;
120
121 update_vertices_ = true;
122 }
123}

197void Sprite::Crop(const std::optional<Aabb> &area) noexcept
198{
199 //Crop by area
200 if (area)
201 {
202 auto min = area->Min().CeilCopy(vector2::Zero).FloorCopy(vector2::UnitScale);
203 auto max = area->Max().CeilCopy(vector2::Zero).FloorCopy(vector2::UnitScale);
204
205 if (min != max)
206 {
207 auto [lower_left, upper_right] =
208 materials::material::detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_, min, max);
209
210 lower_left_tex_coord_ = lower_left;
211 upper_right_tex_coord_ = upper_right;
212
213 auto_repeat_ = false;
214 update_vertices_ = true;
215 }
216 }
217 //Un-crop
218 else if (IsCropped())
219 {
220 auto [lower_left, upper_right] =
221 materials::material::detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_, 0.0_r, 1.0_r);
222
223 lower_left_tex_coord_ = lower_left;
224 upper_right_tex_coord_ = upper_right;
225
226 auto_repeat_ = false;
227 update_vertices_ = true;
228 }
229}

231void Sprite::Repeat(const std::optional<Vector2> &amount) noexcept
232{
233 //Repeat by amount
234 if (amount)
235 {
236 if (auto max = amount->CeilCopy(vector2::Zero); vector2::Zero < max)
237 {
238 auto [lower_left, upper_right] =
239 materials::material::detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_, 0.0_r, max);
240
241 lower_left_tex_coord_ = lower_left;
242 upper_right_tex_coord_ = upper_right;
243
244 auto_repeat_ = false;
245 update_vertices_ = true;
246 }
247 }
248 //Un-repeat
249 else if (IsRepeated())
250 {
251 auto [lower_left, upper_right] =
252 materials::material::detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_, 0.0_r, 1.0_r);
253
254 lower_left_tex_coord_ = lower_left;
255 upper_right_tex_coord_ = upper_right;
256
257 auto_repeat_ = false;
258 update_vertices_ = true;
259 }
