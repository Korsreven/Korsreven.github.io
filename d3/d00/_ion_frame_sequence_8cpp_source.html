29container_type get_frames_from_first_frame(NonOwningPtr<Texture> first_frame, int total_frames)
30{
31 using namespace ion::utilities;
32
33 if (!first_frame->Owner())
34 return {};
35
36 container_type frames;
37 frames.reserve(total_frames);
38 frames.push_back(first_frame);
39
40 //Multiple frames
41 if (total_frames > 1)
42 {
43 auto &name = *first_frame->Name();
44
45 if (auto iter = std::find_if_not(std::rbegin(name), std::rend(name),
46 [](auto c) noexcept
47 {
48 return convert::detail::is_digit(c);
49 }).base(); iter != std::end(name))
50 {
51 auto off = iter - std::begin(name);
52 auto count = std::ssize(name) - off;
53
54 if (auto number = convert::To<int>({std::data(name) + off, static_cast<size_t>(count)}); number)
55 {
56 auto name_prefix = name.substr(0, off);
57 std::string next_name;
58
59 for (auto i = 1; i < total_frames; ++i)
60 {
61 auto digits = math::Log10(++*number) + 1;
62 next_name = name_prefix;
63
64 //Add leading zeroes
65 if (count - digits > 0)
66 next_name.append(count - digits, '0');
67
68 next_name += convert::ToString(*number);
69
70 if (auto next_frame = first_frame->Owner()->GetTexture(next_name); next_frame)
71 frames.push_back(next_frame);
72 else
73 return {};
74 }
75 }
76 }
77 }
78
79 return frames;
80}
