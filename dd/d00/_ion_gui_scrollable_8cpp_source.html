34void GuiScrollable::UpdateScrollBar() noexcept
35{
36 if (scroll_bar_)
37 {
38 auto element_count = TotalElements();
39 auto elements_in_view = ElementsInView();
40 auto max =
41 elements_in_view > 0 && element_count > elements_in_view ?
42 element_count - elements_in_view : 0;
43
44 scroll_bar_->Range(0, max);
45 scroll_bar_->Position(ScrollPosition());
46 scroll_bar_->StepByAmount(scroll_rate_);
47 }
48}

63void GuiScrollable::AttachedScrollBar(NonOwningPtr<GuiScrollBar> scroll_bar) noexcept
64{
65 if (scroll_bar_ != scroll_bar)
66 {
67 //Attach
68 if (scroll_bar)
69 {
70 AttachedScrollBar(nullptr); //Detach previous (if any)
71 scroll_bar_ = scroll_bar;
72 UpdateScrollBar();
73
74 if (auto owner = Owner(); owner)
75 scroll_bar->AttachedScrollable(
76 owner->GetControlAs<GuiScrollable>(*Name())
77 );
78 }
79
80 else //Detach
81 {
82 auto ptr = scroll_bar_.get();
83 scroll_bar_ = nullptr;
84
85 if (ptr)
86 ptr->AttachedScrollable(nullptr);
87 }
88 }
89}

112bool GuiScrollable::KeyReleased(KeyButton button) noexcept
113{
114 switch (button)
115 {
116 case KeyButton::PageUp:
117 Scrolled(-ElementsInView());
118 return true;
119
120 case KeyButton::PageDown:
121 Scrolled(ElementsInView());
122 return true;
123
124 default:
125 return false;
126 }
127}
