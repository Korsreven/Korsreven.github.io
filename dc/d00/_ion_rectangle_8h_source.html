35 inline auto scale_to_fill(const Vector2 &size, const Vector2 &fill_size)
36 {
37 auto [x, y] = size.XY();
38 auto [dst_x, dst_y] = fill_size.XY();
39 return size * std::max(dst_x / x, dst_y / y);
40 }
41
42 inline auto scale_to_fit(const Vector2 &size, const Vector2 &fit_size)
43 {
44 auto [x, y] = size.XY();
45 auto [dst_x, dst_y] = fit_size.XY();
46 return size * std::min(dst_x / x, dst_y / y);
47 }
