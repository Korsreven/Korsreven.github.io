77 constexpr auto is_power_of_two(uint32 x) noexcept
78 {
79 return x && !(x & (x - 1));
80 }
81
82 constexpr auto lower_power_of_two(uint32 x) noexcept
83 {
84 x |= x >> 1;
85 x |= x >> 2;
86 x |= x >> 4;
87 x |= x >> 8;
88 x |= x >> 16;
89 return x - (x >> 1);
90 }
91
92 constexpr auto upper_power_of_two(uint32 x) noexcept
93 {
94 --x;
95 x |= x >> 1;
96 x |= x >> 2;
97 x |= x >> 4;
98 x |= x >> 8;
99 x |= x >> 16;
100 ++x;
101 return x;
102 }
103
104 constexpr auto nearest_power_of_two(uint32 x) noexcept
105 {
106 auto lower = lower_power_of_two(x);
107 auto upper = upper_power_of_two(x);
108 return x - lower < upper - x ? lower : upper;
109 }

111 constexpr auto make_power_of_two(uint32 x, NpotSampling npot_sampling) noexcept
112 {
113 switch (npot_sampling)
114 {
115 case NpotSampling::Up:
116 return upper_power_of_two(x);
117
118 case NpotSampling::Down:
119 return lower_power_of_two(x);
120
121 case NpotSampling::Nearest:
122 default:
123 return nearest_power_of_two(x);
124 }

183 template <typename... Args>
184 void CreateSubTextures(NonOwningPtr<TextureAtlas> texture_atlas, Args &&...args)
185 {
186 using namespace ion::utilities;
187
188 if (texture_atlas)
189 {
190 auto &name = *texture_atlas->Name();
191
192 texture::TextureAtlasRegion atlas_region;
193 atlas_region.Atlas = texture_atlas;
194 atlas_region.Position = {1, 1};
195
196 for (auto i = 0; i < texture_atlas->SubTextures(); ++i)
197 {
198 CreateResource(string::Concat(name, "_", i + 1), texture_atlas->AssetName(), atlas_region,
199 std::forward<Args>(args)...);
200 texture_manager::detail::next_sub_texture_position(atlas_region.Position,
201 texture_atlas->Rows(), texture_atlas->Columns(), texture_atlas->SubTextureOrder());
202 }
203 }
204 }
