74 inline auto perspective_to_frustum(real fov, real aspect_ratio, real z_near, real z_far) noexcept
75 {
76 using namespace ion::utilities;
77 auto height = math::Tan(fov / 360.0_r * math::Pi) * z_near;
78 auto width = height * aspect_ratio;
79 return std::tuple{-width, width, -height, height, z_near, z_far};
80 }
