96 template <typename Iterator>
97 /*constexpr*/ auto median(Iterator first, Iterator last) noexcept
98 {
99 auto count = std::distance(first, last);
100 auto middle = first + count / 2;
101
102 std::nth_element(first, middle, last);
103 auto middle_value = *middle;
104
105 //Even
106 if (count % 2 == 0)
107 {
108 std::nth_element(first, middle - 1, last);
109 return (*(middle - 1) + middle_value) / static_cast<typename Iterator::value_type>(2);
110 }
111 //Odd
112 else
113 return middle_value;
114 }

116 template <typename Iterator>
117 /*constexpr*/ auto mode(Iterator first, Iterator last)
118 {
119 adaptors::FlatMap<typename Iterator::value_type, int> histogram;
120 std::for_each(first, last,
121 [&](auto &&value) mutable noexcept
122 {
123 ++histogram[value]; //Create histogram
124 });
125
126 std::vector dataset(std::begin(histogram), std::end(histogram));
127 std::stable_sort(std::begin(dataset), std::end(dataset),
128 [](auto &&x, auto &&y) noexcept
129 {
130 return x.second > y.second; //Sort descending
131 });
132
133 auto last_in_set =
134 std::find_if_not(std::begin(dataset) + 1, std::end(dataset),
135 [&](auto &&value) noexcept
136 {
137 return value.second == dataset.front().second; //Find all modes (multimodal)
138 });
139
140 std::vector<typename Iterator::value_type> result;
141 result.reserve(last_in_set - std::begin(dataset));
142 std::for_each(std::begin(dataset), last_in_set,
143 [&](auto &&value) mutable noexcept
144 {
145 result.push_back(value.first);
146 });
147
148 return result;
149 }

258 template <typename T>
259 [[nodiscard]] constexpr auto Power(T x, T y) noexcept
260 {
261 static_assert(std::is_integral_v<T>);
262 auto result = T{1};
263
264 while (y)
265 {
266 if (y & T{1})
267 result *= x;
268
269 y >>= T{1};
270 x *= x;
271 }
272
273 return result;

354 template <typename T>
355 [[nodiscard]] constexpr auto Choose(T n, T r) noexcept
356 {
357 static_assert(std::is_integral_v<T>);
358
