45void set_sprite_properties(const SkinPart &part, graphics::scene::shapes::Sprite &sprite) noexcept
46{
47 sprite.Size(sprite.Size() * part.Scaling);
48 sprite.FillColor(part.FillColor);
49
50 if (part.FlipHorizontal)
51 sprite.FlipHorizontal();
52 if (part.FlipVertical)
53 sprite.FlipVertical();
54}
