47 inline auto as_object_requirement(events::event_channel::SubscriptionContract contract) noexcept
48 {
49 switch (contract)
50 {
51 case events::event_channel::SubscriptionContract::NonCancelable:
52 return ObjectRequirement::Mandatory;
53
54 case events::event_channel::SubscriptionContract::Cancelable:
55 default:
56 return ObjectRequirement::Optional;
57 }
58 }

60 inline auto as_subscription_contract(ObjectRequirement requirement) noexcept
61 {
62 switch (requirement)
63 {
64 case ObjectRequirement::Mandatory:
65 return events::event_channel::SubscriptionContract::NonCancelable;
66
67 case ObjectRequirement::Optional:
68 default:
69 return events::event_channel::SubscriptionContract::Cancelable;
70 }
71 }

100 void ObjectRemoved(T &object) noexcept override
101 {
102 if (managed_object_ == &object)
103 {
104 //Execute callback before object is set to nullptr
105 //The callback owner can then know in advance which object is going to be removed
106 if (on_removed_)
107 (*on_removed_)(object);
108
109 managed_object_ = nullptr;
110 this->DoUnsubscribe(true);
111 }
112 }
113
115 bool ObjectRemovable(T&) noexcept override
116 {
117 return this->Contract() == events::event_channel::SubscriptionContract::Cancelable;
118 }

121 void Unsubscribed() noexcept override
122 {
123 //Execute callback before object is set to nullptr
124 //The callback owner can then know in advance which object is going to be removed
125 if (on_removed_ && managed_object_)
126 (*on_removed_)(*managed_object_);
127
128 managed_object_ = nullptr;
129 }

308 inline auto Observe(T &object)
309 {
310 if (managed_object_ != &object)
311 {
312 if (auto owner = object.Owner(); owner && this->Subscribe(*owner))
313 managed_object_ = &object;
314 }
315
316 return !!managed_object_;
317 }
