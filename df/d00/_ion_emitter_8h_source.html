67 inline auto box_coordinate(real coord, real rand, real half_size, real half_inner_size1, real half_inner_size2) noexcept
68 {
69 return coord > -half_inner_size1 && coord < half_inner_size1 ?
70 (half_size - half_inner_size2) * rand + (rand < 0.0_r ? -half_inner_size2 : half_inner_size2) :
71 half_size * rand;
72 }
73
74 inline auto position_inside_box(const Vector2 &half_size, const Vector2 &half_inner_size) noexcept
75 {
76 using namespace ion::utilities;
77
78 if (random::Number(0, 1))
79 {
80 auto x = half_size.X() * random::Number(-1.0_r, 1.0_r);
81 return Vector2{x, box_coordinate(x, random::Number(-1.0_r, 1.0_r), half_size.Y(), half_inner_size.X(), half_inner_size.Y())};
82 }
83 else
84 {
85 auto y = half_size.Y() * random::Number(-1.0_r, 1.0_r);
86 return Vector2{box_coordinate(y, random::Number(-1.0_r, 1.0_r), half_size.X(), half_inner_size.Y(), half_inner_size.X()), y};
87 }
88 }
89
90 inline auto position_inside_ring(const Vector2 &half_size, const Vector2 &half_inner_size) noexcept
91 {
92 using namespace ion::utilities;
93
94 auto theta = math::TwoPi * random::Number();
95 auto radius = (half_size - half_inner_size) * random::Number() + half_inner_size;
96 return Vector2{radius.X() * math::Cos(theta),
97 radius.Y() * math::Sin(theta)};
98 }

101 inline auto particle_position(EmitterType emitter_type, const Vector2 &half_size, const Vector2 &half_inner_size) noexcept
102 {
103 switch (emitter_type)
104 {
105 case EmitterType::Box:
106 return position_inside_box(half_size, half_inner_size);
107
108 case EmitterType::Ring:
109 return position_inside_ring(half_size, half_inner_size);
110
111 default: //Point
112 return utilities::vector2::Zero;
113 }
