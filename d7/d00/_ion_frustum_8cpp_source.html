27Aabb keep_proportion(const Aabb &clip_plane, real aspect_ratio, AspectRatioFormat aspect_format,
28 real base_viewport_height, const Vector2 &viewport_size) noexcept
29{
30 auto height = clip_plane.ToSize().Y();
31 auto width = height * aspect_ratio;
32
33 auto [viewport_width, viewport_height] = viewport_size.XY();
34 auto viewport_aspect_ratio = viewport_width / viewport_height;
35
36 switch (aspect_format)
37 {
38 case AspectRatioFormat::PanAndScan:
39 {
40 //Crop left/right
41 if (aspect_ratio > viewport_aspect_ratio)
42 width *= (viewport_width / (viewport_height * aspect_ratio));
43 //Crop top/bottom
44 else if (aspect_ratio < viewport_aspect_ratio)
45 height *= viewport_height / (viewport_width / aspect_ratio);
46
47 break;
48 }
49
50 case AspectRatioFormat::Letterbox:
51 {
52 //Letterbox
53 if (aspect_ratio > viewport_aspect_ratio)
54 height *= viewport_height / (viewport_width / aspect_ratio);
55 //Pillarbox
56 else if (aspect_ratio < viewport_aspect_ratio)
57 width *= (viewport_width / (viewport_height * aspect_ratio));
58
59 break;
60 }
61
62 case AspectRatioFormat::Windowbox:
63 {
64 auto base_viewport_width = base_viewport_height * aspect_ratio;
65 width *= (viewport_width / base_viewport_width);
66 height *= (viewport_height / base_viewport_height);
67 break;
68 }
69 }
70
71 return Aabb::Size({width , height}, clip_plane.Center());
72}

74std::tuple<real, real, real, real, real, real> to_ortho(const std::optional<Aabb> &clip_plane, real z_near, real z_far,
75 const std::optional<real> &aspect_ratio, AspectRatioFormat aspect_format, real base_viewport_height, const Vector2 &viewport_size) noexcept
76{
77 auto plane = clip_plane ?
78 *clip_plane :
79 Aabb{graphics::utilities::vector2::Zero, viewport_size};
80
81 if (aspect_ratio)
82 plane = keep_proportion(plane, *aspect_ratio, aspect_format, base_viewport_height, viewport_size);
83
84 auto [left, bottom] = plane.Min().XY();
85 auto [right, top] = plane.Max().XY();
86
87 return std::tuple{left, right, bottom, top, z_near, z_far};
88}

108Matrix4 get_projection_matrix(ProjectionType projection, const std::optional<Aabb> &clip_plane, real near_clip_distance, real far_clip_distance, real fov,
109 const std::optional<real> &aspect_ratio, AspectRatioFormat aspect_format, real base_viewport_height, const Vector2 &viewport_size)
110{
111 switch (projection)
112 {
113 case ProjectionType::Perspective:
114 {
115 auto [left, right, bottom, top, z_near, z_far] =
116 to_frustum(clip_plane, near_clip_distance, far_clip_distance, fov,
117 aspect_ratio, aspect_format, base_viewport_height, viewport_size);
118
119 #ifdef ION_ROW_MAJOR
120 return {2.0_r * z_near / (right - left), 0.0_r, 0.0_r, 0.0_r,
121 0.0_r, 2.0_r * z_near / (top - bottom), 0.0_r, 0.0_r,
122 (right + left) / (right - left), (top + bottom) / (top - bottom), -(z_far + z_near) / (z_far - z_near), -1.0,
123 0.0_r, 0.0_r, -(2 * z_far * z_near) / (z_far - z_near), 0.0_r};
124 #else
125 return {2.0_r * z_near / (right - left), 0.0_r, (right + left) / (right - left), 0.0_r,
126 0.0_r, 2.0_r * z_near / (top - bottom), (top + bottom) / (top - bottom), 0.0_r,
127 0.0_r, 0.0_r, -(z_far + z_near) / (z_far - z_near), -(2 * z_far * z_near) / (z_far - z_near),
128 0.0_r, 0.0_r, -1.0_r, 0.0_r};
129 #endif
130 }
131
132 case ProjectionType::Orthographic:
133 default:
134 {
135 auto [left, right, bottom, top, z_near, z_far] =
136 to_ortho(clip_plane, near_clip_distance, far_clip_distance,
137 aspect_ratio, aspect_format, base_viewport_height, viewport_size);
138
139 #ifdef ION_ROW_MAJOR
140 return {2.0_r / (right - left), 0.0_r, 0.0_r, 0.0_r,
141 0.0_r, 2.0_r / (top - bottom), 0.0_r, 0.0_r,
142 0.0_r, 0.0_r, -2.0_r / (z_far - z_near), 0.0_r,
143 -(right + left) / (right - left), -(top + bottom) / (top - bottom), -(z_far + z_near) / (z_far - z_near), 1.0_r};
144 #else
145 return {2.0_r / (right - left), 0.0_r, 0.0_r, -(right + left) / (right - left),
146 0.0_r, 2.0_r / (top - bottom), 0.0_r, -(top + bottom) / (top - bottom),
147 0.0_r, 0.0_r, -2.0_r / (z_far - z_near), -(z_far + z_near) / (z_far - z_near),
148 0.0_r, 0.0_r, 0.0_r, 1.0_r};
149 #endif
150 }
151 }

154void project_through_frustum(ProjectionType projection, const std::optional<Aabb> &clip_plane, real near_clip_distance, real far_clip_distance, real fov,
155 const std::optional<real> &aspect_ratio, AspectRatioFormat aspect_format, real base_viewport_height, const Vector2 &viewport_size) noexcept
156{
157 glMatrixMode(GL_PROJECTION);
158 glLoadIdentity();
159
160 switch (projection)
161 {
162 case ProjectionType::Perspective:
163 {
164 auto [left, right, bottom, top, z_near, z_far] =
165 to_frustum(clip_plane, near_clip_distance, far_clip_distance, fov,
166 aspect_ratio, aspect_format, base_viewport_height, viewport_size);
167 glFrustum(left, right, bottom, top, z_near, z_far);
168 break;
169 }
170
171 case ProjectionType::Orthographic:
172 default:
173 {
174 auto [left, right, bottom, top, z_near, z_far] =
175 to_ortho(clip_plane, near_clip_distance, far_clip_distance,
176 aspect_ratio, aspect_format, base_viewport_height, viewport_size);
177 glOrtho(left, right, bottom, top, z_near, z_far);
178 break;
179 }
180 }
181
182 glMatrixMode(GL_MODELVIEW); //Switch back
183 glLoadIdentity();
184}
