63 inline auto is_cropped(const Vector2 &lower_left, const Vector2 &upper_right) noexcept
64 {
65 auto [ll_s, ll_t] = lower_left.XY();
66 auto [ur_s, ur_t] = upper_right.XY();
67
68 return ll_s > 0.0_r || ll_t > 0.0_r ||
69 ur_s < 1.0_r || ur_t < 1.0_r;
70 }
71
72 inline auto is_repeated(const Vector2 &lower_left, const Vector2 &upper_right) noexcept
73 {
74 auto [ll_s, ll_t] = lower_left.XY();
75 auto [ur_s, ur_t] = upper_right.XY();
76
77 return ll_s < 0.0_r || ll_t < 0.0_r ||
78 ur_s > 1.0_r || ur_t > 1.0_r;
79 }
