56std::pair<bool, real> Ray::Intersects(const Aabb &aabb) const noexcept
57{
58 if (aabb.Empty())
59 return {false, 0.0_r};
60
61 auto [x, y] = origin_.XY();
62 auto [min_x, min_y] = aabb.Min().XY();
63 auto [max_x, max_y] = aabb.Max().XY();
64
65 //Ray origin is already inside aabb
66 if (x > min_x && y > min_y &&
67 x < max_x && y < max_y)
68
69 return {true, 0.0_r};
70
71 auto [dir_x, dir_y] = direction_.XY();
72 auto hit = false;
73 auto min_units = 0.0_r;
74
75 //Min x
76 if (x <= min_x && dir_x > 0.0_r)
77 {
78 if (auto units = (min_x - x) / dir_x; units >= 0.0)
79 {
80 auto hit_point = origin_ + direction_ * units;
81 auto [hit_x, hit_y] = hit_point.XY();
82
83 if (hit_y >= min_y && hit_y <= max_y &&
84 (!hit || units < min_units))
85 {
86 hit = true;
87 min_units = units; //Closest intersection (so far)
88 }
89 }
90 }
91
92 //Max x
93 if (x >= max_x && dir_x < 0.0_r)
94 {
95 if (auto units = (max_x - x) / dir_x; units >= 0.0)
96 {
97 auto hit_point = origin_ + direction_ * units;
98 auto [hit_x, hit_y] = hit_point.XY();
99
100 if (hit_y >= min_y && hit_y <= max_y &&
101 (!hit || units < min_units))
102 {
103 hit = true;
104 min_units = units; //Closest intersection (so far)
105 }
106 }
107 }
108
109 //Min y
110 if (y <= min_y && dir_y > 0.0_r)
111 {
112 if (auto units = (min_y - y) / dir_y; units >= 0.0)
113 {
114 auto hit_point = origin_ + direction_ * units;
115 auto [hit_x, hit_y] = hit_point.XY();
116
117 if (hit_x >= min_x && hit_x <= max_x &&
118 (!hit || units < min_units))
119 {
120 hit = true;
121 min_units = units; //Closest intersection (so far)
122 }
123 }
124 }
125
126 //Max y
127 if (y >= max_y && dir_y < 0.0_r)
128 {
129 if (auto units = (max_y - y) / dir_y; units >= 0.0)
130 {
131 auto hit_point = origin_ + direction_ * units;
132 auto [hit_x, hit_y] = hit_point.XY();
133
134 if (hit_x >= min_x && hit_x <= max_x &&
135 (!hit || units < min_units))
136 {
137 hit = true;
138 min_units = units; //Closest intersection (so far)
139 }
140 }
141 }
142
143 return {hit, min_units};
144}

146std::pair<bool, real> Ray::Intersects(const Obb &obb) const noexcept
147{
148 if (obb.Empty())
149 return {false, 0.0_r};
150
151 auto c0 = obb.Corners()[0];
152 auto c1 = obb.Corners()[1];
153 auto c2 = obb.Corners()[2];
154
155 //Rotate obb to align with axis
156 auto angle = -vector2::UnitX.SignedAngleBetween(c1 - c0);
157 auto center = obb.Center();
158 c0.Rotate(angle, center); //Min
159 c2.Rotate(angle, center); //Max
160
161 //Reduce problem to ray-aabb intersection, by rotating ray correspondingly
162 return RotateCopy(angle, center).Intersects(Aabb{c0, c2});
163}

165std::pair<bool, real> Ray::Intersects(const Sphere &sphere) const noexcept
166{
167 if (sphere.Empty())
168 return {false, 0.0_r};
169
170 auto ray_origin = origin_ - sphere.Center();
171 auto radius = sphere.Radius();
172
173 //Ray origin is already inside sphere
174 if (ray_origin.SquaredLength() <= radius * radius)
175 return {true, 0.0_r};
176
177 //Quadratics
178 auto a = direction_.DotProduct(direction_);
179 auto b = ray_origin.DotProduct(direction_) * 2.0_r;
180 auto c = ray_origin.DotProduct(ray_origin) - radius * radius;
181
182 //Check determinant
183 if (auto det = b * b - 4 * a * c; det < 0.0_r)
184 return {false, 0.0_r}; //No intersection
185
186 else //Intersection
187 {
188 auto det_sqrt = std::sqrt(det);
189 auto units = (-b - det_sqrt) / (2.0_r * a);
190
191 if (units < 0.0_r)
192 units = (-b + det_sqrt) / (2.0_r * a);
193
194 return {true, units}; //Closest intersection
195 }
196}
