55 template <typename T>
56 inline auto get_object_by_name(std::string_view name, const object_register::detail::registered_objects<T> &objects) noexcept
57 {
58 auto iter =
59 std::find_if(std::begin(objects), std::end(objects),
60 [&](auto &object) noexcept
61 {
62 if (auto registered_name = object.name; registered_name)
63 return *registered_name == name;
64 else
65 return false;
66 });
67
68 return iter != std::end(objects) ? iter->ptr : NonOwningPtr<T>{};
69 }

73 inline auto get_object_as(const object_register::detail::registered_objects<T> &objects)

164 auto Register(NonOwningPtr<T> object)
165 {
166 auto iter =
167 std::find_if(std::begin(objects_), std::end(objects_),
168 [&](auto &x) noexcept
169 {
170 return x.ptr == object;
171 });
172
173 //Object not already registered
174 if (iter == std::end(objects_))
175 {
176 objects_.emplace_back(std::nullopt, object);
177 return true;
178 }
179 else
180 return false;
181 }

184 auto Register(std::string name, NonOwningPtr<T> object)
185 {
186 if (!object_register::detail::get_object_by_name(name, objects_))
187 {
188 if (Register(object))
189 {
190 objects_.back().name = std::move(name);
191 return true;
192 }
193 }
194
195 return false;
196 }
