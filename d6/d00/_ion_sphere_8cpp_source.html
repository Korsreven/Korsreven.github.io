33void draw_bounds(real radius, const Vector2 &center, const Color &color, int steps) noexcept
34{
35 #if defined(ION_DOUBLE_PRECISION) || defined(ION_EXTENDED_PRECISION)
36 glColor4dv(color.Channels());
37 #else
38 glColor4fv(color.Channels());
39 #endif
40
41 auto angle = 0.0_r;
42 auto delta_angle = math::TwoPi / (steps - 1);
43
44 glBegin(GL_LINE_STRIP);
45
46 for (auto i = 0; i < steps; ++i, angle += delta_angle)
47 {
48 auto v =
49 center + Vector2{radius * math::Cos(angle), radius * math::Sin(angle)};
50
51 #if defined(ION_DOUBLE_PRECISION) || defined(ION_EXTENDED_PRECISION)
52 glVertex2dv(v.Components());
53 #else
54 glVertex2fv(v.Components());
55 #endif
56 }
57
58 glEnd();
59}

175bool Sphere::Intersects(const Aabb &aabb) const noexcept
176{
177 if (aabb.Empty())
178 return false;
179
180 auto aabb_center = aabb.Center();
181 auto half_size = aabb.ToHalfSize();
182 auto closest = detail::clamp(center_ - aabb_center, -half_size, half_size);
183
184 return Intersects(aabb_center + closest);
185}

187bool Sphere::Intersects(const Obb &obb) const noexcept
188{
189 if (obb.Empty())
190 return false;
191
192 auto c0 = obb.Corners()[0];
193 auto c1 = obb.Corners()[1];
194 auto c2 = obb.Corners()[2];
195
196 //Rotate obb to align with axis
197 auto angle = -vector2::UnitX.SignedAngleBetween(c1 - c0);
198 auto center = obb.Center();
199 c0.Rotate(angle, center); //Min
200 c2.Rotate(angle, center); //Max
201
202 //Reduce problem to sphere-aabb intersection, by rotating sphere correspondingly
203 return Sphere{radius_, center_.RotateCopy(angle, obb.Center())}.Intersects(Aabb{c0, c2});
204}

221Sphere Sphere::MergeCopy(const Sphere &sphere) const noexcept
222{
223 if (sphere.Empty())
224 return *this;
225 else if (Empty())
226 return sphere;
227
228 auto center_diff = sphere.center_ - center_;
229 auto length_squared = center_diff.SquaredLength();
230
231 auto radius_diff = sphere.radius_ - radius_;
232 auto radius_diff_squared = radius_diff * radius_diff;
233
234 if (radius_diff_squared >= length_squared)
235 return radius_diff_squared >= 0.0_r ?
236 sphere :
237 *this;
238 else
239 {
240 auto length = std::sqrt(length_squared);
241 auto theta = radius_diff / (2.0_r * length) + 0.5_r;
242 return {(radius_ + sphere.radius_ + length) * 0.5_r,
243 center_ + center_diff * theta};
244 }
245}

277Sphere Sphere::TransformCopy(const Matrix3 &matrix) const noexcept
278{
279 #ifdef ION_ROW_MAJOR
280 //Row-major layout (Direct3D)
281 #ifdef ION_LEFT_HAND_ROTATION
282 //Left-hand rotation CW
283 auto v = Vector2{matrix.M00(), matrix.M10()} * radius_;
284 #else
285 //Right-hand rotation CCW
286 auto v = Vector2{matrix.M00(), matrix.M01()} * radius_;
287 #endif
288 #else
289 //Column-major layout (OpenGL)
290 #ifdef ION_LEFT_HAND_ROTATION
291 //Left-hand rotation CW
292 auto v = Vector2{matrix.M00(), matrix.M01()} * radius_;
293 #else
294 //Right-hand rotation CCW
295 auto v = Vector2{matrix.M00(), matrix.M10()} * radius_;
296 #endif
297 #endif
298
299 return {((center_ + v) - (center_ - v)).Length() * 0.5_r,
300 matrix.TransformPoint(center_)};
301}
