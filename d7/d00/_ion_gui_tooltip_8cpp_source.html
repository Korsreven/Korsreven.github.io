35Vector2 in_view_offset(const Aabb &tooltip_area, const Aabb &view_area) noexcept
36{
37 auto [tooltip_min, tooltip_max] = tooltip_area.MinMax();
38 auto [view_min, view_max] = view_area.MinMax();
39
40 auto x = 0.0_r;
41 //Outside right edge, nudge left
42 if (tooltip_max.X() > view_max.X())
43 x = view_max.X() - tooltip_max.X();
44
45 //Outside left edge, nudge right
46 if (tooltip_min.X() < view_min.X())
47 x = view_min.X() - tooltip_min.X();
48
49 auto y = 0.0_r;
50 //Outside bottom edge, nudge up
51 if (tooltip_min.Y() < view_min.Y())
52 y = view_min.Y() - tooltip_min.Y();
53
54 //Outside top edge, nudge down
55 if (tooltip_max.Y() > view_max.Y())
56 y = view_max.Y() - tooltip_max.Y();
57
58 return {x, y};
59}

61Vector2 hot_spot_offset(gui_mouse_cursor::MouseCursorHotSpot hot_spot, const Vector2 &tooltip_size, const Vector2 &cursor_size) noexcept
62{
63 auto [half_width, half_height] = (tooltip_size * 0.5_r).XY();
64 auto [cursor_width, cursor_height] = cursor_size.XY();
65
66 switch (hot_spot)
67 {
68 case gui_mouse_cursor::MouseCursorHotSpot::TopLeft:
69 return {half_width, -half_height - cursor_height};
70
71 case gui_mouse_cursor::MouseCursorHotSpot::TopCenter:
72 return {0.0_r, -half_height - cursor_height};
73
74 case gui_mouse_cursor::MouseCursorHotSpot::TopRight:
75 return {-half_width, -half_height - cursor_height};
76
77 case gui_mouse_cursor::MouseCursorHotSpot::Left:
78 return {half_width, -half_height - cursor_height * 0.5_r};
79
80 case gui_mouse_cursor::MouseCursorHotSpot::Right:
81 return {-half_width, -half_height - cursor_height * 0.5_r};
82
83 case gui_mouse_cursor::MouseCursorHotSpot::BottomLeft:
84 return {half_width, half_height + cursor_height};
85
86 case gui_mouse_cursor::MouseCursorHotSpot::BottomCenter:
87 return {0.0_r, half_height + cursor_height};
88
89 case gui_mouse_cursor::MouseCursorHotSpot::BottomRight:
90 return {-half_width, half_height + cursor_height};
91
92 default:
93 return {0.0_r, -half_height - cursor_height * 0.5_r};
94 }
95}

143void GuiTooltip::UpdateCaption() noexcept
144{
145 static auto need_update = true;
146
147 if (skin_)
148 {
149 if (auto &part = skin_->Caption; part && auto_size_ && need_update)
150 {
151 //Caption text
152 if (auto &text = part->Get(); text)
153 {
154 //Area size
155 text->AreaSize({});
156
157 //Padding
158 text->Padding(caption_padding_.value_or(gui_control::detail::default_caption_padding_size));
159
160 //Content
161 if (caption_)
162 text->Content(*caption_);
163 else
164 text->Clear();
165
166
167 if (auto size = text->MinimumAreaSize(); size != vector2::Zero)
168 {
169 //Make sure there is enough space (rounding errors)
170 //Round up so both width and height are even
171 size.X(size.X() + (std::fmod(size.X(), 2.0_r) == 0.0_r ? 2.0_r : 1.0_r));
172 size.Y(size.Y() + (std::fmod(size.Y(), 2.0_r) == 0.0_r ? 2.0_r : 1.0_r));
173
174 auto ppu = Engine::PixelsPerUnit();
175 size /= ppu;
176
177 auto border_size =
178 gui_control::detail::get_border_size(*skin_, false).value_or(vector2::Zero);
179
180 need_update = false;
181 Size(size + border_size);
182 need_update = true;
183 return;
184 }
185 }
186 }

192void GuiTooltip::UpdatePosition(Vector2 position) noexcept
193{
194 if (node_)
195 {
196 if (auto controller = GetController(); controller)
197 {
198 auto size =
199 size_.value_or(vector2::Zero) * node_->DerivedScaling();
200
201 //Custom cursor
202 if (auto mouse_cursor = controller->ActiveMouseCursor(); mouse_cursor)
203 {
204 if (auto mouse_cursor_node = mouse_cursor->Node(); mouse_cursor_node)
205 {
206 auto cursor_size =
207 mouse_cursor->Size().value_or(vector2::Zero) * mouse_cursor_node->DerivedScaling();
208
209 //Adjust tooltip position based on cursor hot spot
210 position += detail::hot_spot_offset(mouse_cursor->HotSpot(), size, cursor_size);
211 }
212 }
213 else //OS cursor
214 {
215 auto cursor_size =
216 [&]() noexcept
217 {
218 if (skin_ && skin_->Parts)
219 {
220 auto cursor_size = Vector2{20.0_r, 20.0_r}; //OS cursor size?
221 auto ppu = Engine::PixelsPerUnit();
222 return cursor_size / ppu;
223 }
224 else
225 return vector2::Zero;
226 }();
227
228 //Adjust tooltip position based on cursor hot spot
229 position += detail::hot_spot_offset(gui_mouse_cursor::MouseCursorHotSpot::TopLeft, size, cursor_size);
230 }
231 }
232
233 node_->DerivedPosition(position);
234 AdjustInView();
235 }

238void GuiTooltip::AdjustInView() noexcept
239{
240 if (node_ && skin_)
241 {
242 //Make sure tooltip stays within view area
243 if (auto view_area =
244 [&]() noexcept
245 {
246 //Get bounding box from the default viewport camera
247 if (auto controller = GetController(); controller)
248 {
249 if (auto viewport = controller->DefaultViewport(); viewport)
250 {
251 if (auto camera = viewport->ConnectedCamera(); camera)
252 return camera->WorldAxisAlignedBoundingBox();
253 }
254 }
255
256 return aabb::Zero;
257 }(); view_area != aabb::Zero)
258 {
259 if (auto tooltip_area =
260 [&]() noexcept
261 {
262 if (skin_->Parts)
263 {
264 skin_->Parts->Prepare();
265 return skin_->Parts->WorldAxisAlignedBoundingBox();
266 }
267 else if (skin_->Caption)
268 {
269 skin_->Caption->Prepare();
270 return skin_->Caption->WorldAxisAlignedBoundingBox();
271 }
272
273 return aabb::Zero;
274 }(); tooltip_area != aabb::Zero)
275
276 node_->Position(node_->Position() + detail::in_view_offset(tooltip_area, view_area));
277 }
278 }

365void GuiTooltip::Show() noexcept
366{
367 if (!visible_)
368 {
369 SetOpacity(0.0_r);
370 SetPhase(detail::tooltip_phase::PreFadeIn);
371 phase_duration_.Total(0.0_sec);
372 }
373 else
374 {
375 if (phase_ != detail::tooltip_phase::PreFadeIn &&
376 phase_ != detail::tooltip_phase::FadeIn)
377 {
378 switch (phase_)
379 {
380 case detail::tooltip_phase::FadeOut:
381 SetPhase(detail::tooltip_phase::FadeIn);
382 phase_duration_.Percent(opacity_);
383 break;
384
385 default:
386 SetPhase(detail::tooltip_phase::Hold);
387 phase_duration_.Total(0.0_sec);
388 break;
389 }
390 }
391 }
392
393 update_position_ = true;
394 GuiControl::Show();
395}

397void GuiTooltip::Hide() noexcept
398{
399 if (visible_)
400 {
401 if (phase_ != detail::tooltip_phase::PreFadeOut &&
402 phase_ != detail::tooltip_phase::FadeOut)
403 {
404 switch (phase_)
405 {
406 case detail::tooltip_phase::PreFadeIn:
407 phase_duration_.Total(0.0_sec);
408 GuiControl::Hide();
409 break;
410
411 case detail::tooltip_phase::FadeIn:
412 SetPhase(detail::tooltip_phase::FadeOut);
413 phase_duration_.Percent(1.0_r - opacity_);
414 break;
415
416 default:
417 SetPhase(detail::tooltip_phase::PreFadeOut);
418 phase_duration_.Total(0.0_sec);
419 break;
420 }
421 }
422 }

430void GuiTooltip::FrameStarted(duration time) noexcept
431{
432 if (visible_)
433 {
434 if (phase_duration_ += time)
435 {
436 phase_duration_.ResetWithCarry();
437
438 //Switch to next phase
439 SetPhase(
440 [&]() noexcept
441 {
442 switch (phase_)
443 {
444 case detail::tooltip_phase::PreFadeIn:
445 return detail::tooltip_phase::FadeIn;
446
447 case detail::tooltip_phase::FadeIn:
448 return detail::tooltip_phase::Hold;
449
450 case detail::tooltip_phase::Hold:
451 return detail::tooltip_phase::PreFadeOut;
452
453 case detail::tooltip_phase::PreFadeOut:
454 return detail::tooltip_phase::FadeOut;
455
456 case detail::tooltip_phase::FadeOut:
457 default:
458 return detail::tooltip_phase::PreFadeIn;
459 }
460 }());
461
462 switch (phase_)
463 {
464 case detail::tooltip_phase::Hold:
465 SetOpacity(1.0_r);
466 break;
467
468 case detail::tooltip_phase::PreFadeIn:
469 SetOpacity(0.0_r);
470 GuiControl::Hide();
471 break;
472 }
473 }
474
475 switch (phase_)
476 {
477 case detail::tooltip_phase::FadeIn:
478 SetOpacity(phase_duration_.Percent());
479 break;
480
481 case detail::tooltip_phase::FadeOut:
482 SetOpacity(1.0_r - phase_duration_.Percent());
483 break;
484 }
485 }
486}
