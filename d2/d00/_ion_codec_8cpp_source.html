21bool is_base(std::string_view str, int base) noexcept
22{
23 static const auto first_number = '0';
24
25 if (base <= 10)
26 {
27 auto last_number =
28 static_cast<char>(first_number + (base - 1));
29
30 for (auto c : str)
31 {
32 if (!(c >= first_number && c <= last_number))
33 return false;
34 }
35 }
36 else
37 {
38 static const auto last_number = '9';
39 static const auto first_letter_lower = 'a';
40 static const auto first_letter_upper = 'A';
41
42 auto last_letter_lower =
43 static_cast<char>(first_letter_lower + (base - 11));
44 auto last_letter_upper =
45 static_cast<char>(first_letter_upper + (base - 11));
46
47 for (auto c : str)
48 {
49 if (!((c >= first_number && c <= last_number) ||
50 (c >= first_letter_lower && c <= last_letter_lower) ||
51 (c >= first_letter_upper && c <= last_letter_upper)))
52 return false;
53 }
54 }
55
56 return true;
57}

59std::string string_to_hex(std::string_view str)
60{
61 std::string result;
62 result.reserve(std::size(str) * 2);
63
64 for (unsigned char c : str)
65 {
66 result.push_back(base36_map[c >> 4]);
67 result.push_back(base36_map[c & 15]);
68 }

88std::optional<std::string> hex_to_string(std::string_view str)
89{
90 if (std::empty(str))
91 return "";
92
93 auto size = std::ssize(str);
94
95 std::string result;
96 result.reserve(size / 2);
97
98 for (auto i = 1; i < size; i += 2)
99 {
100 auto first = rfc_base<base36_map, 16>::
101 Unmap[static_cast<unsigned char>(str[i - 1])];
102 auto second = rfc_base<base36_map, 16>::
103 Unmap[static_cast<unsigned char>(str[i])];
104
105 //Input contains invalid character (not hexadecimal)
106 if (first >= 16 || second >= 16)
107 return {};
108
109 result.push_back((first << 4) | second);
110 }
111
112 return result;
113}
