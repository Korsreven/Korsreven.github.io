28mesh::Vertices curve_vertices(const ControlPoints &control_points, const Color &color, int smoothness)
29{
30 using namespace ion::utilities;
31
32 if (std::ssize(control_points) > max_control_points)
33 return {};
34
35 auto vertex_count =
36 std::ssize(control_points) * (smoothness > 1 ? smoothness : 1);
37
38 mesh::Vertices vertices;
39 vertices.reserve(vertex_count);
40
41 //Don't use bezier
42 if (smoothness == 0 || vertex_count < 2)
43 {
44 for (auto &p : control_points)
45 vertices.push_back({p, vector3::UnitZ, color});
46
47 return vertices;
48 }
49
50 //Use bezier
51 auto point = vector3::Zero;
52 auto percent = 1.0_r;
53 auto step = 1.0_r / (vertex_count - 1);
54
55 //Calculate each coordinate in the bezier curve
56 for (auto i = 0; i < vertex_count; ++i, percent -= step, point = vector3::Zero)
57 {
58 for (auto j = 0; auto &p : control_points)
59 {
60 auto bezier =
61 math::Choose(std::ssize(control_points) - 1, j) *
62 std::pow(percent, static_cast<real>(std::ssize(control_points) - j - 1)) *
63 std::pow(1.0_r - percent, static_cast<real>(j));
64
65 point += p * bezier;
66 point.Z(p.Z());
67 ++j;
68 }
69
70 vertices.push_back({point, vector3::UnitZ, color});
71 }
72
73 return vertices;
74}
