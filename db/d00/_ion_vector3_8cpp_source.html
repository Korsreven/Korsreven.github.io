94real Vector3::SignedAngleBetween(const Vector3 &vector) const noexcept
95{
96 #ifdef ION_LEFT_HANDED
97 //Left-hand rotation CW
98 return std::atan2(-CrossProduct(vector).DotProduct(vector3::UnitZ), DotProduct(vector));
99 #else
100 //Right-hand rotation CCW
101 return std::atan2(CrossProduct(vector).DotProduct(vector3::UnitZ), DotProduct(vector));
102 #endif
103}

105real Vector3::AngleTo(const Vector3 &vector) const noexcept
106{

117
118Vector3 Vector3::CrossProduct(const Vector3 &vector) const noexcept
119{
120 return {y_ * vector.z_ - z_ * vector.y_,
121 z_ * vector.x_ - x_ * vector.z_,
122 x_ * vector.y_ - y_ * vector.x_};
123}
124
125real Vector3::DotProduct(const Vector3 &vector) const noexcept
126{
127 return x_ * vector.x_ + y_ * vector.y_ + z_ * vector.z_;
128}

134
135Vector3 Vector3::Deviant(real angle) const noexcept
136{
137 auto sin_of_angle = math::Sin(angle);
138 auto cos_of_angle = math::Cos(angle);
139
140 #ifdef ION_LEFT_HANDED
141 //Left-hand rotation CW
142 return {cos_of_angle * x_ + sin_of_angle * y_,
143 -sin_of_angle * x_ + cos_of_angle * y_,
144 z_};
145 #else
146 //Right-hand rotation CCW
147 return {cos_of_angle * x_ - sin_of_angle * y_,
148 sin_of_angle * x_ + cos_of_angle * y_,
149 z_};
150 #endif
151}

212Vector3 Vector3::Lerp(const Vector3 &vector, real amount) const noexcept
213{
214 if (amount <= 0.0_r)
215 return *this;
216 else if (amount >= 1.0_r)
217 return vector;
218
219 //Do linear interpolation
220 return {math::Lerp(x_, vector.x_, amount),
221 math::Lerp(y_, vector.y_, amount),
222 math::Lerp(z_, vector.z_, amount)};
223}
224
225Vector3 Vector3::Slerp(Vector3 vector, real amount) const noexcept
226{
227 if (amount <= 0.0_r)
228 return *this;
229 else if (amount >= 1.0_r)
230 return vector;
231
232 //Do spherical linear interpolation
233 auto dot = DotProduct(vector);
234
235 if (dot < 0.0_r)
236 {
237 dot = -dot;
238 vector = -vector;
239 }
240
241 //Fixed potential acos domain error and division by zero
242 if (dot > 1.0_r - math::Epsilon)
243 return Lerp(vector, amount);
244 else
245 {
246 auto angle = std::acos(dot);
247 return {math::Slerp(x_, vector.x_, angle, amount),
248 math::Slerp(y_, vector.y_, angle, amount),
249 math::Slerp(z_, vector.z_, angle, amount)};
250 }
251}
