49vertex_metrics get_vertex_metrics(const render::vertex::VertexDeclaration &vertex_declaration) noexcept
50{
51 auto metrics = vertex_metrics{};
52
53 for (auto &element : vertex_declaration.Elements())
54 {
55 if (element.Name == shaders::shader_layout::AttributeName::Vertex_Rotation)
56 metrics.rotation_offset = element.Offset / static_cast<int>(sizeof(real));
57 else if (element.Name == shaders::shader_layout::AttributeName::Vertex_PointSize)
58 metrics.point_size_offset = element.Offset / static_cast<int>(sizeof(real));
59 }
60
61 return metrics;
62}

68void apply_node_rotation(const vertex_metrics &metrics, real node_rotation, render::render_primitive::VertexContainer &data) noexcept
69{
70 auto size = std::ssize(data);
71 auto stride = sizeof(particles::Particle);
72
73 for (auto off = metrics.rotation_offset; off < size; off += stride)
74 data[off] += node_rotation;
75}
76
77void apply_node_scaling(const vertex_metrics &metrics, const Vector2 &node_scaling, render::render_primitive::VertexContainer &data) noexcept
78{
79 auto size = std::ssize(data);
80 auto stride = sizeof(particles::Particle);
81
82 for (auto off = metrics.point_size_offset; off < size; off += stride)
83 data[off] *= (node_scaling.X() + node_scaling.Y()) * 0.5_r; //Average
84}

87void get_emitter_primitives(const particles::ParticleSystem &particle_system, const vertex_metrics &metrics,
88 real node_rotation, const Vector2 &node_scaling, particle_emitter_primitives &emitter_primitives)
89{
90 for (auto off = 0; auto &emitter : particle_system.Emitters())
91 {
92 if (!emitter.HasActiveParticles())
93 continue;
94
95 auto &primitive =
96 [&]() noexcept -> particle_emitter_primitive&
97 {
98 if (off < std::ssize(emitter_primitives))
99 {
100 emitter_primitives[off]->RenderMaterial(emitter.ParticleMaterial());
101 return *emitter_primitives[off];
102 }
103 else
104 return *emitter_primitives.emplace_back(make_owning<particle_emitter_primitive>(emitter.ParticleMaterial()));
105 }();
106
107 auto size = std::size(emitter.Particles()) * sizeof(particles::Particle);
108 primitive.vertex_data.resize(size / sizeof(real), 0.0_r);
109 std::memcpy(std::data(primitive.vertex_data), std::data(emitter.Particles()), size);
110
111 if (node_rotation != 0.0_r)
112 apply_node_rotation(metrics, node_rotation, primitive.vertex_data);
113
114 if (node_scaling != vector2::UnitScale)
115 apply_node_scaling(metrics, node_scaling, primitive.vertex_data);
116
117 ++off;
118 }
119}

126void DrawableParticleSystem::ReloadPrimitives()
127{
128 render_primitives_.clear();
129
130 if (particle_system_)
131 {
132 auto parent_node = ParentNode();
133 auto node_rotation = parent_node ? parent_node->Rotation() : 0.0_r;
134 auto node_scaling = parent_node ? parent_node->Scaling() : vector2::UnitScale;
135
136 detail::get_emitter_primitives(*particle_system_, vertex_metrics_,
137 node_rotation, node_scaling, emitter_primitives_);
138 }
139
140 std::erase_if(emitter_primitives_,
141 [&](auto &primitive) noexcept
142 {
143 if (!std::empty(primitive->vertex_data))
144 {
145 primitive->owner = this;
146 primitive->VertexData(std::move(primitive->vertex_data));
147 AddPrimitive(*primitive);
148 return false; //Keep
149 }
150 else
151 return true;
152 });
153}

189void DrawableParticleSystem::Prepare()
190{
191 if (reload_primitives_)
192 {
193 ReloadPrimitives();
194 reload_primitives_ = false;
195 update_bounding_volumes_ = true;
196 }
197
198 //Prepare primitives
199 for (auto &primitive : emitter_primitives_)
200 primitive->Prepare();
201
202 if (update_bounding_volumes_)
203 {
204 aabb_ = {};
205
206 //Merge all bounding boxes
207 for (auto &primitive : emitter_primitives_)
208 aabb_.Merge(primitive->AxisAlignedBoundingBox());
209
210 obb_ = aabb_;
211 sphere_ = {aabb_.ToHalfSize().Max(), aabb_.Center()};
212
213 update_bounding_volumes_ = false;
214 }
215
216 DrawableObject::Prepare();
217}
