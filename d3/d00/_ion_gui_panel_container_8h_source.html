217 template <typename T, typename... Args>
218 auto CreateControl(std::string name, Args &&...args)
219 {
220 static_assert(std::is_base_of_v<controls::GuiControl, T>);
221
222 //Find the default skin for the given control of type T
223 if (auto skin_name = skins::GuiSkin::GetDefaultSkinName<T>(); skin_name)
224 {
225 if (auto skin = GetSkin(*skin_name); skin)
226 return CreateControl<T>(std::move(name), *skin, std::forward<Args>(args)...);
227 }
228
229 auto ptr = CreateComponent<T>(std::move(name), std::forward<Args>(args)...);
230 return static_pointer_cast<T>(ptr);
231 }
