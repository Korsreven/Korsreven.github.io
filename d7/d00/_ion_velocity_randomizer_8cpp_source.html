29void affect_particles(affector::detail::particle_range particles, duration time,
30 const std::pair<real, real> &velocity, real scope) noexcept
31{
32 for (auto &particle : particles)
33 {
34 if (scope > random::Number())
35 particle.Velocity(particle.Velocity() * random::Number(velocity.first, velocity.second) * time.count());
36 }
37}
