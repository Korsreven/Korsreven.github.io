90void RenderWindow::Moved(const Vector2 &position) noexcept
91{
92 //Track position only in windowed mode
93 if (display_mode_ == WindowDisplayMode::Windowed)
94 {
95 //Keep position nullopt when window is centered
96 if (position_ || !IsCentered())
97 position_ = position;
98 }
99
100 NotifyWindowMoved(position);
101 NotifyWindowActionReceived(events::listeners::WindowAction::Move);
102}
103
104void RenderWindow::Resized(const Vector2 &size) noexcept
105{
106 //Track position only in windowed mode
107 if (display_mode_ == WindowDisplayMode::Windowed)
108 size_ = size;
109
110 NotifyWindowResized(size);
111 NotifyRenderTargetResized(size);
112 NotifyWindowActionReceived(events::listeners::WindowAction::Resize);
113}

115void RenderWindow::DisplayModeChanged() noexcept
116{
117 switch (display_mode_)
118 {
119 case WindowDisplayMode::Windowed:
120 display_mode_ = pending_display_mode_ = WindowDisplayMode::FullScreen;
121 break;
122
123 case WindowDisplayMode::FullScreen:
124 default:
125 display_mode_ = pending_display_mode_ = WindowDisplayMode::Windowed;
126 break;
127 }
128}

223bool RenderWindow::Create() noexcept
224{
225 switch (border_style_)
226 {
227 case WindowBorderStyle::None:
228 return system::Window::CreateBorderless(title_, size_, position_, color_depth_);
229
230 case WindowBorderStyle::Dialog:
231 return system::Window::CreateDialog(title_, size_, position_, color_depth_);
232
233 case WindowBorderStyle::Single:
234 return system::Window::CreateNonResizable(title_, size_, position_, color_depth_);
235
236 case WindowBorderStyle::Sizeable:
237 return system::Window::CreateResizable(title_, size_, position_, color_depth_);
238 }
239
240 return false;
241}
