24size_t case_insensitive_find(std::string_view str1, std::string_view str2, size_t off) noexcept
25{
26 auto iter = std::search(std::begin(str1) + off, std::end(str1),
27 std::begin(str2), std::end(str2), case_insensitive_equal);
28 return iter != std::end(str1) ?
29 //Found
30 iter - std::begin(str1) :
31 //Not found
32 std::string::npos;

63{

75std::string combine(const Strings &parts, std::optional<std::string_view> delimiter,
76 int start, std::optional<int> count)
77{
78 if (start < 0)
79 start = 0;
80
81 if (count && *count <= 0 ||
82 start >= std::ssize(parts))
83 return "";
84
85 auto str = parts[start];
86 auto last = std::min(
87 start + count.value_or(std::size(parts)),
88 std::ssize(parts));
89
90 if (last > ++start)
91 {
92 //Combine with delimiter
93 if (delimiter)
94 {
95 for (auto iter = std::cbegin(parts) + start,
96 end = std::cbegin(parts) + last;
97 iter != end; ++iter)
98 {
99 str += *delimiter;
100 str += *iter;
101 }
102 }
103 //Combine without delimiter
104 else
105 {
106 for (auto iter = std::cbegin(parts) + start,
107 end = std::cbegin(parts) + last;
108 iter != end; ++iter)
109 str += *iter;
110 }
111 }
112
113 return str;
114}

116Strings tokenize(std::string_view str, std::string_view delimiters,
117 std::optional<int> max_splits, StringSplitOptions split_options)
118{
119 if (max_splits && *max_splits <= 0)
120 return {};
121
122 Strings parts;
123
124 for (size_t from = 0, to = 0;
125 to != std::string::npos;
126 from = to + 1)
127 {
128 to = str.find_first_of(delimiters, from);
129 auto part = str.substr(from, to - from);
130
131 if ((split_options == StringSplitOptions::PreserveDelimiters ||
132 split_options == StringSplitOptions::PreserveEmptyEntriesAndDelimiters) &&
133 from > 0)
134 {
135 parts.push_back(std::string(1, str[from - 1]));
136
137 if (max_splits &&
138 *max_splits == std::ssize(parts))
139 break;
140 }
141
142 if (std::empty(part) &&
143 !(split_options == StringSplitOptions::PreserveEmptyEntries ||
144 split_options == StringSplitOptions::PreserveEmptyEntriesAndDelimiters))
145 continue;
146
147 parts.emplace_back(part);
148
149 if (max_splits &&
150 *max_splits == std::ssize(parts))
151 break;
152 }
153
154 return parts;
155}

157std::string format(real number, std::string_view format_str)
158{
159 if (!std::isfinite(number))
160 return convert::ToString(number); //inf, nan
161
162 //Format specifiers
163
164 auto decimal_point = format_str.find(detail::format_decimal_point);
165 auto natural_placeholder = decimal_point != std::string::npos ?
166 format_str.substr(0, decimal_point).find_last_of(detail::format_placeholders) :
167 format_str.find_last_of(detail::format_placeholders);
168 auto mantissa_placeholder = decimal_point != std::string::npos ?
169 format_str.find_first_of(detail::format_placeholders, decimal_point + 1) :
170 std::string::npos;
171
172 auto min_digits =
173 [&]() noexcept
174 {
175 auto digits = 0;
176 auto zero_digits = 0;
177
178 for (; natural_placeholder != std::string::npos;
179 natural_placeholder = format_str.find_last_of(detail::format_placeholders, natural_placeholder - 1))
180 {
181 ++digits;
182
183 if (format_str[natural_placeholder] == detail::format_zero_placeholder)
184 zero_digits = digits;
185
186 if (natural_placeholder == 0)
187 break;
188 }
189
190 return zero_digits;
191 }();
192
193 auto [min_decimal_digits, decimal_digits] =
194 [&]() noexcept -> std::pair<int, int>
195 {
196 auto digits = 0;
197 auto zero_digits = 0;
198
199 for (; mantissa_placeholder != std::string::npos;
200 mantissa_placeholder = format_str.find_first_of(detail::format_placeholders, mantissa_placeholder + 1))
201 {
202 ++digits;
203
204 if (format_str[mantissa_placeholder] == detail::format_zero_placeholder)
205 zero_digits = digits;
206 }
207
208 return {zero_digits, digits};
209 }();
210
211 //Format number
212
213 auto str = (decimal_digits == 0 && decimal_point != std::string::npos) ?
214 //Arbitrary precision
215 convert::ToString(number, {}) :
216 //Fixed precision
217 convert::ToString(number, decimal_digits);
218 auto str_decimal_point = str.find('.');
219 auto natural_digits = str_decimal_point != std::string::npos ?
220 str_decimal_point :
221 std::size(str);
222
223 //Remove rightmost zeroes from decimals
224 for (auto digits = decimal_digits; digits > min_decimal_digits; --digits)
225 {
226 if (str.back() != '0')
227 break;
228
229 str.pop_back();
230 }
231
232 //Do not end with decimal point
233 if (str.back() == '.')
234 str.pop_back();
235
236 //Left pad with zeroes
237 if (min_digits > static_cast<int>(natural_digits))
238 {
239 str.insert(0, min_digits - natural_digits, '0');
240 natural_digits += min_digits - natural_digits;
241 }
242
243 //Add thousands separators
244 if (format_str.find(detail::format_thousands_separator) != std::string::npos)
245 {
246 if (natural_digits > 3)
247 {
248 auto count = (natural_digits - 1) / 3;
249 for (auto off = natural_digits - 3; count > 0; --count, off -= 3)
250 str.insert(off, 1, ' ');
251 }
252 }
253
254 return str;
255}

277bool StartsWith(std::string_view str1, std::string_view str2,
278 StringCase string_case) noexcept
279{
280 if (std::empty(str2))
281 return std::empty(str1);
282
283 return string_case == StringCase::Insensitive ?
284 //Insensitive
285 std::size(str1) >= std::size(str2) &&
286 std::equal(std::begin(str1), std::begin(str1) + std::size(str2),

288 //Sensitive
289 str1.compare(0, std::size(str2), str2) == 0;

303 //Sensitive
304 str1.compare(std::size(str1) - std::size(str2), std::size(str2), str2) == 0;

307bool Match(std::string_view str, std::string_view pattern,
308 StringCase string_case) noexcept
309{
310 //No pattern
311 if (std::empty(pattern))
312 return std::empty(str);
313
314 //Find first wildcard
315 auto [off, count] = detail::find_wildcard(pattern, 0);
316
317 //Pattern contains no wildcards
318 // _
319 if (off == std::string::npos)
320 return detail::compare_pattern(std::begin(str), std::end(str),
321 std::begin(pattern), std::end(pattern), string_case);
322
323 //Pattern contains only wildcards
324 // %*
325 if (off == 0 && count == std::size(pattern))
326 return true;
327
328 //Pattern contains only wildcard at back
329 // _%*
330 if (off == std::size(pattern) - count)
331 return detail::compare_pattern(std::begin(str), std::begin(str) + (std::size(pattern) - count),
332 std::begin(pattern), std::end(pattern) - count, string_case);
333
334 //Find second wildcard
335 auto [off2, count2] = detail::find_wildcard(pattern, off + count);
336
337 //Pattern contains only one wildcard
338 if (off2 == std::string::npos)
339 {
340 //Pattern contains only one wildcard at front
341 // %*_
342 if (off == 0)
343 return detail::compare_pattern(
344 std::size(str) < std::size(pattern) - (count - 1) ?
345 //Str less than pattern
346 std::begin(str) :
347 //Str greater or equal to pattern
348 std::end(str) - (std::size(pattern) - count),
349 std::end(str),
350 std::begin(pattern) + count, std::end(pattern), string_case);
351
352 //Patterns contains only one wildcard in middle
353 // _%*_
354 return //Compare left
355 detail::compare_pattern(std::begin(str), std::begin(str) + off,
356 std::begin(pattern), std::begin(pattern) + off, string_case)
357 &&
358 //Compare right
359 detail::compare_pattern(std::end(str) - (std::size(pattern) - off - count), std::end(str),
360 std::begin(pattern) + off + count, std::end(pattern), string_case);
361 }
362
363 //Pattern contains only wildcard at front and back
364 // %*_%*
365 if (off == 0 && off2 == std::size(pattern) - count2)
366 return detail::find_pattern(std::begin(str), std::end(str),
367 std::begin(pattern) + count, std::end(pattern) - count2, string_case) != std::string::npos;
368
369 //First wildcard (not leftmost)
370 // _%*_
371 if (off > 0 &&
372 !detail::compare_pattern(std::begin(str), std::begin(str) + off,
373 std::begin(pattern), std::begin(pattern) + off, string_case))
374 return false;
375
376 //Search through remaining wildcards
377 for (size_t str_off = off; off2 != std::string::npos;)
378 {
379 //More wildcards to the right
380 auto pos = detail::find_pattern(std::begin(str) + str_off, std::end(str),
381 std::begin(pattern) + off + count, std::begin(pattern) + off2, string_case);
382
383 if (pos == std::string::npos)
384 return false;
385
386 str_off += pos + (off2 - (off + count));
387
388 //Get next wildcard
389 auto [i, n] = detail::find_wildcard(pattern, off2 + count2);
390 off = off2;
391 count = count2;
392 off2 = i;
393 count2 = n;
394 }
395
396 //Last wildcard (not rightmost)
397 return off + count < std::size(pattern) ?
398 detail::compare_pattern(std::end(str) - (std::size(pattern) - (off + count)), std::end(str),
399 std::begin(pattern) + off + count, std::end(pattern), string_case) :
400 true;
401}

480std::string& PadLeft(std::string &str, int length, std::string_view characters)
481{
482 //String is too short, pad with the extra characters
483 if (std::ssize(str) < length &&
484 !std::empty(characters))
485 {
486 auto pad_length = length - std::size(str);
487 str.insert(std::begin(str), pad_length, '\0');
488
489 for (size_t i = 0; i < pad_length; ++i)
490 str[i] = characters[i % std::size(characters)];
491 }
492
493 return str;
494}
495
496std::string& PadRight(std::string &str, int length, std::string_view characters)
497{
498 //String is too short, pad with the extra characters
499 if (std::ssize(str) < length &&
500 !std::empty(characters))
501 {
502 if (static_cast<int>(str.capacity()) < length)
503 str.reserve(length);
504
505 auto pad_length = length - std::size(str);
506
507 for (size_t i = 0; i < pad_length; ++i)
508 str += characters[i % std::size(characters)];
509 }
510
511 return str;
512}

593std::string& ReplaceAll(std::string &str, std::string_view what, std::string_view with_what,
594 StringCase string_case)
595{
596 if (string_case == StringCase::Insensitive)
597 {
598 for (size_t off = 0;
599 (off = detail::case_insensitive_find(str, what, off)) != std::string::npos;
600 off += std::size(with_what))
601 str.replace(off, std::size(what), with_what);
602 }
603 //Case sensitive
604 else
605 {
606 for (size_t off = 0;
607 (off = str.find(what, off)) != std::string::npos;
608 off += std::size(with_what))
609 str.replace(off, std::size(what), with_what);
610 }
611
612 return str;
613}
