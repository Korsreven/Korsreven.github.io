60bool AsyncTimerManager::Elapse() noexcept
61{
62 auto time = stopwatch_.Elapsed();
63
64 if (time >= detail::minimum_time_elapse_resolution)
65 {
66 for (auto &timer : Objects())
67 timer.Elapse(time);
68
69 stopwatch_.Restart();
70 }
71
72 return true;
73}
