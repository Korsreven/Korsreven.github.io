55void MovableSoundListener::Elapse(duration time) noexcept
56{
57 if (auto parent_node = ParentNode(); parent_node)
58 {
59 auto world_position = position_ + parent_node->DerivedPosition();
60
61 if (sound_listener_)
62 {
63 auto velocity = world_position - previous_world_position_.value_or(world_position);
64 auto velocity_units_sec = velocity * (1.0_r / time.count());
65 sound_listener_->Attributes(world_position, velocity_units_sec);
66 }
67
68 previous_world_position_ = world_position;
69 }
70 else
71 previous_world_position_ = {};
72}
