62 inline void get_eligible_objects(SceneNode &node, uint32 mask, uint32 type_mask, bool only_visible,
63 query_objects &objects) noexcept
64 {
65 const auto default_flags = (mask == ~0_ui32 ? mask : 0_ui32);
66
67 //Check if node is eligible
68 if (!only_visible || node.Visible())
69 {
70 for (auto &attached_object : node.AttachedObjects())
71 {
72 auto object =
73 std::visit([](auto &&x) noexcept -> MovableObject* { return x; }, attached_object);
74
75 //Check if object is eligible
76 if ((!only_visible || object->Visible()) &&
77 object->QueryTypeFlags() & type_mask &&
78 object->QueryFlags().value_or(default_flags) & mask)
79
80 objects.emplace_back(object, true); //Eligible for querying
81 }
82 }
83
84 for (auto &child_node : node.ChildNodes())
85 get_eligible_objects(child_node, mask, type_mask, only_visible, objects); //Recursive
86 }
87
88 inline auto get_eligible_objects(SceneNode &node, uint32 mask, uint32 type_mask, bool only_visible) noexcept
89 {
90 query_objects objects;
91 get_eligible_objects(node, mask, type_mask, only_visible, objects);
92 return objects;
93 }

95 inline void derive_bounding_volumes(query_objects &objects) noexcept
96 {
97 for (auto &object : objects)
98 {
99 switch (object.first->PreferredBoundingVolume())
100 {
101 case movable_object::PreferredBoundingVolumeType::BoundingSphere:
102 {
103 [[maybe_unused]] auto &sphere = object.first->WorldBoundingSphere();
104 break;
105 }
106
107 case movable_object::PreferredBoundingVolumeType::BoundingBox:
108 default:
109 {
110 [[maybe_unused]] auto &aabb = object.first->WorldAxisAlignedBoundingBox();
111 break;
112 }
113 }
114 }

117 inline void remove_objects_outside_region(query_objects &objects, const Aabb &region) noexcept
118 {
119 objects.erase(
120 std::remove_if(std::begin(objects), std::end(objects),
121 [&](auto &object) noexcept
122 {
123 switch (object.first->PreferredBoundingVolume())
124 {
125 case movable_object::PreferredBoundingVolumeType::BoundingSphere:
126 return !object.first->WorldBoundingSphere(false).Intersects(region);
127
128 case movable_object::PreferredBoundingVolumeType::BoundingBox:
129 default:
130 return !object.first->WorldAxisAlignedBoundingBox(false).Intersects(region);
131 }
132 }),
133 std::end(objects));
134 }
