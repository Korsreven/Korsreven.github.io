67 constexpr auto is_power_of_two(uint32 x) noexcept
68 {
69 return x && !(x & (x - 1));
70 }
71
72 constexpr auto lower_power_of_two(uint32 x) noexcept
73 {
74 x |= x >> 1;
75 x |= x >> 2;
76 x |= x >> 4;
77 x |= x >> 8;
78 x |= x >> 16;
79 return x - (x >> 1);
80 }
81
82 constexpr auto upper_power_of_two(uint32 x) noexcept
83 {
84 --x;
85 x |= x >> 1;
86 x |= x >> 2;
87 x |= x >> 4;
88 x |= x >> 8;
89 x |= x >> 16;
90 ++x;
91 return x;
92 }

94 constexpr auto nearest_power_of_two(uint32 x) noexcept
95 {
96 auto lower = lower_power_of_two(x);
97 auto upper = upper_power_of_two(x);
98 return x - lower < upper - x ? lower : upper;
99 }

101 constexpr auto make_power_of_two(uint32 x, NpotScale npot_scale) noexcept
102 {
103 switch (npot_scale)
104 {
105 case NpotScale::ToLarger:
106 return upper_power_of_two(x);
107
108 case NpotScale::ToSmaller:
109 return lower_power_of_two(x);
110
111 case NpotScale::ToNearest:
112 default:
113 return nearest_power_of_two(x);
114 }

123 inline auto power_of_two_adjusted_size(int width, int height,
124 NpotScale npot_scale, std::optional<NpotScaleFit> npot_scale_fit) noexcept
125 {
126 auto aspect_ratio = static_cast<real>(width) / height;
127
128 //Max texture limit reached
129 if (width > max_texture_size() || height > max_texture_size())
130 {
131 if (width > height)
132 {
133 width = max_texture_size();
134 height = static_cast<int>(width / aspect_ratio);
135 }
136 else
137 {
138 height = max_texture_size();
139 width = static_cast<int>(height * aspect_ratio);
140 }
141 }
142 else
143 {
144 auto pot_width = static_cast<int>(make_power_of_two(width, npot_scale));
145 auto pot_height = static_cast<int>(make_power_of_two(height, npot_scale));
146
147 if (!npot_scale_fit)
148 npot_scale_fit =
149 [&]() noexcept
150 {
151 //Choose minimum npot->pot difference
152 if (std::abs(width - pot_width) < std::abs(height - pot_height))
153 return NpotScaleFit::Horizontally;
154 else
155 return NpotScaleFit::Vertically;
156 }();
157
158 switch (*npot_scale_fit)
159 {
160 case NpotScaleFit::Horizontally:
161 {
162 width = pot_width;
163 height = static_cast<int>(width / aspect_ratio);
164 break;
165 }
166
167 case NpotScaleFit::Vertically:
168 {
169 height = pot_height;
170 width = static_cast<int>(height * aspect_ratio);
171 break;
172 }
173 }
174 }
175
176 return std::pair{width, height};
177 }

179 inline auto power_of_two_padding(int width, int height) noexcept
180 {
181 if (auto is_width_pot = is_power_of_two(width),
182 is_height_pot = is_power_of_two(height); !is_width_pot || !is_height_pot)
183 {
184 //Pad left and right
185 if (!is_width_pot)
186 {
187 auto padding_width = static_cast<int>(upper_power_of_two(width)) - width;
188 auto padding_half_width = padding_width / 2;
189 return std::tuple{padding_half_width, 0, padding_half_width + padding_width % 2, 0};
190 }
191 //Pad top and bottom
192 else
193 {
194 auto padding_height = static_cast<int>(upper_power_of_two(height)) - height;
195 auto padding_half_height = padding_height / 2;
196 return std::tuple{0, padding_half_height, 0, padding_half_height + padding_height % 2};
197 }
198 }
199 else
200 return std::tuple{0, 0, 0, 0};
201 }

244 template <typename... Args>
245 void CreateSubTextures(NonOwningPtr<TextureAtlas> texture_atlas, Args &&...args)
246 {
247 using namespace ion::utilities;
248
249 if (texture_atlas)
250 {
251 auto &name = *texture_atlas->Name();
252
253 texture::TextureAtlasRegion atlas_region;
254 atlas_region.Atlas = texture_atlas;
255 atlas_region.Position = {1, 1};
256
257 for (auto i = 0; i < texture_atlas->SubTextures(); ++i)
258 {
259 CreateResource(string::Concat(name, "_", i + 1), texture_atlas->AssetName(), atlas_region,
260 std::forward<Args>(args)...);
261 texture_manager::detail::next_sub_texture_position(atlas_region.Position,
262 texture_atlas->Rows(), texture_atlas->Columns(), texture_atlas->SubTextureOrder());
263 }
264 }
