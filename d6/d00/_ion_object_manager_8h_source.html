41 inline auto get_object_by_name(std::string_view name, const object_manager::detail::container_type<T> &objects) noexcept
42 {
43 auto iter =
44 std::find_if(std::begin(objects), std::end(objects),
45 [&](auto &object) noexcept
46 {
47 if (auto object_name = object->Name(); object_name)
48 return *object_name == name;
49 else
50 return false;
51 });
52
53 return iter != std::end(objects) ? NonOwningPtr<T>{*iter} : NonOwningPtr<T>{};
54 }

96 auto NotifyRemovable(ObjectT &object) noexcept
97 {
98 auto removable = Removable(object); //Notify derived first
99
100 for (auto &listener : this->Listeners())
101 removable &= this->Notify(&ListenerT::ObjectRemovable, listener, object).value_or(true);
102 //If one listener disagrees (returns false), the object can not be removed
103
104 return removable;
105 }

108 {
109 Removed(object); //Notify derived first

288 auto Extract(ObjectT &object) noexcept
289 {
290 auto iter =
291 std::find_if(std::begin(objects_), std::end(objects_),
292 [&](auto &x) noexcept
293 {
294 return x.get() == &object;
295 });
296
297 //object found
298 if (iter != std::end(objects_) &&
299 NotifyRemovable(object))
300 {
301 RemovalStarted();
302
303 typename decltype(objects_)::value_type object_ptr = std::move(*iter); //Extend lifetime
304 objects_.erase(iter);
305 NotifyRemoved(object); //Still valid
306
307 RemovalEnded();
308 return object_ptr;
309 }
310 else
311 return typename decltype(objects_)::value_type{};
312 }
313
314 auto ExtractAll() noexcept
315 {
316 //[objects to keep, objects to remove]
317 auto iter =
318 std::stable_partition(std::begin(objects_), std::end(objects_),
319 [&](auto &object) noexcept
320 {
321 return !NotifyRemovable(*object);
322 });
323
324 //Something to remove
325 if (iter != std::end(objects_))
326 {
327 RemovalStarted();
328
329 decltype(objects_) objects;
330 std::move(iter, std::end(objects_), std::back_inserter(objects)); //Extend lifetime
331 objects_.erase(iter, std::end(objects_)); //Erase the moved range
332 objects_.shrink_to_fit();
333 NotifyRemovedAll(objects); //Still valid
334
335 RemovalEnded();
336 return objects;
337 }
338 else
339 return decltype(objects_){};
340 }

342 void Tidy() noexcept
343 {
344 if (!std::empty(objects_))
345 {
346 NotifyRemovableAll();
347 RemovalStarted();
348
349 decltype(objects_) objects = std::move(objects_);
350 objects_.shrink_to_fit();
351 NotifyRemovedAll(objects);
352
353 RemovalEnded();
354 }

435 auto Adopt(typename decltype(objects_)::value_type &object_ptr) noexcept
436 {
437 assert(object_ptr);
438
439 if (auto object_name = object_ptr->Name(); object_name)
440 {
441 //Check if an object with that name already exists
442 if (auto ptr = object_manager::detail::get_object_by_name(*object_name, objects_); ptr)
443 return NonOwningPtr<ObjectT>{}; //Object exists, could not adopt
444 }
445
446 AdditionStarted();
447
448 auto &ptr = objects_.emplace_back(std::move(object_ptr));
449 ptr->Owner(static_cast<OwnerT&>(*this));
450 NotifyCreated(*ptr);
451
452 AdditionEnded();
453 return NonOwningPtr<ObjectT>{ptr};
454 }

465 void AdoptAll(typename decltype(objects_) &objects) noexcept
466 {
467 decltype(objects_) adoptable_objects;
468
469 for (auto iter = std::begin(objects); iter != std::end(objects) && *iter;)
470 {
471 if (auto object_name = (*iter)->Name(); object_name)
472 {
473 if (auto ptr = object_manager::detail::get_object_by_name(*object_name, objects_); ptr)
474 {
475 ++iter;
476 continue; //Object exists, could not adopt
477 }
478 }
479
480 adoptable_objects.push_back(std::move(*iter));
481 iter = objects.erase(iter);
482 }
483
484 if (!std::empty(adoptable_objects))
485 {
486 AdditionStarted();
487
488 std::move(std::begin(adoptable_objects), std::end(adoptable_objects), std::back_inserter(objects_));
489 std::for_each(std::end(objects_) - std::size(adoptable_objects), std::end(objects_),
490 [&](auto &ptr) mutable noexcept
491 {
492 ptr->Owner(static_cast<OwnerT&>(*this));
493 NotifyCreated(*ptr);
494 });
495
496 AdditionEnded();
497 }
