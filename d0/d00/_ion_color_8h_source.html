40 constexpr auto hex_to_red(uint32 value) noexcept
41 {
42 if (value > max_hex_triplet)
43 return (value >> 24) / 255.0_r; //Hex quartet (rgba)
44 else if (value > max_hex_four_digit)
45 return (value >> 16) / 255.0_r; //Hex triplet (rgb)
46
47 //Shorthand
48 else if (value > max_hex_three_digit)
49 return (((value >> 12) << 4) | (value >> 12)) / 255.0_r; //Four-digit (rgba)
50 else
51 return (((value >> 8) << 4) | (value >> 8)) / 255.0_r; //Three-digit (rgb)
52 }
53
54 constexpr auto hex_to_green(uint32 value) noexcept
55 {
56 if (value > max_hex_triplet)
57 return (value >> 16 & 0xff) / 255.0_r; //Hex quartet (rgba)
58 else if (value > max_hex_four_digit)
59 return (value >> 8 & 0xff) / 255.0_r; //Hex triplet (rgb)
60
61 //Shorthand
62 else if (value > max_hex_three_digit)
63 return (((value >> 8 & 0xf) << 4) | (value >> 8 & 0xf)) / 255.0_r; //Four-digit (rgba)
64 else
65 return (((value >> 4 & 0xf) << 4) | (value >> 4 & 0xf)) / 255.0_r; //Three-digit (rgb)
66 }
67
68 constexpr auto hex_to_blue(uint32 value) noexcept
69 {
70 if (value > max_hex_triplet)
71 return (value >> 8 & 0xff) / 255.0_r; //Hex quartet (rgba)
72 else if (value > max_hex_four_digit)
73 return (value & 0xff) / 255.0_r; //Hex triplet (rgb)
74
75 //Shorthand
76 else if (value > max_hex_three_digit)
77 return (((value >> 4 & 0xf) << 4) | (value >> 4 & 0xf)) / 255.0_r; //Four-digit (rgba)
78 else
79 return (((value & 0xf) << 4) | (value & 0xf)) / 255.0_r; //Three-digit (rgb)
80 }
81
82 constexpr auto hex_to_alpha(uint32 value) noexcept
83 {
84 if (value > max_hex_triplet)
85 return (value & 0xff) / 255.0_r; //Hex quartet (rgba)
86 else if (value > max_hex_four_digit)
87 return 1.0_r; //Hex triplet (rgb)
88
89 //Shorthand
90 else if (value > max_hex_three_digit)
91 return (((value & 0xf) << 4) | (value & 0xf)) / 255.0_r; //Four-digit (rgba)
92 else
93 return 1.0_r; //Three-digit (rgb)
94 }
