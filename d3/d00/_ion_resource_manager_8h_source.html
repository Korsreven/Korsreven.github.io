182 void ProcessPreparedResource(ResourceT &resource, bool prepared) noexcept
183 {
184 if (prepared)
185 {
186 ChangeResourceLoadingState(resource, resource::LoadingState::Prepared);
187
188 //Check loading action
189 switch (resource.LoadingAction())
190 {
191 case resource::LoadingAction::Load:
192 ChangeResourceLoadingState(resource, resource::LoadingState::LoadPending);
193 break;
194 }
195 }
196 else
197 ChangeResourceLoadingState(resource, resource::LoadingState::Failed);
198 }

200 void ProcessPreparedResources() noexcept
201 {
202 auto result = processes_.Get(parallel::worker_pool::Synchronization::NonBlocking);
203
204 //Some resources are done preparing
205 if (!std::empty(result))
206 {
207 for (auto &resource : Resources())
208 {
209 //Check if resource was prepared
210 if (auto iter = result.find(&resource); iter != std::end(result))
211 ProcessPreparedResource(resource, iter->second);
212 }
213 }
214 }

217 void JoinAsyncResourceProcess(ResourceT &resource) noexcept
218 {
219 if (auto prepared = processes_.Get(&resource); prepared) //Blocking
220 ProcessPreparedResource(resource, *prepared);
221 }
222
223 void JoinAllAsyncResourceProcesses() noexcept
224 {
225 auto result = processes_.Get(); //Blocking
226
227 for (auto [resource, prepared] : result)
228 ProcessPreparedResource(*resource, prepared);
229 }

232 void ExecutePrepareResource(ResourceT &resource, resource_manager::ExecutionModel execution_model = resource_manager::ExecutionModel::Synchronous)
233 {
234 ChangeResourceLoadingState(resource, resource::LoadingState::Preparing);
235
236 //Non-blocking
237 if (execution_model == resource_manager::ExecutionModel::Asynchronous)
238 processes_.RunTask(&resource, &ResourceManager::PrepareResource, std::ref(*this), std::ref(resource));
239 //Blocking
240 else
241 ProcessPreparedResource(resource, PrepareResource(resource));
242 }
243
244 void ExecuteLoadResource(ResourceT &resource)
245 {
246 ChangeResourceLoadingState(resource, resource::LoadingState::Loading);
247
248 if (LoadResource(resource))
249 ChangeResourceLoadingState(resource, resource::LoadingState::Loaded);
250 else
251 ChangeResourceLoadingState(resource, resource::LoadingState::Failed);
252 }
253
254 void ExecuteUnloadResource(ResourceT &resource) noexcept
255 {
256 ChangeResourceLoadingState(resource, resource::LoadingState::Unloading);
257
258 if (UnloadResource(resource))
259 {
260 ChangeResourceLoadingState(resource, resource::LoadingState::Unloaded);
261
262 //Check loading action
263 switch (resource.LoadingAction())
264 {
265 case resource::LoadingAction::Prepare:
266 case resource::LoadingAction::Load:
267 ChangeResourceLoadingState(resource, resource::LoadingState::PreparePending);
268 break;
269 }
270 }
271 else
272 ChangeResourceLoadingState(resource, resource::LoadingState::Failed);
273 }

276 void PreparePendingResources(resource_manager::ExecutionModel execution_model = resource_manager::ExecutionModel::Asynchronous)
277 {
278 //Check if any resources needs preparing
279 for (auto &resource : Resources())
280 {
281 //Prepare
282 if (resource.LoadingState() == resource::LoadingState::PreparePending)
283 {
284 NotifyResourceLoadingStateChanged(resource);
285 //Make sure to notify the pending state (in case someone is listening)
286 ExecutePrepareResource(resource, execution_model);
287
288 if (execution_model == resource_manager::ExecutionModel::Synchronous)
289 break; //One at a time
290 }
291 }
292
293 ProcessPreparedResources();
294 //Must be called even if the execution model is synchronous
295 }

434 {

485 auto Prepare(ResourceT &resource, resource_manager::EvaluationStrategy strategy = resource_manager::EvaluationStrategy::Eager)
486 {
487 if (resource.Owner() != this)
488 return false;
489
490 //Resource already preparing
491 if (resource.LoadingState() == resource::LoadingState::Preparing &&
492 strategy == resource_manager::EvaluationStrategy::Eager)
493 {
494 JoinAsyncResourceProcess(resource);
495 return resource.IsPrepared();
496 }
497
498 if (auto prepare = resource.Prepare();
499 prepare && strategy == resource_manager::EvaluationStrategy::Eager)
500 {
501 NotifyResourceLoadingStateChanged(resource);
502 //Make sure to notify the pending state (in case someone is listening)
503 ExecutePrepareResource(resource);
504 return resource.IsPrepared();
505 }
506 else
507 return prepare;
508 }

571 auto Load(ResourceT &resource, resource_manager::EvaluationStrategy strategy = resource_manager::EvaluationStrategy::Eager)
572 {
573 if (resource.Owner() != this)
574 return false;
575
576 //Resource already preparing
577 if (resource.LoadingState() == resource::LoadingState::Preparing &&
578 strategy == resource_manager::EvaluationStrategy::Eager)
579 JoinAsyncResourceProcess(resource);
580
581 if (auto load = resource.Load();
582 load && strategy == resource_manager::EvaluationStrategy::Eager)
583 {
584 NotifyResourceLoadingStateChanged(resource);
585 //Make sure to notify the pending state (in case someone is listening)
586
587 if (resource.LoadingState() == resource::LoadingState::PreparePending)
588 ExecutePrepareResource(resource);
589
590 if (resource.LoadingState() == resource::LoadingState::LoadPending)
591 ExecuteLoadResource(resource);
592
593 return resource.IsLoaded();
594 }
595 else
596 return load;
