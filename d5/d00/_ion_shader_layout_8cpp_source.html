46adaptors::FlatMap<StructName, std::string> make_struct_binding_map(StructBindings struct_bindings) noexcept
47{
48 //Sort on name (asc)
49 std::stable_sort(std::begin(struct_bindings), std::end(struct_bindings),
50 [](auto &x, auto &y) noexcept
51 {
52 return x.second < y.second;
53 });
54
55 //Remove duplicate names
56 struct_bindings.erase(
57 std::unique(std::begin(struct_bindings), std::end(struct_bindings),
58 [](auto &x, auto &y) noexcept
59 {
60 return x.second == y.second;
61 }), std::end(struct_bindings));
62
63 return adaptors::FlatMap<StructName, std::string>{std::move(struct_bindings)};
64}
65
66bool is_struct_unique(std::string_view name, const struct_binding_map &struct_bindings) noexcept
67{
68 //No duplicate names allowed
69 for (auto &binding : struct_bindings)
70 {
71 if (binding.second == name)
72 return false;
73 }
74
75 return true;
76}
