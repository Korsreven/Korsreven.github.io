212 inline void Position(real position) noexcept
213 {
214 if (progress_.Position() != position)

217 auto percent = progress_.Percent();
218 progress_.Position(position);

226 inline void Range(real min, real max) noexcept
227 {
228 if (progress_.Min() != min || progress_.Max() != max)
229 {
230 //Go from or to an empty range
231 auto empty_range =
232 progress_.Min() == progress_.Max() || min == max;
233
234 progress_.Extents(min, max);
235
236 if (empty_range)
237 SetState(state_);
238
239 UpdateBars();
240 }
241 }
