82void set_text_properties(const script_tree::ObjectNode &object, Text &text)
83{
84 for (auto &property : object.Properties())
85 {
86 if (property.Name() == "alignment")
87 {
88 if (property[0].Get<ScriptType::Enumerable>()->Get() == "left")
89 text.Alignment(text::TextAlignment::Left);
90 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "center")
91 text.Alignment(text::TextAlignment::Center);
92 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "right")
93 text.Alignment(text::TextAlignment::Right);
94 }
95 else if (property.Name() == "area-size")
96 text.AreaSize(property[0].Get<ScriptType::Vector2>()->Get());
97 else if (property.Name() == "default-background-color")
98 text.DefaultBackgroundColor(property[0].Get<ScriptType::Color>()->Get());
99 else if (property.Name() == "default-decoration")
100 {
101 if (property[0].Get<ScriptType::Enumerable>()->Get() == "overline")
102 text.DefaultDecoration(text::TextDecoration::Overline);
103 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "line-through")
104 text.DefaultDecoration(text::TextDecoration::LineThrough);
105 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "underline")
106 text.DefaultDecoration(text::TextDecoration::Underline);
107 }
108 else if (property.Name() == "default-decoration-color")
109 text.DefaultDecorationColor(property[0].Get<ScriptType::Color>()->Get());
110 else if (property.Name() == "default-font-style")
111 {
112 if (property[0].Get<ScriptType::Enumerable>()->Get() == "bold")
113 text.DefaultFontStyle(text::TextFontStyle::Bold);
114 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "italic")
115 text.DefaultFontStyle(text::TextFontStyle::Italic);
116 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "bold-italic")
117 text.DefaultFontStyle(text::TextFontStyle::BoldItalic);
118 }
119 else if (property.Name() == "default-foreground-color")
120 text.DefaultForegroundColor(property[0].Get<ScriptType::Color>()->Get());
121 else if (property.Name() == "from-line")
122 text.FromLine(property[0].Get<ScriptType::Integer>()->As<int>());
123 else if (property.Name() == "formatting")
124 {
125 if (property[0].Get<ScriptType::Enumerable>()->Get() == "none")
126 text.Formatting(text::TextFormatting::None);
127 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "html")
128 text.Formatting(text::TextFormatting::HTML);
129 }
130 else if (property.Name() == "line-height")
131 text.LineHeight(property[0].Get<ScriptType::FloatingPoint>()->As<real>());
132 else if (property.Name() == "line-height-factor")
133 text.LineHeightFactor(property[0].Get<ScriptType::FloatingPoint>()->As<real>());
134 else if (property.Name() == "max-lines")
135 text.MaxLines(property[0].Get<ScriptType::Integer>()->As<int>());
136 else if (property.Name() == "overflow")
137 {
138 if (property[0].Get<ScriptType::Enumerable>()->Get() == "truncate")
139 text.Overflow(text::TextOverflow::Truncate);
140 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "truncate-ellipsis")
141 text.Overflow(text::TextOverflow::TruncateEllipsis);
142 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "word-truncate")
143 text.Overflow(text::TextOverflow::WordTruncate);
144 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "word-wrap")
145 text.Overflow(text::TextOverflow::WordWrap);
146 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "wrap")
147 text.Overflow(text::TextOverflow::Wrap);
148 }
149 else if (property.Name() == "padding")
150 text.Padding(property[0].Get<ScriptType::Vector2>()->Get());
151 else if (property.Name() == "vertical-alignment")
152 {
153 if (property[0].Get<ScriptType::Enumerable>()->Get() == "top")

157 else if (property[0].Get<ScriptType::Enumerable>()->Get() == "bottom")

163NonOwningPtr<Text> create_text(const script_tree::ObjectNode &object,
164 TextManager &text_manager, const ManagerRegister &managers)
165{
166 auto name = object
167 .Property("name")[0]
168 .Get<ScriptType::String>()->Get();
169 auto content = object
170 .Property("content")[0]
171 .Get<ScriptType::String>().value_or(""s).Get();
172 auto type_face_name = object
173 .Property("type-face")[0]
174 .Get<ScriptType::String>()->Get();
175
176 auto text = text_manager.CreateText(std::move(name), std::move(content),
177 get_type_face(type_face_name, managers));
178
179 if (text)
180 set_text_properties(object, *text);
181
182 return text;
183}

185void create_texts(const ScriptTree &tree,
186 TextManager &text_manager, const ManagerRegister &managers)
187{
188 for (auto &object : tree.Objects())
189 {
190 if (object.Name() == "text")
191 create_text(object, text_manager, managers);
192 }
