63int get_next_texture_unit(int &next_texture_unit) noexcept
64{
65 static auto max_texture_units = gl::MaxTextureUnits();
66
67 if (next_texture_unit < max_texture_units)
68 return next_texture_unit++;
69 else
70 return -1; //Invalid unit

96void ShaderProgram::Created(variables::UniformVariable &uniform_variable) noexcept
97{
98 if (shader_layout_)
99 detail::remap_uniform(
100 uniform_variable.Owner()->GetUniform(*uniform_variable.Name()), *shader_layout_, mapped_uniforms_);
101
102 uniform_variable.Visit(
103 //1D sampler types
104 [&](variables::glsl::uniform<variables::glsl::isampler1D> &sampler)
105 {
106 sampler = detail::get_next_texture_unit(next_texture_unit_);
107 },
108 [&](variables::glsl::uniform<variables::glsl::usampler1D> &sampler)
109 {
110 sampler = detail::get_next_texture_unit(next_texture_unit_);
111 },
112 [&](variables::glsl::uniform<variables::glsl::sampler1D> &sampler)
113 {
114 sampler = detail::get_next_texture_unit(next_texture_unit_);
115 },
116 [&](variables::glsl::uniform<variables::glsl::isampler1DArray> &sampler)
117 {
118 sampler = detail::get_next_texture_unit(next_texture_unit_);
119 },
120 [&](variables::glsl::uniform<variables::glsl::usampler1DArray> &sampler)
121 {
122 sampler = detail::get_next_texture_unit(next_texture_unit_);
123 },
124 [&](variables::glsl::uniform<variables::glsl::sampler1DArray> &sampler)
125 {
126 sampler = detail::get_next_texture_unit(next_texture_unit_);
127 },
128
129 //2D sampler types
130 [&](variables::glsl::uniform<variables::glsl::isampler2D> &sampler)
131 {
132 sampler = detail::get_next_texture_unit(next_texture_unit_);
133 },
134 [&](variables::glsl::uniform<variables::glsl::usampler2D> &sampler)

145 },
146 [&](variables::glsl::uniform<variables::glsl::usampler2DArray> &sampler)
147 {

159void ShaderProgram::Removed(ShaderStruct &shader_struct) noexcept
160{
161 if (shader_layout_)
162 {
163 for (auto i = 0; auto &s_struct : mapped_structs_)
164 {
165 if (s_struct.get() == &shader_struct)
166 {
167 mapped_structs_[i] = nullptr;
168 break;
169 }
170 else
171 ++i;
172 }

284void ShaderProgram::Layout(NonOwningPtr<ShaderLayout> shader_layout) noexcept
285{
286 if (shader_layout_ != shader_layout)
287 {
288 shader_layout_ = shader_layout;
289
290 mapped_structs_.fill(nullptr);
291 mapped_attributes_.fill(nullptr);
292 mapped_uniforms_.fill(nullptr);
293
294 if (shader_layout_)
295 {
296 //Remap all structs
297 for (auto &shader_struct : Structs())
298 Created(shader_struct); //Reuse functionality
299
300 //Remap all attributes
301 for (auto &attribute : AttributeVariables())
302 Created(attribute); //Reuse functionality
303
304 //Remap all uniforms
305 for (auto &uniform : UniformVariables())
306 Created(uniform); //Reuse functionality
307 }
308 }
