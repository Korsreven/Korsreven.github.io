67NonOwningPtr<FrameSequence> create_frame_sequence(const script_tree::ObjectNode &object,
68 FrameSequenceManager &frame_sequence_manager, const ManagerRegister &managers)
69{
70 auto name = object
71 .Property("name")[0]
72 .Get<ScriptType::String>()->Get();
73 auto first_frame_name = object
74 .Property("frames")[0]
75 .Get<ScriptType::String>().value_or(""s).Get();
76 auto total_frames = object
77 .Property("frames")[1]
78 .Get<ScriptType::Integer>().value_or(0).As<int>();
79
80 //Construct from first frame and total frames
81 if (total_frames > 0)
82 return frame_sequence_manager.CreateFrameSequence(std::move(name),
83 get_texture(first_frame_name, managers), total_frames);
84 else //Construct from one or more given frame
85 {
86 frame_sequence::detail::container_type frames;
87
88 for (auto &property : object.Properties())
89 {
90 if (property.Name() == "frame")
91 frames.push_back(get_texture(property[0].Get<ScriptType::String>()->Get(), managers));
92 }
93
94 if (!std::empty(frames))
95 return frame_sequence_manager.CreateFrameSequence(std::move(name), frames);
96 }
97
98 return nullptr;
99}
