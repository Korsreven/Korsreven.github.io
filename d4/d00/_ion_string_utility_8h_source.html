194 template <size_t N>
195 struct substitute_with_impl
196 {
197 template <typename T>
198 auto operator()(std::string &str, size_t off, size_t length,
199 T &&values, size_t index, std::optional<std::string_view> format_str) const noexcept
200 {
201 //Found correct index
202 if (N - 1 == index)
203 {
204 if (format_str)
205 {
206 auto result = format(realify(std::get<N - 1>(std::forward<T>(values))), *format_str);
207 str.replace(off, length, result);
208 return off + std::size(result);
209 }
210
211 auto &&result = stringify(std::get<N - 1>(std::forward<T>(values)));
212
213 if constexpr (types::is_char_v<decltype(result)>)
214 {
215 str.replace(off, length, 1, result);
216 return off + 1;
217 }
218 else
219 {
220 str.replace(off, length, result);
221 return off + std::size(result);
222 }
223 }
224
225 //Check previous index
226 if constexpr (N - 1 > 0)
227 return substitute_with_impl<N - 1>()(str, off, length,
228 std::forward<T>(values), index, format_str);
229 else
230 return off + length;
231 }
232 };

242 template <typename T>
243 inline auto substitute_curly_brace(std::string &str, size_t off, size_t length, T &&values)
244 {
245 if (length > 2)
246 {
247 auto item = std::string_view{std::data(str) + off + 1, length - 2};
248 auto index = convert::FirstPartTo<int>(item);
249
250 if (index && *index < std::tuple_size_v<T>)
251 {
252 auto format_delimiter = item.find(':');
253 return (format_delimiter != std::string::npos) ?
254 //Substitute with formatted tuple value
255 substitute_with(str, off, length,
256 std::forward<T>(values), *index, item.substr(format_delimiter + 1)) :
257 //Substitute with unformatted tuple value
258 substitute_with(str, off, length,
259 std::forward<T>(values), *index, {});
260 }
261 }
262
263 return off + length;
264 }
265
266 template <typename T>
267 inline auto &substitute_curly_braces(std::string &str, T &&values)
268 {
269 auto inside_brace = false;
270
271 for (size_t i = 0, off = 0; i < std::size(str);)
272 {
273 switch (str[i])
274 {
275 case '{':
276 {
277 //Final brace (no consecutive)
278 if (!(i + 1 < std::size(str) &&
279 str[i + 1] == str[i]))
280 {
281 inside_brace = true;
282 off = i;
283 }
284
285 break;
286 }
287 case '}':
288 {
289 //Substitute
290 if (inside_brace)
291 {
292 inside_brace = false;
293 i = substitute_curly_brace(str, off, i - off + 1, std::forward<T>(values));
294 continue;
295 }
296
297 break;
298 }
299 }
300
301 ++i;
302 }
303
304 return str;
305 }

563 [[nodiscard]] inline auto Serialize(const T &object) noexcept
564 {
565 static_assert(std::is_trivially_copyable_v<T>);
566
567 std::array<std::byte, sizeof(T)> bytes;
568 auto first = reinterpret_cast<const std::byte*>(std::addressof(object));
569 auto last = first + std::size(bytes);
570 std::copy(first, last, std::begin(bytes));
571
572 return bytes;
573 }
