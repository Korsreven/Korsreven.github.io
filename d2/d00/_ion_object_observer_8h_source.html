40 template <typename T,
41 typename = std::enable_if_t<observed_object::detail::has_owner_type_v<T>>>
42 class ObjectObserver final :
43 protected events::EventChannel<events::Listenable<events::listeners::listener_of_t<T, typename T::owner_type>>, typename T::owner_type>
44 {
45 static_assert(std::is_base_of_v<ManagedObject<typename T::owner_type>, T>);

50 object_observer::detail::container_type<T> managed_objects_;
51 std::optional<events::Callback<void, T&>> on_removed_;
52 std::optional<events::Callback<>> on_removed_all_;
53
54 void Tidy() noexcept
55 {
56 managed_objects_.clear();
57 managed_objects_.shrink_to_fit();
58 }
59
60 protected:
61
68 void ObjectRemoved(T &object) noexcept override
69 {
70 if (auto iter = managed_objects_.find(&object); iter != std::end(managed_objects_))
71 {
72 //Execute callback before object is erased
73 //The callback owner can then know in advance which object is going to be removed
74 if (on_removed_)
75 (*on_removed_)(object);
76
77 managed_objects_.erase(iter);
78
79 //Object erased
80 //Unsubscribe from publisher if empty
81 if (std::empty(managed_objects_))
82 {
83 this->DoUnsubscribe(true);
84 Tidy();
85 }
86 }
87 }
88
90 bool ObjectRemovable(T&) noexcept override
91 {
92 return this->Contract() == events::event_channel::SubscriptionContract::Cancelable;
93 }
94
96
98 void Unsubscribed() noexcept override
99 {
100 //Execute callback before all objects are erased
101 //The callback owner can then know in advance which object is going to be removed
102 if (on_removed_all_ && !std::empty(managed_objects_))
103 (*on_removed_all_)();
104
105 Tidy();
106 }

288 inline auto Observe(T &object)
289 {
290 //Is object observable
291 if (auto owner = object.Owner(); owner &&
292 ((std::empty(managed_objects_) && this->Subscribe(*owner)) ||
293 (!std::empty(managed_objects_) && (*std::begin(managed_objects_))->Owner() == owner)))
294 {
295 managed_objects_.insert(&object);
296 return true;
297 }
298 else
299 return false;
300 }
301
303
311 inline auto ReleaseAll() noexcept
312 {
313 if (this->Unsubscribe())
314 Unsubscribed();
315
316 return std::empty(managed_objects_);
317 }
318
321 inline auto Release(T &object) noexcept
322 {
323 if (ObjectRemovable(object) && ObjectRemoved(object))
324 {
325 //Object released
326 //Unsubscribe from publisher if empty
327 if (std::empty(managed_objects_))
328 ReleaseAll();
329
330 return true;
331 }
332 else
333 return false;
334 }
