57 struct default_shader_program
58 {
59 uint32 type_mask = 0_ui32;
60 NonOwningPtr<shaders::ShaderProgram> shader_program;
61 };
62
63 using default_shader_programs = std::vector<default_shader_program>;
64
65
66 inline auto get_default_shader_programs_by_type(uint32 type_flags, const default_shader_programs &def_shader_programs)
67 {
68 default_shader_programs matches;
69
70 for (auto &def_shader_program : def_shader_programs)
71 {
72 if (def_shader_program.type_mask & type_flags)
73 matches.push_back(def_shader_program);
74 }
75
76 return matches;
77 }
78 }
