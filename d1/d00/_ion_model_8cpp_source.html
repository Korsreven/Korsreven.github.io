31void add_emissive_mesh(shapes::Mesh &mesh, emissive_meshes &meshes)
32{
33 meshes.emplace_back(&mesh, create_emissive_light(mesh));
34}

36bool remove_emissive_mesh(const shapes::Mesh &mesh, emissive_meshes &meshes) noexcept
37{
38 if (auto iter = std::find_if(std::begin(meshes), std::end(meshes),
39 [&](auto &x) noexcept
40 {
41 return x.first == &mesh;
42 }); iter != std::end(meshes))
43 {
44 meshes.erase(iter);
45 return true;
46 }
47 else
48 return false;
49}

51bool is_mesh_emissive(const shapes::Mesh &mesh) noexcept
52{
53 if (auto material = mesh.RenderMaterial(); material)
54 return material->EmissiveColor().RGB() != color::Black.RGB() &&
55 material->EmissiveLightRadius().value_or(1.0_r) > 0.0_r;
56 else
57 return false;
58}

60
61OwningPtr<Light> create_emissive_light(const shapes::Mesh &mesh)
62{
63 if (auto material = mesh.RenderMaterial(); material)
64 {
65 auto &aabb = mesh.AxisAlignedBoundingBox();
66 auto sphere_radius = aabb.ToHalfSize().Max();
67 auto light_radius = material->EmissiveLightRadius().value_or(sphere_radius * 2.0_r);
68 //Increase default light radius by two times the sphere radius
69
70 return make_owning<Light>(Light::Point({}, aabb.Center(), light_radius, material->EmissiveColor()));
71 }
72 else
73 return make_owning<Light>();
74}

85void Model::Created(shapes::Mesh &mesh) noexcept
86{
87 update_bounding_volumes_ |= mesh.VertexCount() > 0;
88
89 if (detail::is_mesh_emissive(mesh))
90 {
91 detail::add_emissive_mesh(mesh, emissive_meshes_);
92 update_emissive_lights_ = true;
93 }
94
95 AddPrimitive(mesh);
96}
97
98void Model::Removed(shapes::Mesh &mesh) noexcept
99{
100 update_bounding_volumes_ |= mesh.IncludeBoundingVolumes();
101 update_emissive_lights_ |= detail::remove_emissive_mesh(mesh, emissive_meshes_);
102
103 RemovePrimitive(mesh);
104}

179void Model::Prepare()
180{
181 //Prepare all meshes
182 for (auto &mesh : Meshes())
183 mesh.Prepare();
184
185 if (update_bounding_volumes_)
186 {
187 aabb_ = {};
188
189 //Merge all bounding boxes
190 for (auto &mesh : Meshes())
191 {
192 if (mesh.IncludeBoundingVolumes())
193 aabb_.Merge(mesh.AxisAlignedBoundingBox());
194 }
195
196 obb_ = aabb_;
197 sphere_ = {aabb_.ToHalfSize().Max(), aabb_.Center()};
198
199 update_bounding_volumes_ = false;
200 }
201
202 if (update_emissive_lights_)
203 {
204 emissive_lights_.clear();
205
206 //Update all emissive lights
207 for (auto &[mesh, light] : emissive_meshes_)
208 {
209 light->ParentNode(ParentNode());
210 light->Prepare();
211 emissive_lights_.push_back(light.get());
212 }
213
214 update_emissive_lights_ = false;
215 }
216
217 DrawableObject::Prepare();
218}
