32std::pair<Vector2, Vector2> get_tex_coords(const Vector2 &lower_left, const Vector2 &upper_right,
33 const Vector2 &new_lower_left, const Vector2 &new_upper_right) noexcept
34{
35 auto [ll_s, ll_t] = lower_left.XY();
36 auto [ur_s, ur_t] = upper_right.XY();
37 auto [new_ll_s, new_ll_t] = new_lower_left.XY();
38 auto [new_ur_s, new_ur_t] = new_upper_right.XY();
39
40 if (is_flipped_horizontally(lower_left, upper_right))
41 std::swap(new_ll_s, new_ur_s);
42
43 if (is_flipped_vertically(lower_left, upper_right))
44 std::swap(new_ll_t, new_ur_t);
45
46 return {{new_ll_s, new_ll_t}, {new_ur_s, new_ur_t}};
47}

49std::pair<Vector2, Vector2> get_unflipped_tex_coords(const Vector2 &lower_left, const Vector2 &upper_right) noexcept
50{
51 auto [ll_s, ll_t] = lower_left.XY();
52 auto [ur_s, ur_t] = upper_right.XY();
53
54 if (is_flipped_horizontally(lower_left, upper_right))
55 std::swap(ll_s, ur_s);
56
57 if (is_flipped_vertically(lower_left, upper_right))
58 std::swap(ll_t, ur_t);
59
60 return {{ll_s, ll_t}, {ur_s, ur_t}};
61}

64Vector2 get_normalized_tex_coord(const Vector2 &tex_coord, const Vector2 &min, const Vector2 &max,
65 const Vector2 &new_min, const Vector2 &new_max) noexcept
66{
67 using namespace ion::utilities;
68
69 auto [s, t] = tex_coord.XY();
70 auto [min_s, min_t] = min.XY();
71 auto [max_s, max_t] = max.XY();
72 auto [new_min_s, new_min_t] = new_min.XY();
73 auto [new_max_s, new_max_t] = new_max.XY();
74
75 return {math::Normalize(s, min_s, max_s, new_min_s, new_max_s),
76 math::Normalize(t, min_t, max_t, new_min_t, new_max_t)};
77}

91std::pair<NonOwningPtr<Animation>, NonOwningPtr<Texture>> get_texture_maps(const texture_map_type &texture_map) noexcept
92{
93 using pair_type = std::pair<NonOwningPtr<Animation>, NonOwningPtr<Texture>>;
94
95 return std::visit(types::overloaded{
96 [](std::monostate) { return pair_type{nullptr, nullptr}; },
97 [](NonOwningPtr<Animation> animation) { return pair_type{animation, nullptr}; },
98 [](NonOwningPtr<Texture> texture) { return pair_type{nullptr, texture}; }
99 }, texture_map);
100}

102NonOwningPtr<Texture> get_texture_map(const texture_map_type &texture_map) noexcept
103{
104 if (auto [animation, texture] = get_texture_maps(texture_map); animation)
105 return animation->UnderlyingFrameSequence() ?
106 animation->UnderlyingFrameSequence()->FirstFrame() : nullptr;
107 else
108 return texture;

111NonOwningPtr<Texture> get_first_texture_map(const texture_map_type &diffuse_map, const texture_map_type &normal_map,
112 const texture_map_type &specular_map, const texture_map_type &emissive_map) noexcept
113{

121 return texture;

123 if (auto texture = get_texture_map(emissive_map); texture)
124 return texture;
125

130std::pair<bool, bool> is_texture_map_repeatable(const Texture &texture,
131 const Vector2 &lower_left, const Vector2 &upper_right) noexcept
132{
133 if (auto repeatable = texture.IsRepeatable(); repeatable)
134 {
135 auto &[s_repeatable, t_repeatable] = *repeatable;
136
137 auto [ll_s, ll_t] = lower_left.XY();
138 auto [ur_s, ur_t] = upper_right.XY();
139 return {s_repeatable && ll_s <= 0.0_r && ur_s >= 1.0_r,
140 t_repeatable && ll_t <= 0.0_r && ur_t >= 1.0_r};
141 }
142 else
143 return {false, false};
144}

344void Material::Crop(const std::optional<Aabb> &area) noexcept
345{
346 //Crop by area
347 if (area)
348 {
349 auto min = area->Min().CeilCopy(vector2::Zero).FloorCopy(vector2::UnitScale);
350 auto max = area->Max().CeilCopy(vector2::Zero).FloorCopy(vector2::UnitScale);
351
352 if (min != max)
353 {
354 auto [lower_left, upper_right] =
355 detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_, min, max);
356
357 lower_left_tex_coord_ = lower_left;
358 upper_right_tex_coord_ = upper_right;
359 }
360 }
361 //Un-crop
362 else if (IsCropped())
363 {
364 auto [lower_left, upper_right] =
365 detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_, 0.0_r, 1.0_r);
366
367 lower_left_tex_coord_ = lower_left;
368 upper_right_tex_coord_ = upper_right;
369 }
370}

372void Material::Repeat(const std::optional<Vector2> &amount) noexcept
373{
374 //Repeat by amount
375 if (amount)
376 {
377 if (auto max = amount->CeilCopy(vector2::Zero); vector2::Zero < max)
378 {
379 auto [lower_left, upper_right] =
380 detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_, 0.0_r, max);
381
382 lower_left_tex_coord_ = lower_left;

455std::pair<Vector2, Vector2> Material::WorldTexCoords() const noexcept
456{
457 if (auto texture = detail::get_first_texture_map(diffuse_map_, normal_map_, specular_map_, emissive_map_); texture)
458 {
459 //Use local tex coords relative to world tex coords
460 if (auto world_tex_coords = texture->TexCoords(); world_tex_coords)
461 {
462 auto [lower_left, upper_right] =
463 detail::get_unflipped_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_);
464 auto &[world_lower_left, world_upper_right] = *world_tex_coords;
465
466 auto [s_repeatable, t_repeatable] =
467 detail::is_texture_map_repeatable(*texture, lower_left, upper_right);
468
469 //Discard any repetition on s
470 if (!s_repeatable)
471 {
472 lower_left.X(std::max(lower_left.X(), 0.0_r));
473 upper_right.X(std::min(upper_right.X(), 1.0_r));
474 }
475
476 //Discard any repetition on t
477 if (!t_repeatable)
478 {
479 lower_left.Y(std::max(lower_left.Y(), 0.0_r));
480 upper_right.Y(std::min(upper_right.Y(), 1.0_r));
481 }
482
483 auto [norm_lower_left, norm_upper_right] =
484 detail::get_normalized_tex_coords(lower_left, upper_right, world_lower_left, world_upper_right);
485
486 return detail::get_tex_coords(lower_left_tex_coord_, upper_right_tex_coord_,
487 norm_lower_left, norm_upper_right);
488 }
489 }
490
491 //Use local tex coords
492 return {lower_left_tex_coord_, upper_right_tex_coord_};
493}
