51 inline auto cycle_duration_to_frame_duration(duration current_cycle_duration, duration total_cycle_duration, int frame_count, bool reverse) noexcept
52 {
53 auto frame_position = current_cycle_duration / total_cycle_duration * frame_count;
54 auto current_frame = static_cast<int>(frame_position);
55 auto frame_duration = total_cycle_duration / frame_count * (frame_position - current_frame);
56
57 if (reverse)
58 current_frame = frame_count - current_frame - 1;
59
60 return std::pair{frame_duration, current_frame};
61 }
62
63 inline auto frame_duration_to_cycle_duration(duration current_frame_duration, duration total_frame_duration, int current_frame, int frame_count, bool reverse) noexcept
64 {
65 if (reverse)
66 current_frame = frame_count - current_frame - 1;
67
68 return total_frame_duration * current_frame + current_frame_duration;
69 }
