26Vector2 hot_spot_offset(MouseCursorHotSpot hot_spot, const Vector2 &cursor_size) noexcept
27{
28 auto [half_width, half_height] = (cursor_size * 0.5_r).XY();
29
30 switch (hot_spot)
31 {
32 case MouseCursorHotSpot::TopLeft:
33 return {half_width, -half_height};
34
35 case MouseCursorHotSpot::TopCenter:
36 return {0.0_r, -half_height};
37
38 case MouseCursorHotSpot::TopRight:
39 return {-half_width, -half_height};
40
41 case MouseCursorHotSpot::Left:
42 return {half_width, 0.0_r};
43
44 case MouseCursorHotSpot::Right:
45 return {-half_width, 0.0_r};
46
47 case MouseCursorHotSpot::BottomLeft:
48 return {half_width, half_height};
49
50 case MouseCursorHotSpot::BottomCenter:
51 return {0.0_r, half_height};
52
53 case MouseCursorHotSpot::BottomRight:
54 return {-half_width, half_height};
55
56 default:
57 return vector2::Zero;
58 }
59}

103void GuiMouseCursor::UpdatePosition(Vector2 position) noexcept
104{
105 if (node_ && skin_)
106 {
107 auto cursor_size =
108 skin_->Parts->AxisAlignedBoundingBox().ToSize() * node_->DerivedScaling();
109
110 //Adjust from center to hot spot
111 auto hot_spot_off =
112 detail::hot_spot_offset(hot_spot_, cursor_size);
113 node_->DerivedPosition(position + hot_spot_off);
114 }
115}
