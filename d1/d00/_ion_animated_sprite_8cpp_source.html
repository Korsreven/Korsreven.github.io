93void AnimatedSprite::Revert()
94{
95 if (material_ && initial_material_)
96 {
97 if (diffuse_animation_ && initial_material_->DiffuseMap().first)
98 *diffuse_animation_ = *initial_material_->DiffuseMap().first;
99
100 if (specular_animation_ && initial_material_->SpecularMap().first)
101 *specular_animation_ = *initial_material_->SpecularMap().first;
102
103 if (normal_animation_ && initial_material_->NormalMap().first)
104 *normal_animation_ = *initial_material_->NormalMap().first;
105
106 *material_ = *initial_material_;
107 material_->DiffuseMap(diffuse_animation_);
108 material_->SpecularMap(specular_animation_);
109 material_->NormalMap(normal_animation_);
110 }
111}
