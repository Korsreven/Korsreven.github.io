27 template <typename KeyType, typename ValueType, typename Compare>
28 struct value_compare
29 {
30 Compare compare;
31
32 template <typename T1, typename T2>
33 constexpr auto operator()(const T1 &x, const T2 &y) const noexcept
34 {
35 if constexpr (std::is_same_v<KeyType, ValueType>) //Set
36 {
37 return compare(x, y);
38 }
39 else //Map
40 {
41 if constexpr (std::is_same_v<ValueType, T1> && std::is_same_v<ValueType, T2>)
42 return compare(x.first, y.first);
43 else if constexpr (std::is_same_v<ValueType, T1>)
44 return compare(x.first, y);
45 else if constexpr (std::is_same_v<ValueType, T2>)
46 return compare(x, y.first);
47 else
48 return compare(x, y);
49 }
50 }
51 };

57 return value; //Set
58 else
59 return value.first; //Map

75 template <typename Iterator, typename T, typename Compare>
76 inline auto find(Iterator first, Iterator last, const T &value, Compare compare) noexcept
77 {
78 first = std::lower_bound(first, last, value, compare);
79 return std::pair{first, first != last && !compare(value, *first)};

82 template <typename Iterator, typename T, typename Compare>
83 inline auto find_with_hint(Iterator hint, Iterator first, Iterator last, const T &value, Compare compare) noexcept
84 {
85 first =
86 [&]() noexcept
87 {
88 //No range
89 if (first == last)
90 return first;
91
92 //First
93 if (hint == first)
94 {
95 //Value is not greater than hint
96 if (!compare(*hint, value))
97 return hint;
98 }
99
100 //Last
101 else if (hint == last)
102 {
103 //Value is not less than hint - 1
104 if (!compare(value, *(hint - 1)))
105 return hint;
106 }
107
108 //Middle
109 //Value is not greater than hint
110 //Value is not less than hint - 1
111 else
112 {
113 auto left_of = !compare(*hint, value);
114 auto right_of = !compare(value, *(hint - 1));
115
116 if (left_of && right_of)
117 return hint;
118 else if (left_of)
119 return std::lower_bound(first, hint, value, compare);
120 else if (right_of)
121 return std::lower_bound(hint, last, value, compare);
122 }
123
124 return std::lower_bound(first, last, value, compare);
125 }();
126
127 return std::pair{first, first != last && !compare(value, *first)};
128 }

132 inline void sort_keys(Iterator first, Iterator last, Compare compare) noexcept

134 std::sort(first, last, compare);

137 template <typename Container, typename Iterator, typename Compare>
138 inline void erase_duplicate_keys(Container &container, Iterator first, Iterator last, Compare compare) noexcept
139 {
140 //Container must be sorted!
141 container.erase(
142 std::unique(first, last,
143 [&](const auto &x, const auto &y) noexcept
144 {
145 return !compare(x, y) && !compare(y, x);
146 }), std::end(container));
147 }
148
149 template <typename Container, typename Compare>
150 inline Container make_ordered_associative(Container container, Compare compare) noexcept
151 {
152 sort_keys(std::begin(container), std::end(container), compare);
153 erase_duplicate_keys(container, std::begin(container), std::end(container), compare);
154 return container;
155 }
156
157 template <typename Container, typename Compare>
158 inline void merge_added(Container &container, typename Container::iterator first, Compare compare)
159 {
160 //Sort added keys
161 sort_keys(first, std::end(container), compare);
162
163 //Container contains previously added entries
164 if (first != std::begin(container))
165 {
166 //Merge newly added entries with previous added entries
167 std::inplace_merge(
168 std::begin(container), first,
169 std::end(container), compare);
170 }
171
172 //Make sure there are no duplicate keys
173 erase_duplicate_keys(container, std::begin(container), std::end(container), compare);
174 }

313 template <typename T,
314 typename = std::enable_if_t<std::is_same_v<T, key_type> || types::is_transparent_comparator_v<Compare>>>
315 [[nodiscard]] inline auto count(const T &key) const noexcept
316 {
317 auto [iter, found] =
318 flat_associative_adaptor::detail::find(
319 std::begin(this->container_), std::end(this->container_),
320 key, value_compare{});
321 auto key_count = 0;
322
323 if (found)
324 {
325 ++key_count;
326
327 for (; ++iter != std::end(this->container_);)
328 {
329 if (!value_compare{}(*iter, key) && !value_compare{}(key, *iter))
330 ++key_count;
331 else
332 break;
333 }
334 }
335
336 return key_count;
337 }
