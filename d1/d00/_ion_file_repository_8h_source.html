64 template <typename T>
65 inline auto filter_by_file_extensions(utilities::file::Paths file_paths, const T &file_extensions) noexcept
66 {
67 //Something to filter on
68 if (!std::empty(file_extensions))
69 //Erase all files paths that does not match any files extensions
70 file_paths.erase(
71 std::remove_if(std::begin(file_paths), std::end(file_paths),
72 [&](const auto &file_path) noexcept
73 {
74 return !std::binary_search(
75 std::begin(file_extensions), std::end(file_extensions),
76 file_path.extension().generic_string());
77 }),
78 std::end(file_paths));
79
80 return file_paths;
81 }
