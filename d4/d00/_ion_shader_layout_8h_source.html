248 template <typename T>
249 auto make_variable_binding_map(T variable_bindings) noexcept
250 {
251 //Sort on location (asc)
252 std::stable_sort(std::begin(variable_bindings), std::end(variable_bindings),
253 [](auto &x, auto &y) noexcept
254 {
255 return x.second.Location() < y.second.Location();
256 });
257
258 //Remove duplicate locations
259 variable_bindings.erase(
260 std::unique(std::begin(variable_bindings), std::end(variable_bindings),
261 [](auto &x, auto &y) noexcept
262 {
263 return x.second.Location() && y.second.Location() &&
264 x.second.Location() == y.second.Location();
265 }), std::end(variable_bindings));
266
267
268 //Sort on name (asc)
269 std::stable_sort(std::begin(variable_bindings), std::end(variable_bindings),
270 [](auto &x, auto &y) noexcept
271 {
272 return x.second.Name() < y.second.Name();
273 });
274
275 //Remove duplicate names
276 variable_bindings.erase(
277 std::unique(std::begin(variable_bindings), std::end(variable_bindings),
278 [](auto &x, auto &y) noexcept
279 {
280 return x.second.Name() && y.second.Name() &&
281 x.second.Name() == y.second.Name();
282 }), std::end(variable_bindings));
283
284 return adaptors::FlatMap<typename T::value_type::first_type, typename T::value_type::second_type>{std::move(variable_bindings)};
285 }

290 template <typename T>
291 auto is_declaration_unique(const VariableDeclaration &declaration, const T &variable_bindings) noexcept
292 {
293 //Declaration has location
294 if (declaration.Location())
295 {
296 //No duplicate locations allowed
297 for (auto &binding : variable_bindings)
298 {
299 if (binding.second.Location() == declaration.Location())
300 return false;
301 }
302 }
303
304 //Declaration has name
305 if (declaration.Name())
306 {
307 //No duplicate names allowed
308 for (auto &binding : variable_bindings)
309 {
310 if (binding.second.Name() == declaration.Name())
311 return false;
312 }
313 }
314
315 return true;
316 }
