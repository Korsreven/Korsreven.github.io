23utilities::file::Paths filter_by_file_extension(utilities::file::Paths file_paths, std::string_view file_extension) noexcept
24{
25 //Erase all file paths that does not match file extension
26 file_paths.erase(
27 std::remove_if(std::begin(file_paths), std::end(file_paths),
28 [&](const auto &file_path) noexcept
29 {
30 return file_path.extension().generic_string() != file_extension;
31 }),
32 std::end(file_paths));
33
34 return file_paths;
35}

93std::optional<std::string> FileRepository::FileData(std::string_view name) const noexcept
94{
95 auto iter = files_.find(name);
96
97 if (iter != std::end(files_))
98 {
99 std::string data;
100
101 //File is stored in a data file
102 if (iter->second.DataFile)
103 {
104 //Load file from data file [position, count)
105 if (utilities::file::LoadPartOf(iter->second.DataFile->Path, data,
106 iter->second.DataFile->Position, iter->second.DataFile->Count,
107 ion::utilities::file::FileLoadMode::Binary))
108 return data;
109 }
110 //Load file from path
111 else if (utilities::file::Load(iter->second.Path, data, ion::utilities::file::FileLoadMode::Binary))
112 return data;
113 }
114
115 return {};
116}

141bool FileRepository::AddDirectory(const std::filesystem::path &directory_path,

151bool FileRepository::AddDirectories(const utilities::file::Paths &directory_paths,
152 utilities::file::DirectoryIteration directory_iteration)
153{
154 if (std::empty(directory_paths))
155 return false;
156
157 utilities::file::Paths file_paths;
158
159 for (auto &directory_path : directory_paths)
160 {
161 if (utilities::file::IsDirectory(directory_path))
162 {
163 auto files = utilities::file::FilesIn(directory_path, directory_iteration);
164 file_paths.insert(
165 std::end(file_paths),
166 std::make_move_iterator(std::begin(files)),
167 std::make_move_iterator(std::end(files)));
168 }
169 }
170
171 return AddFiles(detail::filter_by_file_extensions(file_paths, file_extensions_));
172}

197bool FileRepository::AddFiles(utilities::file::Paths file_paths)
198{
199 if (std::empty(file_paths))
200 return false;
201
202 decltype(files_)::container_type file_container;
203 file_container.reserve(std::size(file_paths));
204
205 for (auto &file_path : file_paths)
206 {
207 if (utilities::file::IsFile(file_path))
208 {
209 auto file = FileEntry{std::move(file_path), std::nullopt};
210 file_container.emplace_back(detail::file_path_to_name(file.Path, naming_convention_),
211 std::move(file));
212 }
213 }
214
215 return files_.insert(file_container);
216}

280bool FileRepository::RemoveDirectory(const std::filesystem::path &directory_path,
281 utilities::file::DirectoryIteration directory_iteration) noexcept
282{
283 if (!utilities::file::IsDirectory(directory_path))
284 return false;
285
286 auto count = std::size(files_);
287
288 files_.erase_if(
289 [&](const auto &x) noexcept
290 {
291 return std::filesystem::equivalent(x.second.Path.parent_path(), directory_path);
292 });
293
294 if (directory_iteration == utilities::file::DirectoryIteration::Recursive)
295 RemoveDirectories(utilities::file::DirectoriesIn(directory_path, directory_iteration),
296 utilities::file::DirectoryIteration::NonRecursive);
297
298 return count > std::size(files_);
299}
300
301bool FileRepository::RemoveDirectories(const utilities::file::Paths &directory_paths,
302 utilities::file::DirectoryIteration directory_iteration) noexcept
303{
304 auto count = std::size(files_);
305
306 for (auto &directory_path : directory_paths)
307 RemoveDirectory(directory_path, directory_iteration);
308
309 return count > std::size(files_);
