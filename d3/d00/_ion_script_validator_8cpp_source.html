51const ClassDefinition* find_inherited_class_definition_impl(const class_pointers &classes, const class_reference &class_ref)
52{
53 if (!std::empty(classes))
54 {
55 class_pointers more_classes;
56
57 for (auto &class_def : classes)
58 {
59 if (class_def->Name() == class_ref.Name)
60 return class_def; //A class definition found, stop searching
61
62 //Search inner classes of base
63 for (auto &inner_class : class_def->InnerClasses())
64 {
65 if (inner_class.HasDefinition() &&
66 inner_class.Definition().Name() == class_ref.Name)
67 {
68 if (inner_class.Definition().Name() == class_ref.Name)
69 return &inner_class.Definition(); //A class definition found, stop searching
70 }
71 }
72
73 //Search deeper in inheritance hierarchy
74 for (auto &base_class : class_def->BaseClasses())
75 {
76 if (base_class.HasDefinition())
77 more_classes.push_back(&base_class.Definition());
78 }
79 }
80
81 if (auto result = find_inherited_class_definition_impl(more_classes, class_ref); result)
82 return result;
83 }
84
85 return nullptr;
86}

88const ClassDefinition* find_class_definition_impl(bool &unwinding, bool inheriting,
89 const ClassDefinition &class_def, const class_reference &class_ref)
90{
91 //Class owner found
92 if (&class_def == class_ref.Owner)
93 {
94 if (class_def.Name() == class_ref.Name)
95 return &class_def; //Class definition found, stop searching
96
97 //Search inheritance hierarchy of class reference owner
98 if (auto result = find_inherited_class_definition(class_def, {class_ref.Name, class_def}); result)
99 return result; //Class definition found in inheritance hierarchy, stop searching
100 else
101 unwinding = true; //Continue searching for a class definition while unwinding the stack
102 }
103 else
104 {
105 if (!unwinding)
106 {
107 //Search base classes
108 for (auto &base_class : class_def.BaseClasses())
109 {
110 if (base_class.HasDefinition())
111 {
112 if (auto result = find_class_definition_impl(unwinding, inheriting = true, base_class.Definition(), class_ref); result)
113 return result;
114
115 else if (unwinding)
116 break;
117 }
118 }
119 }
120
121 if (!unwinding)
122 {
123 //Search inner classes
124 for (auto &class_decl : class_def.InnerClasses())
125 {
126 if (class_decl.HasDefinition())
127 {
128 if (auto result = find_class_definition_impl(unwinding, inheriting, class_decl.Definition(), class_ref); result)
129 return result;
130
131 else if (unwinding)
132 break;
133 }
134 }
135 }
136
137 if (unwinding && !inheriting)
138 {
139 //Search siblings
140 for (auto &sibling : class_def.InnerClasses())
141 {
142 if (sibling.HasDefinition() &&
143 sibling.Name() == class_ref.Name)
144 return &sibling.Definition(); //Class definition found, stop searching
145 }
146 }
147 }
148
149 return nullptr;
150}

152void find_all_declarations_impl(declarations_result &result, inhertied_class_pointers &inherited_classes,
153 const ClassDefinition &root, const ClassDefinition &class_def, int depth)
154{
155 if (auto [iter, inserted] = inherited_classes.insert(&class_def); !inserted)
156 return; //Do not inherit from the same class multiple times
157
158 //Add and group inner classes by name
159 for (auto &class_decl : class_def.InnerClasses())
160 result.InnerClasses[class_decl.Name()].push_back({&class_decl, &class_def, depth});
161
162 //Add and group properties by name
163 for (auto &property : class_def.Properties())
164 result.Properties[property.Name()].push_back({&property, &class_def, depth});
165
166
167 //Search further down the inheritance hierarchy
168 for (auto &base_class : class_def.BaseClasses())
169 {
170 auto base_class_def =
171 [&]() noexcept
172 {
173 if (base_class.HasDefinition())
174 return &base_class.Definition();
175 else
176 return find_class_definition(root, {base_class.Name(), class_def});
177 }();
178
179 if (base_class_def)
180 find_all_declarations_impl(result, inherited_classes, root, *base_class_def, depth + 1);
181 }
182}

185const ClassDefinition* find_inherited_class_definition(const ClassDefinition &class_def, const class_reference &class_ref)
186{
187 class_pointers classes;
188
189 for (auto &base_class : class_def.BaseClasses())
190 {
191 if (base_class.HasDefinition())
192 classes.push_back(&base_class.Definition());
193 }
194
195 return find_inherited_class_definition_impl(classes, class_ref);
196}

198const ClassDefinition* find_class_definition(const ClassDefinition &root, const class_reference &class_ref)
199{
200 auto unwinding = false;
201 return find_class_definition_impl(unwinding, false, root, class_ref);

204declarations_result find_all_declarations(const ClassDefinition &root, const ClassDefinition &class_def)
205{
206 declarations_result result;
207 inhertied_class_pointers inherited_classes;
208 find_all_declarations_impl(result, inherited_classes, root, class_def, 0);
209
210 //Sort inner classes by depth
211 for (auto &inner_class : result.InnerClasses)
212 std::sort(std::begin(inner_class.second), std::end(inner_class.second));
213
214 //Sort properties by depth
215 for (auto &property : result.Properties)
216 std::sort(std::begin(property.second), std::end(property.second));
217
218 return result;
219}

255adaptors::FlatSet<std::string_view> get_required_properties(const properties_map &properties)
256{
257 adaptors::FlatSet<std::string_view> required_properties;
258
259 for (auto &property : properties)
260 {
261 auto visible_depth = !std::empty(property.second) ?
262 property.second.front().Depth : 0;
263
264 for (auto &candidate : property.second)
265 {
266 if (candidate.Depth > visible_depth)
267 break; //Hide names deeper than visible depth
268
269 if (candidate.Declaration->Required())
270 required_properties.insert(candidate.Declaration->Name());
271 }
272 }
273
274 return required_properties;
275}
276
277adaptors::FlatSet<std::string_view> get_required_classes(const inner_classes_map &inner_classes)
278{
279 adaptors::FlatSet<std::string_view> required_classes;
280
281 for (auto &inner_class : inner_classes)
282 {
283 for (auto &candidate : inner_class.second)
284 {
285 if (candidate.Declaration->Required())
286 required_classes.insert(candidate.Declaration->Name());
287 }
288 }
289
290 return required_classes;
291}

294bool validate_property(const script_tree::PropertyNode &property, const property_declarations &overload_set)
295{
296 auto match = false;
297 auto visible_depth = !std::empty(overload_set) ?
298 overload_set.front().Depth : 0;
299
300 for (auto &candidate : overload_set)
301 {
302 if (candidate.Depth > visible_depth)
303 break; //Hide names deeper than visible depth
304
305 //Candidate has no definition
306 //Nothing more to validate
307 if (!candidate.Declaration->HasDefinition())
308 return true;
309
310 auto &candidate_def = candidate.Declaration->Definition();
311
312 //Check if candidate has correct number of arguments
313 if (std::ssize(property.Arguments()) >= candidate_def.RequiredParameters() &&
314 std::ssize(property.Arguments()) <= std::ssize(candidate_def.Parameters()))
315 {
316 match = true;
317 auto arg_iter = std::begin(property.Arguments());
318 auto param_iter = std::begin(candidate_def.Parameters());
319
320 //Check if each argument type is equal to each parameter type
321 for (; arg_iter != std::end(property.Arguments()); ++arg_iter, ++param_iter)
322 {
323 switch (param_iter->Type())
324 {
325 case ParameterType::Boolean:
326 {
327 match &= arg_iter->Visit(
328 [](const ScriptType::Boolean&) noexcept
329 {
330 return true;
331 },
332 [](auto&&) noexcept
333 {
334 return false;
335 });
336 break;
337 }
338
339 case ParameterType::Color:
340 {
341 match &= arg_iter->Visit(
342 [](const ScriptType::Color&) noexcept
343 {
344 return true;
345 },
346 [](auto&&) noexcept
347 {
348 return false;
349 });
350 break;
351 }
352
353 case ParameterType::Enumerable:
354 {
355 match &= arg_iter->Visit(
356 [&](const ScriptType::Enumerable &value) noexcept
357 {
358 return param_iter->HasValue(value.Get());
359 },
360 [](auto&&) noexcept
361 {
362 return false;
363 });
364 break;
365 }
366
367 case ParameterType::FloatingPoint:
368 {
369 match &= arg_iter->Visit(
370 [](const ScriptType::FloatingPoint&) noexcept
371 {
372 return true;
373 },
374 [](const ScriptType::Integer&) noexcept //Okay, non-narrowing
375 {
376 return true;
377 },
378 [](auto&&) noexcept
379 {
380 return false;
381 });
382 break;
383 }
384
385 case ParameterType::Integer:
386 {
387 match &= arg_iter->Visit(
388 [](const ScriptType::Integer&) noexcept
389 {
390 return true;
391 },
392 [](auto&&) noexcept
393 {
394 return false;
395 });
396 break;
397 }
398
399 case ParameterType::String:
400 {
401 match &= arg_iter->Visit(
402 [](const ScriptType::String&) noexcept
403 {
404 return true;
405 },
406 [](auto&&) noexcept
407 {
408 return false;
409 });
410 break;
411 }
412
413 case ParameterType::Vector2:
414 {
415 match &= arg_iter->Visit(
416 [](const ScriptType::Vector2&) noexcept
417 {
418 return true;
419 },
420 [](auto&&) noexcept
421 {
422 return false;
423 });
424 break;
425 }
426
427 case ParameterType::Vector3:
428 {
429 match &= arg_iter->Visit(
430 [](const ScriptType::Vector3&) noexcept
431 {
432 return true;
433 },
434 [](auto&&) noexcept
435 {
436 return false;
437 });
438 break;
439 }
440
441 default:
442 match = false;
443 break;
444 }
445
446 //An argument/parameter type mismatch found
447 //Skip candidate
448 if (!match)
449 break;
450 }
451
452 //A candidate match found
453 if (match)
454 break;
455 }
456 }
457
458 return match;
459}

461bool validate_properties(const ScriptTree &tree, const script_tree::ObjectNode &object, const ClassDefinition &class_def,
462 class_declarations_cacher &declarations_cacher, std::vector<ValidateError> &errors)
463{
464 auto &[inner_classes, properties] = declarations_cacher.Get(class_def);
465 auto required_properties = get_required_properties(properties);
466 ValidateError error;
467
468 for (auto &property : object.Properties())
469 {
470 if (auto iter = properties.find(property.Name());
471 iter != std::end(properties))
472 {
473 if (!validate_property(property, iter->second))
474 {
475 error = {ValidateErrorCode::InvalidPropertyArguments, *tree.GetFullyQualifiedName(object, property)};
476 errors.push_back(error);
477 }
478
479 required_properties.erase(property.Name());
480 }
481 else
482 {
483 error = {ValidateErrorCode::UnexpectedProperty, *tree.GetFullyQualifiedName(object, property)};
484 errors.push_back(error);
485 }
486 }
487
488 for (auto &required_property : required_properties)
489 {
490 error = {ValidateErrorCode::MissingRequiredProperty, *tree.GetFullyQualifiedName(object) + "." +
491 std::string{required_property}};
492 errors.push_back(error);
493 }
494
495 return !error;
496}

498const ClassDefinition* validate_class(const ScriptTree &tree, const script_tree::ObjectNode &object, const ClassDefinition &class_owner,
499 class_definition_cacher &definition_cacher, class_declarations_cacher &declarations_cacher, std::vector<ValidateError> &errors)
500{
501 auto &[inner_classes, properties] = declarations_cacher.Get(class_owner);
502 const inner_class_declarations::value_type *class_candidate = nullptr;
503 ValidateError error;
504
505 if (auto iter = inner_classes.find(object.Name());
506 iter != std::end(inner_classes))
507 {
508 //Check closest candidate with definition
509 for (auto &candidate : iter->second)
510 {
511 if (class_candidate &&
512 class_candidate->Depth < candidate.Depth)
513 break; //Unambiguous definition found
514
515 //Ambiguous definition found
516 if (class_candidate)
517 {
518 error = {ValidateErrorCode::AmbiguousClass, *tree.GetFullyQualifiedName(object)};
519 errors.push_back(error);
520 break;
521 }
522 //A candidate found
523 else
524 class_candidate = &candidate;
525 }
526 }
527
528 if (error)
529 return nullptr;
530 else if (!class_candidate)
531 {
532 error = {ValidateErrorCode::UnexpectedClass, *tree.GetFullyQualifiedName(object)};
533 errors.push_back(error);
534 return nullptr;
535 }
536 else if (!class_candidate->Declaration->Instantiatable())
537 {
538 error = {ValidateErrorCode::AbstractClassInstantiated, *tree.GetFullyQualifiedName(object)};
539 errors.push_back(error);
540 return nullptr;
541 }
542
543
544 auto class_def = class_candidate->Declaration->HasDefinition() ?
545 &class_candidate->Declaration->Definition() :
546 definition_cacher.Get({class_candidate->Declaration->Name(), *class_candidate->Owner});
547
548 //Class definition found, validate properties
549 if (class_def)
550 {
551 validate_properties(tree, object, *class_def, declarations_cacher, errors); //Ignoring return value
552 return class_def;
553 }
554 else
555 return nullptr;
556}

558bool validate(const ScriptTree &tree, const ClassDefinition &root, ValidateError &error, std::vector<ValidateError> &errors)
559{
560 //Start by validating the given tree
561 auto result = validate_tree(tree, root, errors);
562
563 //Set to first error
564 if (!std::empty(errors))
565 error = errors.front();
566
567 return result;
568}

570bool validate_tree(const ScriptTree &tree, const ClassDefinition &root, std::vector<ValidateError> &errors)
571{
572 //No top level (global) classes found
573 if (std::empty(root.InnerClasses()))
574 return true; //Nothing to validate against
575
576 struct scope
577 {
578 const script_tree::ObjectNode *object;
579 const ClassDefinition &class_def;
580 adaptors::FlatSet<std::string_view> required_classes;
581 };
582
583 class_definition_cacher definition_cacher{root};
584 class_declarations_cacher declarations_cacher{root};
585
586 auto &[root_classes, root_properties] = declarations_cacher.Get(root);
587 std::vector<scope> scopes{{nullptr, root, get_required_classes(root_classes)}};
588 auto next_search_depth = -1;
589 ValidateError error;
590
591 for (auto &[object, parent, depth] : tree.DepthFirstSearch())
592 {
593 if (next_search_depth >= 0)
594 {
595 //Next search depth reached
596 if (next_search_depth >= depth)
597 next_search_depth = -1;
598 //Skip object
599 else
600 continue;
601 }
602
603 //Pop back until depth is equal to scope size
604 while (depth + 1 < std::ssize(scopes))
605 {
606 const auto &scope = scopes.back();
607
608 for (auto &required_class : scope.required_classes)
609 {
610 error = {ValidateErrorCode::MissingRequiredClass, *tree.GetFullyQualifiedName(*scope.object) + "." +
611 std::string{required_class}};
612 errors.push_back(error);
613 }
614
615 scopes.pop_back();
616 }
617
618 if (auto class_def = validate_class(tree, object, scopes.back().class_def, definition_cacher, declarations_cacher, errors); class_def)
619 {
620 auto &[inner_classes, properties] = declarations_cacher.Get(*class_def);
621
622 scopes.back().required_classes.erase(class_def->Name());
623 scopes.push_back({&object, *class_def, get_required_classes(inner_classes)});
624 }
625 else
626 {
627 next_search_depth = depth; //Forward to next search depth
628 scopes.back().required_classes.erase(object.Name());
629 }
630 }
631
632 //Check if some required classes are missing
633 for (const auto &scope : adaptors::ranges::ReverseIterable<decltype(scopes)&>{scopes})
634 {
635 for (auto &required_class : scope.required_classes)
636 {
637 error = {ValidateErrorCode::MissingRequiredClass, (scope.object ? *tree.GetFullyQualifiedName(*scope.object) + "." : "") +
638 std::string{required_class}};
639 errors.push_back(error);
640 }
641 }
642
643 return std::empty(errors);
644}

650
651std::string print_output(duration validate_time, const std::vector<ValidateError> &errors, OutputOptions output_options)
652{
653 std::string output;
654 {
655 auto error = !std::empty(errors) ? errors.front() : ValidateError{};
656
657 //Find first error (if any)
658 if (!error)
659 {
660 if (auto iter = std::find_if(std::begin(errors), std::end(errors),
661 [](auto &error) noexcept
662 {
663 return !error;
664 }); iter != std::end(errors))
665 //Has error
666 error = *iter;
667 }
668
669 if (output_options == OutputOptions::Summary ||
670 output_options == OutputOptions::SummaryAndErrors)
671 {
672 if (!std::empty(output))
673 output += "\n\n";
674
675 output += ion::utilities::string::Concat(
676 "[Validator summary]\n"
677 "Message - ",
678 !error ?
679 "Validation succeeded!" :
680 ion::utilities::string::Concat("Validation failed. ", error.Condition.message(), " (", error.FullyQualifiedName, ")"),
681 "\n"
682 "Validate time - ", ion::utilities::string::Format(validate_time.count(), "0.0000"sv), " seconds\n"
683 "Validation errors - ", std::size(errors));
684 }
685 }
686
687 if (!std::empty(errors) &&
688 (output_options == OutputOptions::Errors ||
689 output_options == OutputOptions::SummaryAndErrors))
690 {
691 if (!std::empty(output))
692 output += "\n\n";
693
694 output += "[Validation errors]";
695
696 for (auto &error : errors)
697 output += ion::utilities::string::Concat(
698 "\n", "Error. ", error.Condition.message(), " (", error.FullyQualifiedName, ")");
699 }
700
701 return output;
702}
