117 template <typename VariantT, typename T, size_t Index = 0>
118 constexpr size_t variant_index() noexcept
119 {
120 static_assert(Index < std::variant_size_v<VariantT>);
121
122 if constexpr (std::is_same_v<std::variant_alternative_t<Index, VariantT>, T>)
123 return Index;
124 else
125 return variant_index<VariantT, T, Index + 1>();
126 }

129 template <typename T>
130 inline void serialize_value(const T &value, std::vector<std::byte> &bytes)
131 {
132 if constexpr (std::is_same_v<T, std::string>)
133 {
134 auto data_size = utilities::string::Serialize(std::size(value));
135 auto data = reinterpret_cast<const std::byte*>(std::data(value));
136 bytes.insert(std::end(bytes), std::begin(data_size), std::end(data_size));
137 bytes.insert(std::end(bytes), data, data + std::size(value));
138 }
139 else
140 {
141 auto data = utilities::string::Serialize(value);
142 bytes.insert(std::end(bytes), std::begin(data), std::end(data));
143 }
144 }

146 template <typename T>
147 inline void serialize_argument(const T &value, std::vector<std::byte> &bytes)
148 {
149 static_assert(is_contained_in_v<T, ArgumentType>);
150
151 bytes.push_back(static_cast<std::byte>(serialization_section_token::Argument));
152 bytes.push_back(static_cast<std::byte>(variant_index<ArgumentType, T>()));
153 serialize_value(value.Get(), bytes);
154 }

157 template <typename T>
158 inline auto deserialize_value(std::string_view bytes, T &value)
159 {
160 if constexpr (std::is_same_v<T, std::string>)
161 {
162 if (std::size(bytes) >= sizeof(size_t))
163 {
164 std::array<std::byte, sizeof(size_t)> output;
165 std::copy(std::begin(bytes), std::begin(bytes) + std::size(output), reinterpret_cast<char*>(std::data(output)));
166
167 size_t data_size;
168 utilities::string::Deserialize(output, data_size);
169
170 if (std::size(bytes) >= std::size(output) + data_size)
171 {
172 value = T{bytes.substr(std::size(output), data_size)};
173 return std::ssize(output) + static_cast<int>(data_size);
174 }
175 }
176 }
177 else
178 {
179 if (std::size(bytes) >= sizeof(T))
180 {
181 std::array<std::byte, sizeof(T)> output;
182 std::copy(std::begin(bytes), std::begin(bytes) + std::size(output), reinterpret_cast<char*>(std::data(output)));
183 utilities::string::Deserialize(output, value);
184 return std::ssize(output);
185 }
186 }
187
188 return 0;
189 }

191 template <typename T>
192 inline auto deserialize_argument(std::string_view bytes, ArgumentNodes &arguments)
193 {
194 typename T::value_type value;
195 auto bytes_deserialized = deserialize_value(bytes, value);
196
197 std::string unit;
198 auto unit_bytes_deserialized = bytes_deserialized > 0 ?
199 deserialize_value(bytes.substr(bytes_deserialized), unit) : 0;
200
201 if (unit_bytes_deserialized > 0)
202 arguments.emplace_back(T{value}, unit);
203
204 return bytes_deserialized + unit_bytes_deserialized;
205 }

223 template <typename DestNodes, typename SrcNodes>
224 void append_nodes(DestNodes &destination, const SrcNodes &source, AppendCondition append_condition)
225 {
226 auto count = std::size(destination); //Count, before appending
227
228 switch (append_condition)
229 {
230 case AppendCondition::Unconditionally:
231 {
232 destination.insert(std::end(destination), std::begin(source), std::end(source));
233 break;
234 }
235
236 case AppendCondition::NoDuplicateClasses:
237 {
238 if constexpr (std::is_same_v<std::remove_cvref_t<decltype(*std::begin(source))>, ObjectNode>)
239 {
240 for (auto &candidate : source)
241 {
242 if (auto iter = std::find_if(std::cbegin(destination), std::cbegin(destination) + count,
243 [&](auto &node) noexcept
244 {
245 return node.Name() == candidate.Name() && node.Classes() == candidate.Classes();
246 }); iter == std::cbegin(destination) + count)
247 //Candidate is approved
248 destination.push_back(candidate);
249 }
250 break;
251 }
252 else
253 [[fallthrough]];
254 }
255
256 case AppendCondition::NoDuplicateNames:
257 {
258 for (auto &candidate : source)
259 {
260 if (auto iter = std::find_if(std::cbegin(destination), std::cbegin(destination) + count,
261 [&](auto &node) noexcept
262 {
263 return node.Name() == candidate.Name();
264 }); iter == std::cbegin(destination) + count)
265 //Candidate is approved
266 destination.push_back(candidate);
267 }
268 break;
269 }
270 }
271 }

629 template <typename T>
630 [[nodiscard]] inline auto Get() const noexcept
631 {
632 auto value = *this ? std::get_if<T>(&*argument_) : nullptr;
633
634 if constexpr (std::is_same_v<T, ScriptType::FloatingPoint>)
635 {
636 if (!value)
637 {
638 //Try to get as integer
639 if (auto val = *this ? std::get_if<ScriptType::Integer>(&*argument_) : nullptr; val)
640 //Okay, non-narrowing
641 return std::make_optional(ScriptType::FloatingPoint{val->As<ScriptType::FloatingPoint::value_type>()});
642 }
643 }
644
645 return value ? std::make_optional(*value) : std::nullopt;
646 }
