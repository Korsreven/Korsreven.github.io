59render_primitive::VertexContainer vertices_to_vertex_data(const Vertices &vertices)
60{
61 render_primitive::VertexContainer vertex_data;
62 vertex_data.reserve(std::ssize(vertices) * vertex_components);
63
64 for (auto &vertex : vertices)
65 {
66 vertex_data.insert(std::end(vertex_data), vertex.Position.Components(), vertex.Position.Components() + position_components);
67 vertex_data.insert(std::end(vertex_data), vertex.Normal.Components(), vertex.Normal.Components() + normal_components);
68 vertex_data.insert(std::end(vertex_data), vertex.BaseColor.Channels(), vertex.BaseColor.Channels() + color_components);
69 vertex_data.insert(std::end(vertex_data), vertex.TexCoord.Components(), vertex.TexCoord.Components() + tex_coord_components);
70 }
71
72 return vertex_data;
73}

76void generate_tex_coords(render_primitive::VertexContainer &vertex_data, const Aabb &aabb) noexcept
77{
78 //Generate each vertex tex coords (s,t) from position (x,y) in range [0, 1]
79 for (auto i = detail::position_offset, j = detail::tex_coord_offset; i < std::ssize(vertex_data);
80 i += detail::vertex_components, j += detail::vertex_components)
81 {
82 auto tex_coord =
83 materials::material::detail::get_normalized_tex_coord(
84 {vertex_data[i], vertex_data[i + 1]},
85 aabb.Min(), aabb.Max(), vector2::Zero, vector2::UnitScale);
86
87 auto [s, t] = tex_coord.XY();
88 vertex_data[j] = s;
89 vertex_data[j + 1] = t;
90 }
91}

93void normalize_tex_coords(render_primitive::VertexContainer &vertex_data, const materials::Material *material) noexcept
94{
95 if (auto [s_repeatable, t_repeatable] = material ?
96 material->IsRepeatable() : std::pair{false, false}; !s_repeatable || !t_repeatable)
97 {
98 //Clamp each vertex tex coords (s,t) to range [0, 1]
99 for (auto i = detail::tex_coord_offset; i < std::ssize(vertex_data);
100 i += detail::vertex_components)
101 {
102 if (!s_repeatable)
103 vertex_data[i] = std::clamp(vertex_data[i], 0.0_r, 1.0_r);
104
105 if (!t_repeatable)
106 vertex_data[i + 1] = std::clamp(vertex_data[i + 1], 0.0_r, 1.0_r);
107 }
108 }
109
110
111 auto [world_lower_left_tex_coord, world_upper_right_tex_coord] = material ?
112 material->WorldTexCoords() :
113 std::pair{vector2::Zero, vector2::UnitScale};
114 auto [world_lower_left, world_upper_right] =
115 materials::material::detail::get_unflipped_tex_coords(world_lower_left_tex_coord, world_upper_right_tex_coord);
116
117 auto [mid_s, mid_t] = world_lower_left.Midpoint(world_upper_right).XY();
118 auto flip_s = materials::material::detail::is_flipped_horizontally(world_lower_left_tex_coord, world_upper_right_tex_coord);
119 auto flip_t = materials::material::detail::is_flipped_vertically(world_lower_left_tex_coord, world_upper_right_tex_coord);
120
121 //Normalize each vertex tex coords (s,t)
122 for (auto i = detail::tex_coord_offset; i < std::ssize(vertex_data);
123 i += detail::vertex_components)
124 {
125 auto norm_tex_coord =
126 materials::material::detail::get_normalized_tex_coord(
127 {vertex_data[i], vertex_data[i + 1]},
128 vector2::Zero, vector2::UnitScale,
129 world_lower_left, world_upper_right);
130
131 auto [s, t] = norm_tex_coord.XY();
132
133 //Make sure mesh texture is flipped the same way as material texture
134 if (flip_s)
135 s = 2.0_r * mid_s - s; //Reflect s across middle point
136 if (flip_t)
137 t = 2.0_r * mid_t - t; //Reflect t across middle point
138
139 vertex_data[i] = s;
140 vertex_data[i + 1] = t;
141 }
142}

245void Mesh::Prepare()
246{
247 if (update_tex_coords_)
248 {
249 //Auto generate tex coords
250 if (tex_coord_mode_ == mesh::MeshTexCoordMode::Auto)
251 detail::generate_tex_coords(VertexData(), AxisAlignedBoundingBox());
252
253 //Normalize tex coords
254 if (tex_coord_mode_ == mesh::MeshTexCoordMode::Manual || RenderMaterial())
255 detail::normalize_tex_coords(VertexData(), RenderMaterial().get());
256
257 update_tex_coords_ = false;
258 }
259
260 RenderPrimitive::Prepare();
261}
