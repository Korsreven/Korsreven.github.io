208 constexpr auto is_alpha(char c) noexcept
209 {
210 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z');
211 }
212
213 constexpr auto is_white_space(char c) noexcept
214 {
215 //Unroll all true cases
216 switch (c)
217 {
218 case ' ':
219 case '\n': //Line feed
220 case '\r': //Carriage return
221 case '\t': //Horizontal tab
222 case '\v': //Vertical tab
223 return true;
224
225 default:
226 return false;
227 }
228 }
229
230 constexpr auto is_valid_in_identifier(char c)
231 {
232 if (!is_alpha(c) &&
233 !ion::utilities::convert::detail::is_digit(c))
234 {
235 //Unroll all true cases
236 switch (c)
237 {
238 case '-': //Hyphen
239 case '_': //Underscore
240 return true;
241
242 default:
243 return false;
244 }
245 }
246
247 return true;
248 }

250 constexpr auto is_class_selector(char c) noexcept
251 {
252 //Unroll all true cases
253 switch (c)
254 {
255 case '.': //Class
256 case '#': //Id
257 case '*': //Both
258 return true;
259
260 default:
261 return false;
262 }
263 }
264
265 constexpr auto is_combinator_selector(char c) noexcept
266 {
267 //Unroll all true cases
268 switch (c)
269 {
270 case ',': //Both
271 case '>': //Child
272 case '+': //Adjacent sibling
273 case '~': //General sibling
274 return true;
275
276 default:
277 return false;
278 }
279 }

412 constexpr auto is_start_of_comment(char c, char next_c) noexcept
413 {
414 if (c == '/')
415 {
416 //Unroll all true cases
417 switch (next_c)
418 {
419 case '/':
420 case '*':
421 return true;
422 }
423 }
424
425 return false;

428 constexpr auto is_start_of_hex_literal(char c, char next_c) noexcept
429 {
430 return c == '#' && is_hex_digit(next_c);
431 }
432
433 constexpr auto is_start_of_identifier(char c, char next_c = '\0') noexcept
434 {
435 //A digit
436 if (ion::utilities::convert::detail::is_digit(c) ||
437 //Two hyphens
438 (c == '-' && (next_c == '-' ||
439 //Hyphen followed by a non valid identifier or digit
440 !is_valid_in_identifier(next_c) ||
441 ion::utilities::convert::detail::is_digit(next_c))))
442 return false;
443 else
444 return is_valid_in_identifier(c);
445 }

447 constexpr auto is_start_of_numeric_literal(char c, char next_c = '\0') noexcept
448 {
449 return ion::utilities::convert::detail::is_digit(c) ||
450 (c == '.' && ion::utilities::convert::detail::is_digit(next_c));
451 }
