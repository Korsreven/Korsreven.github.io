52Aabb MovableObject::DeriveWorldAxisAlignedBoundingBox(Aabb aabb, bool apply_extent) const noexcept
53{
54 if (parent_node_ && !aabb.Empty())
55 {
56 if (apply_extent)
57 {
58 if (auto [min, max] = bounding_volume_extent_.MinMax();
59 min != vector2::Zero || max != vector2::UnitScale)
60 {
61 auto size = aabb.ToSize();
62 aabb = Aabb{aabb.Min() + min * size,
63 aabb.Max() + (max - vector2::UnitScale) * size};
64 }
65 }
66
67 aabb.Transform(Matrix3::Transformation(parent_node_->FullTransformation()));
68 }
69
70 return aabb;
71}

95Sphere MovableObject::DeriveWorldBoundingSphere(Sphere sphere, Aabb aabb, bool apply_extent) const noexcept
96{
97 if (parent_node_ && !sphere.Empty())
98 {
99 if (apply_extent)
100 {
101 if (auto [min, max] = bounding_volume_extent_.MinMax();
102 min != vector2::Zero || max != vector2::UnitScale)
103 {
104 auto size = aabb.ToSize();
105 aabb = Aabb{aabb.Min() + min * size,
106 aabb.Max() + (max - vector2::UnitScale) * size};
107 sphere = {aabb.ToHalfSize().Max(), aabb.Center()};
108 }
109 }
110
111 sphere.Transform(Matrix3::Transformation(parent_node_->FullTransformation()));
112 }
113
114 return sphere;
115}

118void MovableObject::DrawBoundingVolumes(const Aabb &aabb, const Obb &obb, const Sphere &sphere,
119 const Color &aabb_color, const Color &obb_color, const Color &sphere_color, real z) const noexcept
120{
121 if (parent_node_)
122 {
123 graph::scene_graph::detail::push_gl_matrix(); //New view matrix
124 graph::scene_graph::detail::mult_gl_model_view_matrix(
125 Matrix4::Translation({0.0_r, 0.0_r, z})); //view * z translation matrix
126 }
127
128
129 //Enable fixed-function pipeline
130 auto active_program = shaders::shader_program_manager::detail::get_active_shader_program();
131 if (active_program > 0)
132 shaders::shader_program_manager::detail::use_shader_program(0);
133
134 //Draw bounding sphere
135 if (!sphere.Empty() && sphere_color != color::Transparent)
136 sphere.Draw(sphere_color);
137
138 //Draw oriented bounding box
139 if (!obb.Empty() && obb_color != color::Transparent)
140 obb.Draw(obb_color);
141
142 //Draw axis-aligned bounding box
143 if (!aabb.Empty() && aabb_color != color::Transparent)
144 aabb.Draw(aabb_color);
145
146 //Re-enable active shader program (if any)
147 if (active_program > 0)
148 shaders::shader_program_manager::detail::use_shader_program(active_program);
149
150
151 if (parent_node_)
152 graph::scene_graph::detail::pop_gl_matrix(); //Restore
153}
