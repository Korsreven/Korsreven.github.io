31void GuiButton::ExecuteActions() noexcept
32{
33 auto owner = Owner();
34 if (!owner)
35 return;
36
37 auto parent_frame = owner->ParentFrame();
38 if (!parent_frame)
39 return;
40
41 auto controller = parent_frame->Owner();
42 if (!controller)
43 return;
44
45
46 //Execute actions
47 for (auto &[type, name] : actions_)
48 {
49 auto active_frame = controller->FocusedFrame();
50 if (!active_frame)
51 active_frame = parent_frame;
52
53
54 switch (type)
55 {
56 //Controller actions
57 case ButtonActionType::ShowGui:
58 controller->Show();
59 break;
60
61 case ButtonActionType::HideGui:
62 controller->Hide();
63 break;
64
65 case ButtonActionType::EnableGui:
66 controller->Enable();
67 break;
68
69 case ButtonActionType::DisableGui:
70 controller->Disable();
71 break;
72
73
74 //Frame actions
75 case ButtonActionType::ShowFrame:
76 if (auto frame = controller->GetFrame(name); frame)
77 frame->Show();
78 break;
79
80 case ButtonActionType::ShowFrameModal:
81 if (auto frame = controller->GetFrame(name); frame)
82 frame->Show(gui_frame::FrameMode::Modal);
83 break;
84
85 case ButtonActionType::HideFrame:
86 if (auto frame = controller->GetFrame(name); frame)
87 frame->Hide();
88 break;
89
90 case ButtonActionType::EnableFrame:
91 if (auto frame = controller->GetFrame(name); frame)
92 frame->Enable();
93 break;
94
95 case ButtonActionType::DisableFrame:
96 if (auto frame = controller->GetFrame(name); frame)
97 frame->Disable();
98 break;
99
100 case ButtonActionType::FocusFrame:
101 if (auto frame = controller->GetFrame(name); frame)
102 frame->Focus();
103 break;
104
105 case ButtonActionType::DefocusFrame:
106 if (auto frame = controller->GetFrame(name); frame)
107 frame->Defocus();
108 break;
109
110
111 //Panel actions
112 case ButtonActionType::ShowPanel:
113 if (auto panel = active_frame->SearchPanel(name); panel)
114 panel->Show();
115 break;
116
117 case ButtonActionType::HidePanel:
118 if (auto panel = active_frame->SearchPanel(name); panel)
119 panel->Hide();
120 break;
121
122 case ButtonActionType::EnablePanel:
123 if (auto panel = active_frame->SearchPanel(name); panel)
124 panel->Enable();
125 break;
126
127 case ButtonActionType::DisablePanel:
128 if (auto panel = active_frame->SearchPanel(name); panel)
129 panel->Disable();
130 break;
131
132
133 //Control actions
134 case ButtonActionType::ShowControl:
135 if (auto control = active_frame->SearchControl(name); control)
136 control->Show();
137 break;
138
139 case ButtonActionType::HideControl:
140 if (auto control = active_frame->SearchControl(name); control)
141 control->Hide();
142 break;
143
144 case ButtonActionType::EnableControl:
145 if (auto control = active_frame->SearchControl(name); control)
146 control->Enable();
147 break;
148
149 case ButtonActionType::DisableControl:
150 if (auto control = active_frame->SearchControl(name); control)
151 control->Disable();
152 break;
153
154 case ButtonActionType::FocusControl:
155 if (auto control = active_frame->SearchControl(name); control)
156 control->Focus();
157 break;
158
159 case ButtonActionType::DefocusControl:
160 if (auto control = active_frame->SearchControl(name); control)
161 control->Defocus();
162 break;
163 }
164 }
165}

187OwningPtr<gui_control::ControlSkin> GuiButton::AttuneSkin(OwningPtr<gui_control::ControlSkin> skin) const
188{
189 //Not fully compatible
190 if (skin && !dynamic_cast<ButtonSkin*>(skin.get()))
191 {
192 auto button_skin = make_owning<ButtonSkin>();
193 button_skin->Assign(*skin);
194 return button_skin;
195 }
196 else
197 return skin;
