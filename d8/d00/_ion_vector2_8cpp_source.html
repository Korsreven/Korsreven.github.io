93real Vector2::SignedAngleBetween(const Vector2 &vector) const noexcept
94{
95 #ifdef ION_LEFT_HANDED
96 //Left-hand rotation CW

129 Deviant
130*/
131
132Vector2 Vector2::Deviant(real angle) const noexcept
133{
134 auto sin_of_angle = math::Sin(angle);
135 auto cos_of_angle = math::Cos(angle);
136
137 #ifdef ION_LEFT_HANDED
138 //Left-hand rotation CW
139 return {cos_of_angle * x_ + sin_of_angle * y_,
140 -sin_of_angle * x_ + cos_of_angle * y_};
141 #else
142 //Right-hand rotation CCW
143 return {cos_of_angle * x_ - sin_of_angle * y_,
144 sin_of_angle * x_ + cos_of_angle * y_};
145 #endif
146}

148Vector2 Vector2::RandomDeviant(real angle) const noexcept

204
205Vector2 Vector2::Lerp(const Vector2 &vector, real amount) const noexcept
206{
207 if (amount <= 0.0_r)
208 return *this;
209 else if (amount >= 1.0_r)
210 return vector;
211
212 //Do linear interpolation
213 return {math::Lerp(x_, vector.x_, amount),
214 math::Lerp(y_, vector.y_, amount)};
215}
216
217Vector2 Vector2::Slerp(Vector2 vector, real amount) const noexcept
218{
219 if (amount <= 0.0_r)
220 return *this;
221 else if (amount >= 1.0_r)
222 return vector;
223
224 //Do spherical linear interpolation
225 auto dot = DotProduct(vector);
226
227 if (dot < 0.0_r)
228 {
229 dot = -dot;
230 vector = -vector;
231 }
232
233 //Fixed potential acos domain error and division by zero
234 if (dot > 1.0_r - math::Epsilon)
235 return Lerp(vector, amount);
236 else
237 {
238 auto angle = std::acos(dot);
239 return {math::Slerp(x_, vector.x_, angle, amount),
240 math::Slerp(y_, vector.y_, angle, amount)};
241 }
242}
