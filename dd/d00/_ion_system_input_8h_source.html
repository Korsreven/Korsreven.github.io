233 template <typename T>
234 inline auto get_mapped_code(const T &code_map, CodeType code) noexcept
235 {
236 auto iter = std::lower_bound(std::begin(code_map), std::end(code_map), code,
237 [](const typename T::value_type &x, CodeType y) noexcept
238 {
239 return x.first < y;
240 });
241
242 return iter != std::end(code_map) &&
243 iter->first == code ?
244
245 std::make_optional(iter->second) :
246 std::nullopt;
247 }
248 } //detail
249

257 [[nodiscard]] inline auto GetMappedKeyButton(CodeType code) noexcept
258 {
259 return detail::get_mapped_code(detail::key_button_map, code);
260 }

263 [[nodiscard]] inline auto GetMappedMouseButton(CodeType code) noexcept
264 {

266 }
