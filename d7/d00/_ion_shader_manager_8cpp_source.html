26std::optional<int> load_shader(shader::ShaderType shader_type, const std::string &shader_source,
27 const std::filesystem::path &shader_source_path, std::optional<InfoLogLevel> log_level) noexcept
28{
29 auto shader_handle = 0;
30 auto compile_status = 0;
31
32 //Compile shader
33 switch (gl::Shader_Support())
34 {
35 case gl::Extension::Core:
36 {
37 switch (shader_type)
38 {
39 case shader::ShaderType::Vertex:
40 shader_handle = GL_VERTEX_SHADER;
41 break;
42
43 case shader::ShaderType::Fragment:
44 shader_handle = GL_FRAGMENT_SHADER;
45 break;
46
47 default:
48 return {};
49 }
50
51 auto source_ptr = std::data(shader_source);
52 shader_handle = glCreateShader(shader_handle);
53 glShaderSource(shader_handle, 1, &source_ptr, 0); //Set GLSL code
54
55 glCompileShader(shader_handle); //Compile GLSL
56 glGetShaderiv(shader_handle, GL_COMPILE_STATUS, &compile_status);
57 break;
58 }
59
60 case gl::Extension::ARB:
61 {
62 switch (shader_type)
63 {
64 case shader::ShaderType::Vertex:
65 shader_handle = GL_VERTEX_SHADER_ARB;
66 break;
67
68 case shader::ShaderType::Fragment:
69 shader_handle = GL_FRAGMENT_SHADER_ARB;
70 break;
71
72 default:
73 return {};
74 }
75
76 auto source_ptr = std::data(shader_source);
77 shader_handle = glCreateShaderObjectARB(shader_handle);
78 glShaderSourceARB(shader_handle, 1, &source_ptr, 0); //Set GLSL code
79
80 glCompileShaderARB(shader_handle); //Compile GLSL
81 glGetObjectParameterivARB(shader_handle, GL_OBJECT_COMPILE_STATUS_ARB, &compile_status);
82 break;
83 }
84
85 default:
86 return {};
87 }
88
89 //Save shader info log
90 if (log_level &&
91 (compile_status == 0 || *log_level == InfoLogLevel::Warning))
92 //Compile error, or log anyways (could be warnings)
93 {
94 if (auto log = print_info_log(shader_handle); log)
95 ion::utilities::file::Save(shader_source_path.string() + ".output.txt", *log);
96 }
97
98 //Shader did not compile properly, unload
99 if (compile_status == 0)
100 {
101 unload_shader(shader_handle);
102 return {};
103 }
104 else
105 return shader_handle;
106}

108void unload_shader(int shader_handle) noexcept
109{
110 switch (gl::Shader_Support())
111 {
112 case gl::Extension::Core:
113 glDeleteShader(shader_handle);
114 break;
115
116 case gl::Extension::ARB:
117 glDeleteObjectARB(shader_handle);
118 break;
119 }
120}

123std::optional<std::string> print_info_log(int shader_handle)
124{
125 auto length = 0;
126
127 //Retrieve log length from shader object
128 switch (gl::Shader_Support())
129 {
130 case gl::Extension::Core:
131 glGetShaderiv(shader_handle, GL_INFO_LOG_LENGTH, &length);
132 break;
133
134 case gl::Extension::ARB:
135 glGetObjectParameterivARB(shader_handle, GL_OBJECT_INFO_LOG_LENGTH_ARB, &length);
136 break;
137 }
138
139 //Retrieve log from shader object
140 if (length > 0)
141 {
142 auto written = 0;
143 std::string log(length, '\0');
144
145 switch (gl::Shader_Support())
146 {
147 case gl::Extension::Core:
148 glGetShaderInfoLog(shader_handle, length, &written, &log[0]);
149 break;
150
151 case gl::Extension::ARB:
152 glGetInfoLogARB(shader_handle, length, &written, &log[0]);
153 break;
154 }
155
156 return std::move(log);
157 }
158 else
159 return {};
160}

171bool ShaderManager::LoadResource(Shader &shader)
172{
173 auto type = shader.Type();
174 auto &source = shader.FileData();
175 auto &source_path = shader.FilePath();
176
177 if (type && source && source_path)
178 {
179 shader.Handle(detail::load_shader(*type, *source, *source_path, log_level_));
180 return shader.Handle().has_value();
181 }
182 else
183 return false;
184}

186bool ShaderManager::UnloadResource(Shader &shader) noexcept
187{
188 if (auto handle = shader.Handle(); handle)
189 {
190 detail::unload_shader(*handle);
191 shader.Handle({});
192 return true;
193 }
194 else
195 return false;
196}
