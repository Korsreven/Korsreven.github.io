45ParticleSystem ParticleSystem::Clone() const
46{
47 auto particle_system = ParticleSystem{*name_};
48 particle_system.particle_primitive_ = particle_primitive_;
49
50 //Clone emitters
51 for (auto &emitter : Emitters())
52 particle_system.EmitterManager::Adopt(emitter.Clone());
53
54 //Clone global affectors
55 for (auto &affector : Affectors())
56 particle_system.AffectorManager::Adopt(affector.Clone());
57
58 return particle_system;
59}

95void ParticleSystem::Elapse(duration time) noexcept
96{
97 //Elapse emitters
98 for (auto &emitter : Emitters())
99 {
100 emitter.Elapse(time);
101
102 //Affect particles (in emitter)
103 for (auto &affector : Affectors())
104 affector.Affect(emitter.Particles(), time);
105 }
