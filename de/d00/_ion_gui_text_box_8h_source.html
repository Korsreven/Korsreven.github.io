308 inline void Content(std::string content) noexcept
309 {
310 content = gui_text_box::detail::trim_content(std::move(content), text_mode_, character_set_);
311
312 if (max_characters_)
313 content = gui_text_box::detail::truncate_content(std::move(content), *max_characters_);
314
315 if (content_ != content)
316 {
317 content_ = content;
318 CursorPosition(std::ssize(content));
319 UpdateText();
320 Changed();
321 }
322 }

325 inline void PlaceholderContent(std::optional<std::string> content) noexcept

335 UpdateText();
336 }

340 inline void MaxCharacters(std::optional<int> max) noexcept
341 {
342 if (max && *max < 0)
343 max = 0;
344

347 max_characters_ = max;

351 content_ = gui_text_box::detail::truncate_content(std::move(content_), *max);
352 CursorPosition(cursor_position_);
353 UpdateText();
354 Changed();
355 }

440 inline void CursorPosition(int position) noexcept
441 {
442 position = std::clamp(position, 0, std::ssize(content_));
443
444 if (cursor_position_ != position)

446 cursor_position_ = position;
447
448 //Out of view

451 UpdateText();

453 UpdateCursor();
