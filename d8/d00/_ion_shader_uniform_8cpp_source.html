52bool UniformVariable::HasNewValue() noexcept
53{
54 auto changed =
55 current_value_ ?
56 Visit(
57 [&](auto &&value) noexcept
58 {
59 return uniform_variable::detail::is_value_different(
60 value,
61 std::get<std::remove_cvref_t<decltype(value)>>(*current_value_));
62 }) :
63 true;
64
65 if (changed)
66 current_value_ = value_;
67
68 return changed;
69}
