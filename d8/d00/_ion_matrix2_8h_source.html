195 [[nodiscard]] inline auto operator*(const Matrix2 &rhs) const noexcept
196 {
197 #ifdef ION_ROW_MAJOR
198 //Row-major layout (Direct3D)
199 return Matrix2{m_[0][0] * rhs.m_[0][0] + m_[1][0] * rhs.m_[0][1],
200 m_[0][1] * rhs.m_[0][0] + m_[1][1] * rhs.m_[0][1],
201
202 m_[0][0] * rhs.m_[1][0] + m_[1][0] * rhs.m_[1][1],
203 m_[0][1] * rhs.m_[1][0] + m_[1][1] * rhs.m_[1][1]};
204 #else
205 //Column-major layout (OpenGL)
206 return Matrix2{m_[0][0] * rhs.m_[0][0] + m_[0][1] * rhs.m_[1][0],
207 m_[0][0] * rhs.m_[0][1] + m_[0][1] * rhs.m_[1][1],
208
209 m_[1][0] * rhs.m_[0][0] + m_[1][1] * rhs.m_[1][0],
210 m_[1][0] * rhs.m_[0][1] + m_[1][1] * rhs.m_[1][1]};
211 #endif
