36void DrawableObject::AddPrimitive(render::RenderPrimitive &primitive)
37{
38 if (auto renderer = ParentRenderer(); renderer)
39 {
40 if (std::empty(primitive.RenderPasses()))
41 primitive.RenderPasses(render_passes_);
42
43 primitive.Opacity(opacity_);
44
45 renderer->AddPrimitive(primitive);
46 render_primitives_.push_back(&primitive);
47 }
48}

50void DrawableObject::RemovePrimitive(render::RenderPrimitive &primitive) noexcept
51{
52 if (auto iter = std::find(std::begin(render_primitives_), std::end(render_primitives_), &primitive);
53 iter != std::end(render_primitives_))
54 render_primitives_.erase(iter);
55}

123void DrawableObject::Prepare()
124{
125 //No render passes added, add render pass with default shader program (if any)
126 if (std::empty(render_passes_))
127 {
128 auto default_shader_program =
129 [&]() noexcept -> NonOwningPtr<shaders::ShaderProgram>
130 {
131 if (auto owner = Owner(); owner)
132 return owner->GetDefaultShaderProgram(query_type_flags_);
133 else
134 return nullptr;
135 }();
136
137 AddRenderPass(render::RenderPass{default_shader_program});
138 }
139
140 if (update_render_passes_)
141 {
142 shader_programs_.clear();
143
144 //Update all shader programs
145 for (auto &primitive : render_primitives_)
146 {
147 for (auto &pass : primitive->RenderPasses())
148 {
149 if (auto shader_program = pass.RenderProgram(); shader_program)
150 {
151 //There are not many distinct shader programs per drawable object
152 //So std::find with its linear complexity will be the fastest method to make sure each added element is unique
153 if (std::find(std::begin(shader_programs_), std::end(shader_programs_), shader_program) == std::end(shader_programs_))
154 shader_programs_.push_back(shader_program); //Only distinct
155 }
156 }
157 }
158
159 update_render_passes_ = false;
160 }
161
162 //Set render primitive visibility
163 if (!std::empty(render_primitives_))
164 {
165 auto parent_node = ParentNode();
166 auto node_visible = parent_node && parent_node->Visible();
167
168 for (auto &primitive : render_primitives_)
169 {
170 auto world_visible = node_visible && visible_ && primitive->Visible();
171 primitive->WorldVisible(world_visible);
172
173 if (world_visible)
174 primitive->ModelMatrix(parent_node->FullTransformation());
175 }
176 }
177
178 MovableObject::Prepare();
179}

202void DrawableObject::AddRenderPasses(render::render_pass::Passes passes)
203{
204 if (std::empty(render_passes_))
205 render_passes_ = std::move(passes);
206 else
207 std::move(std::begin(passes), std::end(passes), std::back_inserter(render_passes_));
208
209 UpdateRenderPassesOnAllPrimitives(render_passes_);
210}

241bool DrawableObject::RemoveRenderPass(int off) noexcept
242{
243 if (off >= 0 && off < std::ssize(render_passes_))
244 {
245 render_passes_.erase(std::begin(render_passes_) + off);
246 UpdateRenderPassesOnAllPrimitives(render_passes_);
247 return true;
248 }
