24void evolve_particles(container_type<Particle> &particles, duration time) noexcept
25{
26 auto end = std::end(particles);
27 auto last = end;
28
29 for (auto iter = std::begin(particles); iter != last;)
30 {
31 //Particle lifetime has ended
32 if (!iter->Evolve(time))
33 *iter = std::move(*--last); //Move last active particle
34 else
35 ++iter;
36 }
37
38 particles.erase(last, end);
39}

162 Elapse time
163*/
164
165void Emitter::Elapse(duration time) noexcept
166{
167 if (emitting_)
168 {
169 if (emission_duration_ && (*emission_duration_ += time))
170 Reset();
171
172 else if (emission_amount_ += emission_rate_ * time.count())
173 {
174 Emit(static_cast<int>(emission_amount_.Total())); //Emit total number of particles (whole part)
175 emission_amount_.Total(math::Fraction(emission_amount_.Total())); //Keep fractional part for next emission
176 }
177 }
178
179 //Evolve particles (even when not emitting)
180 if (!std::empty(particles_))
181 detail::evolve_particles(particles_, time);
182
183 //Affect particles (even when not emitting)
184 for (auto &affector : Affectors())
185 affector.Affect(Particles(), time);
186}

193void Emitter::Emit(int particle_count) noexcept
194{
195 for (particle_count = std::min(particle_count, particle_quota_ - std::ssize(particles_));
196 particle_count > 0; --particle_count)
197
198 particles_.emplace_back(
199 position_ + detail::particle_position(type_, size_ * 0.5_r, inner_size_ * 0.5_r), //Position
200 detail::particle_direction(direction_, emission_angle_, particle_velocity_.first, particle_velocity_.second), //Direction + velocity
201 detail::particle_size(particle_size_.first, particle_size_.second), //Size
202 detail::particle_mass(particle_mass_.first, particle_mass_.second), //Mass
203 detail::particle_color(particle_color_.first, particle_color_.second), //Color
204 detail::particle_lifetime(particle_lifetime_.first, particle_lifetime_.second), //Lifetime
205 direction_ //Initial direction
206 );
207}
