51Matrix2 Matrix2::Reflection(real angle) noexcept
52{
53 auto sin_of_angle = math::Sin(angle * 2.0_r);
54 auto cos_of_angle = math::Cos(angle * 2.0_r);
55
56 #ifdef ION_ROW_MAJOR
57 //Row-major layout (Direct3D)
58 //Same for both left and right handed rotation
59 return {cos_of_angle, sin_of_angle,
60 sin_of_angle, -cos_of_angle};
61 #else
62 //Column-major layout (OpenGL)
63 //Same for both left and right handed rotation
64 return {cos_of_angle, sin_of_angle,
65 sin_of_angle, -cos_of_angle};
66 #endif
67}

69Matrix2 Matrix2::Rotation(real angle) noexcept
70{
71 auto sin_of_angle = math::Sin(angle);
72 auto cos_of_angle = math::Cos(angle);
73
74 #ifdef ION_ROW_MAJOR
75 //Row-major layout (Direct3D)
76 #ifdef ION_LEFT_HANDED
77 //Left-hand rotation CW
78 return {cos_of_angle, -sin_of_angle,
79 sin_of_angle, cos_of_angle};
80 #else
81 //Right-hand rotation CCW
82 return {cos_of_angle, sin_of_angle,
83 -sin_of_angle, cos_of_angle};
84 #endif
85 #else
86 //Column-major layout (OpenGL)
87 #ifdef ION_LEFT_HANDED
88 //Left-hand rotation CW
89 return {cos_of_angle, sin_of_angle,
90 -sin_of_angle, cos_of_angle};
91 #else
92 //Right-hand rotation CCW
93 return {cos_of_angle, -sin_of_angle,
94 sin_of_angle, cos_of_angle};
95 #endif
96 #endif
97}

118Matrix2 Matrix2::Transformation(real rotation, const Vector2 &scaling) noexcept
119{
120 auto rot = Rotation(rotation);
121
122 #ifdef ION_ROW_MAJOR
123 //Row-major layout (Direct3D)
124 return {rot.M00() * scaling.X(), rot.M01() * scaling.X(),
125 rot.M10() * scaling.Y(), rot.M11() * scaling.Y()};
126 #else
127 //Column-major layout (OpenGL)
128 return {rot.M00() * scaling.X(), rot.M01() * scaling.Y(),
129 rot.M10() * scaling.X(), rot.M11() * scaling.Y()};
130 #endif
131}
