310 inline auto is_caption_outside(ControlCaptionLayout caption_layout) noexcept
311 {
312 switch (caption_layout)
313 {
314 case ControlCaptionLayout::TopLeft:
315 case ControlCaptionLayout::TopCenter:
316 case ControlCaptionLayout::TopRight:
317 case ControlCaptionLayout::Left:
318 case ControlCaptionLayout::Center:
319 case ControlCaptionLayout::Right:
320 case ControlCaptionLayout::BottomLeft:
321 case ControlCaptionLayout::BottomCenter:
322 case ControlCaptionLayout::BottomRight:
323 return false;
324
325 default:
326 return true;
327 }
328 }

330 inline auto caption_layout_to_text_alignment(ControlCaptionLayout caption_layout) noexcept
331 {
332 switch (caption_layout)
333 {
334 case ControlCaptionLayout::Left:
335 case ControlCaptionLayout::TopLeft:
336 case ControlCaptionLayout::BottomLeft:
337 case ControlCaptionLayout::OutsideRightTop:
338 case ControlCaptionLayout::OutsideRightCenter:
339 case ControlCaptionLayout::OutsideRightBottom:
340 case ControlCaptionLayout::OutsideTopLeft:
341 case ControlCaptionLayout::OutsideBottomLeft:
342 return graphics::fonts::text::TextAlignment::Left;
343
344 case ControlCaptionLayout::Right:
345 case ControlCaptionLayout::TopRight:
346 case ControlCaptionLayout::BottomRight:
347 case ControlCaptionLayout::OutsideLeftTop:
348 case ControlCaptionLayout::OutsideLeftCenter:
349 case ControlCaptionLayout::OutsideLeftBottom:
350 case ControlCaptionLayout::OutsideTopRight:
351 case ControlCaptionLayout::OutsideBottomRight:
352 return graphics::fonts::text::TextAlignment::Right;
353
354 default:
355 return graphics::fonts::text::TextAlignment::Center;
356 }
357 }
358
359 inline auto caption_layout_to_text_vertical_alignment(ControlCaptionLayout caption_layout) noexcept
360 {
361 switch (caption_layout)
362 {
363 case ControlCaptionLayout::TopLeft:
364 case ControlCaptionLayout::TopCenter:
365 case ControlCaptionLayout::TopRight:
366 case ControlCaptionLayout::OutsideLeftTop:
367 case ControlCaptionLayout::OutsideRightTop:
368 case ControlCaptionLayout::OutsideBottomLeft:
369 case ControlCaptionLayout::OutsideBottomCenter:
370 case ControlCaptionLayout::OutsideBottomRight:
371 return graphics::fonts::text::TextVerticalAlignment::Top;
372
373 case ControlCaptionLayout::BottomLeft:
374 case ControlCaptionLayout::BottomCenter:
375 case ControlCaptionLayout::BottomRight:
376 case ControlCaptionLayout::OutsideLeftBottom:
377 case ControlCaptionLayout::OutsideRightBottom:
378 case ControlCaptionLayout::OutsideTopLeft:
379 case ControlCaptionLayout::OutsideTopCenter:
380 case ControlCaptionLayout::OutsideTopRight:
381 return graphics::fonts::text::TextVerticalAlignment::Bottom;
382
383 default:
384 return graphics::fonts::text::TextVerticalAlignment::Middle;
385 }
