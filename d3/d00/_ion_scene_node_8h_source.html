94 template <typename Compare = node_comparator>
95 inline void add_node(node_container &dest_nodes, SceneNode *node, Compare compare = Compare{})
96 {
97 //Search for first scene node with greater z-order
98 auto iter = std::upper_bound(std::begin(dest_nodes), std::end(dest_nodes), node, compare);
99 dest_nodes.insert(iter, node);
100 }
101
102 template <typename Compare = node_comparator>
103 inline void move_nodes(node_container &dest_nodes, node_container &source_nodes, Compare compare = Compare{})
104 {
105 auto size = std::size(dest_nodes);
106
107 //Insert source nodes to the back of dest nodes
108 dest_nodes.insert(
109 std::end(dest_nodes),
110 std::begin(source_nodes), std::end(source_nodes));
111
112 //One or more source node has been inserted
113 if (auto first = std::begin(dest_nodes) + size; first != std::begin(dest_nodes))
114 //Merge dest and source nodes
115 std::inplace_merge(
116 std::begin(dest_nodes), first,
117 std::end(dest_nodes), compare);
118
119 source_nodes.clear(); //Moved
120 source_nodes.shrink_to_fit();
121 }

124 template <typename Compare = node_comparator>
125 inline void remove_node(node_container &from_nodes, SceneNode *node, Compare compare = Compare{}) noexcept
126 {
127 //Search for first scene node with equal z-order
128 if (auto first = std::lower_bound(std::begin(from_nodes), std::end(from_nodes), node, compare);
129 first != std::end(from_nodes) && !compare(node, *first)) //Found z-order
130 {
131 //Search for first scene node with greater z-order
132 auto last = std::upper_bound(first, std::end(from_nodes), node, compare);
133
134 //Search for exact scene node in range [first, last)
135 if (first = std::find(first, last, node); first != last) //Found exact
136 from_nodes.erase(first);
137 }
138 }
139
140 template <typename Compare = node_comparator>
141 inline void remove_nodes(node_container &from_nodes, node_container &nodes, Compare compare = Compare{}) noexcept
142 {
143 if (std::empty(nodes))
144 return;
145
146 if (std::size(nodes) == 1)
147 {
148 remove_node(from_nodes, nodes.front(), compare);
149 return;
150 }
151
152 //Search for first scene node with equal z-order
153 if (auto first = std::lower_bound(std::begin(from_nodes), std::end(from_nodes), nodes.front(), compare);
154 first != std::end(from_nodes) && !compare(nodes.front(), *first)) //Found z-order
155 {
156 //Search for first scene node with greater z-order
157 auto last = std::upper_bound(first, std::end(from_nodes), nodes.back(), compare);
158
159 for (auto iter = first; auto &node : nodes)
160 {
161 //Search for each exact scene node in range [first, last)
162 //Range is narrowed for each node found
163 if (iter = std::find(iter, last, node); iter != last) //Found exact
164 {
165 *iter = nullptr; //Tag
166 ++iter;
167 }
168 }
169
170 //Erase all tagged in range [first, last)
171 from_nodes.erase(
172 std::remove_if(first, last,
173 [](auto &node) noexcept
174 {
175 return !node;
176 }),
177 last);
178 }
179 }

227 inline void remove_objects(Container &from_objects, Container &objects, Compare compare = Compare{}) noexcept
228 {

250 *iter = nullptr; //Tag

257 std::remove_if(first, last,
258 [](auto &object) noexcept
259 {
260 return !object;
261 }),
262 last);

584 inline void Position(const Vector2 &position) noexcept
585 {
586 auto position3 =
587 Vector3{position.X(), position.Y(), position_.Z()};
588
589 if (position_ != position3)
590 {
591 position_ = position3;
592 NotifyUpdate();
593 }
594 }
595
597 inline void Position(const Vector3 &position) noexcept
598 {
599 if (position_ != position)
600 {
601 auto z_changed = position_.Z() != position.Z();
602
603 if (z_changed && parent_node_)
604 {
605 GatherNodes(ordered_nodes_);
606 RemoveNodes(RootNode().ordered_nodes_, ordered_nodes_);
607 }
608
609 position_ = position;
610 NotifyUpdate();
611
612 if (z_changed)
613 {
614 NotifyUpdateZ();
615
616 if (parent_node_)
617 MoveNodes(RootNode().ordered_nodes_, ordered_nodes_);
618 }
619 }
