36void move_to(const Vector3 &position) noexcept
37{
38 auto [x, y, z] = position.XYZ();
39
40 if constexpr (std::is_same_v<real, float>)
41 glTranslatef(static_cast<float>(-x), static_cast<float>(-y), static_cast<float>(-z));
42 else
43 glTranslated(static_cast<double>(-x), static_cast<double>(-y), static_cast<double>(-z));
44}
45
46void rotate_by(real angle) noexcept
47{
48 #ifdef ION_LEFT_HANDED
49 //Left-hand rotation CW
50 if constexpr (std::is_same_v<real, float>)
51 glRotatef(static_cast<float>(math::ToDegrees(angle)), 0.0f, 0.0f, 1.0f); //Rotate around the z-axis
52 else
53 glRotated(static_cast<double>(math::ToDegrees(angle)), 0.0, 0.0, 1.0); //Rotate around the z-axis
54 #else
55 //Right-hand rotation CCW
56 if constexpr (std::is_same_v<real, float>)
57 glRotatef(static_cast<float>(-math::ToDegrees(angle)), 0.0f, 0.0f, 1.0f); //Rotate around the z-axis
58 else
59 glRotated(static_cast<double>(-math::ToDegrees(angle)), 0.0, 0.0, 1.0); //Rotate around the z-axis
60 #endif
61}

74void Camera::PrepareBoundingVolumes() noexcept
75{
76 auto [left, right, bottom, top, z_near, z_far] =
77 frustum_.ToOrthoBounds(viewport_size_);
78
79 aabb_ = Aabb::Size({right - left, top - bottom}, position_).RotateCopy(rotation_);
80 obb_ = aabb_;
81 sphere_ = {aabb_.ToHalfSize().Max(), aabb_.Center()};
82}

123void Camera::CaptureScene(const render::Viewport &viewport) noexcept
124{
125 if (auto viewport_size = viewport.Bounds().ToSize();
126 viewport_size_ != viewport_size)
127 {
128 viewport_size_ = viewport_size;
129 update_bounding_volumes_ = true;
130 }
131
132 frustum_.ProjectScene(viewport_size_);
133
134 auto position = position_;
135 auto rotation = rotation_;
136
137 if (auto parent_node = ParentNode(); parent_node)
138 {
139 position += parent_node->DerivedPosition();
140 rotation += parent_node->DerivedRotation();
141 }
142
143 detail::rotate_by(rotation);
144 detail::move_to(position);
145 view_matrix_ = detail::get_view_matrix(position, rotation);
146
147 if (update_bounding_volumes_)
148 {
149 PrepareBoundingVolumes();
150 update_bounding_volumes_ = false;
151 }
152}
