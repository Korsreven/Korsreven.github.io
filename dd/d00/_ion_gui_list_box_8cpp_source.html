38void resize_skin(ListBoxSkin &skin, const Vector2 &from_size, const Vector2 &to_size) noexcept
39{
40 auto delta_size = to_size - from_size;
41 auto delta_position = delta_size * 0.5_r;
42
43 if (skin.Selection)
44 {
45 auto &center = skin.Selection->Position();
46 gui_control::detail::resize_part(skin.Selection, delta_size, delta_position, center);
47 }
48
49 for (auto &icon : skin.Icons)
50 {
51 if (icon)
52 gui_control::detail::resize_part(*icon, delta_size, delta_position, vector2::Zero);
53 }
54}

85Vector2 lines_area_offset(ListBoxIconLayout icon_layout, const Vector2 &icon_column_size) noexcept
86{
87 auto half_size = icon_column_size * 0.5_r;
88
89 switch (icon_layout)
90 {
91 case ListBoxIconLayout::Left:
92 return {half_size.X(), 0.0_r};
93
94 case ListBoxIconLayout::Right:
95 return {-half_size.X(), 0.0_r};
96
97 default:
98 return {0.0_r, 0.0_r};
99 }

131void GuiListBox::Scrolled(int delta) noexcept
132{
133 static auto do_scroll = true;
134
135 if (auto skin = static_cast<ListBoxSkin*>(skin_.get());
136 skin && skin->Lines && do_scroll)
137 {
138 //Lines text
139 if (auto &lines = skin->Lines->Get(); lines)
140 {
141 auto from_line = lines->FromLine();
142
143 //Scrolled up
144 if (delta < 0)
145 {
146 lines->FromLine(
147 from_line > -delta ?
148 from_line + delta : 0);
149 }
150 //Scrolled down
151 else if (delta > 0)
152 {
153 auto line_count = lines->LineCount();
154 auto displayed_line_count = lines->DisplayedLineCount();
155 auto max_from_line =
156 line_count > displayed_line_count ?
157 line_count - displayed_line_count : 0;
158
159 lines->FromLine(
160 from_line + delta < max_from_line ?
161 from_line + delta : max_from_line);
162 }
163
164 UpdateIcons();
165 UpdateSelection();
166
167 do_scroll = false;
168 UpdateScrollBar();
169 do_scroll = true;
170 }
171 }

295void GuiListBox::UpdateLines() noexcept
296{
297 if (auto skin = static_cast<ListBoxSkin*>(skin_.get());
298 skin && skin->Lines)
299 {
300 //Lines text
301 if (auto &text = skin->Lines->Get(); text)
302 {
303 if (auto size = InnerSize(); size)
304 {
305 auto center = CenterArea().value_or(aabb::Zero).Center();
306 auto icon_column_size = show_icons_ ?
307 *size * Vector2{icon_column_width_.value_or(detail::default_icon_column_width_percent), 0.0_r} :
308 vector2::Zero;
309
310 auto ppu = Engine::PixelsPerUnit();
311 text->Overflow(graphics::fonts::text::TextOverflow::WordTruncate);
312 text->AreaSize((*size - icon_column_size) * ppu);
313 text->LineHeightFactor(item_height_factor_.value_or(detail::default_item_height_factor));
314 text->Padding(item_padding_.value_or(detail::default_item_padding_size));
315 text->Alignment(detail::item_layout_to_text_alignment(item_layout_));
316
317 //Refresh content
318 if (text->LineCount() != std::ssize(items_))
319 text->Content(detail::items_to_text_content(items_));
320
321 skin->Lines->Position(center + detail::lines_area_offset(icon_layout_, icon_column_size));
322 }
323 }
324 }
325
326 UpdateIcons();
327 UpdateSelection();
328}

330void GuiListBox::UpdateIcons() noexcept
331{
332 if (auto skin = static_cast<ListBoxSkin*>(skin_.get());
333 skin && skin->Lines)
334 {
335 //Lines text
336 if (auto &text = skin->Lines->GetImmutable();
337 text && text->LineHeight())
338 {
339 if (auto size = InnerSize(); size)
340 {
341 auto [width, height] = size->XY();
342 auto center = CenterArea().value_or(aabb::Zero).Center();
343
344 auto ppu = Engine::PixelsPerUnit();
345 auto item_height = *text->LineHeight() / ppu;
346 auto item_padding = text->Padding().Y() / ppu;
347 auto icon_padding = icon_padding_.value_or(detail::default_icon_padding_size) / ppu;
348
349 auto icon_column_percent = icon_column_width_.value_or(detail::default_icon_column_width_percent);
350 auto icon_column_width = width * icon_column_percent;
351 auto icon_max_size = Vector2{icon_column_width, item_height} - icon_padding * 2.0_r;
352 icon_max_size.Floor(icon_max_size_.value_or(icon_max_size)).Ceil(vector2::Zero);
353
354 auto item_count = show_icons_ ? text->DisplayedLineCount() : 0;
355
356 //Show all icons in view
357 for (auto item_off = text->FromLine(), icon_off = 0;
358 icon_off < item_count; ++item_off, ++icon_off)
359 {
360 //Create icon sprite (if missing)
361 if (icon_off == std::ssize(skin->Icons))
362 skin->Icons.push_back(CreateIcon(items_[item_off].Icon));
363 else if (!skin->Icons[icon_off])
364 skin->Icons[icon_off] = CreateIcon(items_[item_off].Icon);
365 else
366 skin->Icons[icon_off]->SurfaceMaterial(items_[item_off].Icon);
367
368 //Item has icon
369 if (items_[item_off].Icon)
370 {
371 skin->Icons[icon_off]->AutoSize(true);
372 skin->Icons[icon_off]->ResizeToFit(icon_max_size);
373
374 skin->Icons[icon_off]->Position({
375 icon_layout_ == ListBoxIconLayout::Right ?
376 center.X() + width * 0.5_r - icon_column_width * 0.5_r : //Right
377 center.X() - width * 0.5_r + icon_column_width * 0.5_r, //Left
378 height * 0.5_r - item_padding - item_height * 0.5_r - item_height * icon_off,
379 skin->Icons[icon_off]->Position().Z()}
380 );
381 }
382
383 skin->Icons[icon_off]->Visible(!!items_[item_off].Icon);
384 }
385
386 //Hide all icons out of view
387 for (auto iter = std::begin(skin->Icons) + item_count,
388 end = std::end(skin->Icons); iter != end; ++iter)
389 {
390 if (*iter)
391 (*iter)->Visible(false);
392 }
393 }
394 }
395 }

398void GuiListBox::UpdateSelection() noexcept
399{
400 if (auto skin = static_cast<ListBoxSkin*>(skin_.get());
401 skin && skin->Selection)
402 {
403 auto show_selection = false;
404
405 if (skin->Lines && item_index_)
406 {
407 //Lines text
408 if (auto &text = skin->Lines->GetImmutable();
409 text && text->LineHeight())
410 {
411 //Selected item is in view
412 if (auto item_off = *item_index_ - text->FromLine();
413 item_off >= 0 && item_off < text->DisplayedLineCount())
414 {
415 if (auto size = InnerSize(); size)
416 {
417 auto [width, height] = size->XY();
418 auto center = CenterArea().value_or(aabb::Zero).Center();
419
420 auto ppu = Engine::PixelsPerUnit();
421 auto item_height = *text->LineHeight() / ppu;
422 auto item_padding = text->Padding().Y() / ppu;
423 auto item_selection_padding = selection_padding_.
424 value_or(detail::default_selection_padding_size) / ppu;
425
426 skin->Selection->Size(
427 (Vector2{width, item_height} - item_selection_padding * 2.0_r).
428 CeilCopy(vector2::Zero)
429 );
430 skin->Selection->Position({
431 center.X(),
432 height * 0.5_r - item_padding - item_height * 0.5_r - item_height * item_off,
433 skin->Selection->Position().Z()}
434 );
435
436 show_selection = true;
437 }
438 }
439 }
440 }
441
442 skin->Selection->Visible(show_selection);
443 }

576void GuiListBox::ItemIndex(std::optional<int> index) noexcept
577{
578 if (index)
579 {
580 if (*index < 0 || std::empty(items_))
581 index = {};
582 else
583 index = std::clamp(*index, 0, std::ssize(items_) - 1);
584 }
585
586 if (item_index_ != index)
587 {
588 //Go from or to an empty selection
589 auto empty_selection =
590 !item_index_ || !index;
591
592 item_index_ = index;
593
594 if (item_index_)
595 ItemSelected();
596 else
597 ItemDeselected();
598
599 if (empty_selection)
600 SetState(state_);
601
602 UpdateSelection();
603 }
604}

689void GuiListBox::ReplaceItems(int first, int last, ListBoxItems items)
690{
691 if (first >= 0 && first < last)
692 {
693 detail::trim_items(items);
694
695 last = std::clamp(last, first, std::ssize(items_));
696 items_.erase(std::begin(items_) + first, std::begin(items_) + last);
697 items_.insert(std::begin(items_) + first, std::begin(items), std::end(items));
698
699 //Adjust item index
700 if (item_index_)
701 {
702 if (*item_index_ >= first && *item_index_ < last &&
703 *item_index_ - first >= std::ssize(items))
704 ItemIndex({}); //Deselect
705 else if (*item_index_ >= last)
706 ItemIndex(*item_index_ + (std::ssize(items) - (last - first)));
707 }
708
709 ReplaceLines(first, last, items);
710 UpdateLines();
711 }
712}

763/*
764 Key events
765*/
766
767bool GuiListBox::KeyReleased(KeyButton button) noexcept
768{
769 auto update_selection =
770 [&]() noexcept
771 {
772 switch (button)
773 {
774 //Move selection up
775 case KeyButton::UpArrow:
776 case KeyButton::LeftArrow:
777 {
778 if (!item_index_)
779 ItemIndex(0);
780 else if (*item_index_ > 0)
781 ItemIndex(*item_index_ - 1);
782
783 return true;
784 }
785
786 //Move selection down
787 case KeyButton::DownArrow:
788 case KeyButton::RightArrow:
789 {
790 if (!item_index_)
791 ItemIndex(0);
792 else if (*item_index_ + 1 < std::ssize(items_))
793 ItemIndex(*item_index_ + 1);
794
795 return true;
796 }
797
798 //Move selection to top
799 case KeyButton::Home:
800 {
801 if (!std::empty(items_))
802 ItemIndex(0);
803
804 return true;
805 }
806
807 //Move selection to bottom
808 case KeyButton::End:
809 {
810 if (!std::empty(items_))
811 ItemIndex(std::ssize(items_) - 1);
812
813 return true;
814 }
815
816 default:
817 return false;
818 }
819 }();
820
821 if (update_selection)
822 {
823 if (item_index_)
824 {
825 auto first = ScrollPosition();
826 auto view_count = ElementsInView();
827
828 if (*item_index_ < first)
829 Scrolled(*item_index_ - first);
830 else if (*item_index_ > first + view_count - 1)
831 Scrolled(*item_index_ - (first + view_count - 1));
832 }
833
834 return true;
835 }
836
837 return GuiScrollable::KeyReleased(button); //Use base functionality
838}
