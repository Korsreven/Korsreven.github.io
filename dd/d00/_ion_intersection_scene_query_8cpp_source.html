24bool sphere_sphere_hit(const scene_query::detail::query_object &sphere_object, const scene_query::detail::query_object &sphere_object2) noexcept
25{
26 //Sphere vs Sphere
27 return sphere_object.first->WorldBoundingSphere(false).
28 Intersects(sphere_object2.first->WorldBoundingSphere(false));
29}

31bool box_box_hit(scene_query::detail::query_object &box_object, scene_query::detail::query_object &box_object2) noexcept
32{
33 //Aabb vs Aabb
34 if (box_object.first->WorldAxisAlignedBoundingBox(false).
35 Intersects(box_object2.first->WorldAxisAlignedBoundingBox(false)))
36 {
37 if (box_object.first->ParentNode()->AxisAligned() &&
38 box_object2.first->ParentNode()->AxisAligned())
39 return true;
40
41 else //Obb vs Obb
42 return box_object.first->WorldOrientedBoundingBox(derive_once(box_object.second)).
43 Intersects(box_object2.first->WorldOrientedBoundingBox(derive_once(box_object2.second)));
44 }
45
46 return false;
47}

49bool sphere_box_hit(const scene_query::detail::query_object &sphere_object, scene_query::detail::query_object &box_object) noexcept
50{
51 //Sphere vs Aabb
52 if (sphere_object.first->WorldBoundingSphere(false).
53 Intersects(box_object.first->WorldAxisAlignedBoundingBox(false)))
54 {
55 if (box_object.first->ParentNode()->AxisAligned())
56 return true;
57
58 else //Sphere vs Obb
59 return sphere_object.first->WorldBoundingSphere(false).
60 Intersects(box_object.first->WorldOrientedBoundingBox(derive_once(box_object.second)));
61 }
62
63 return false;
64}

67ResultType intersects(scene_query::detail::query_objects &objects) noexcept
68{
69 ResultType result;
70
71 for (auto iter = std::begin(objects), end = std::end(objects); iter != end; ++iter)
72 {
73 for (auto iter2 = iter + 1; iter2 != end; ++iter2)
74 {
75 if (iter->first->QueryFlags().value_or(~0_ui32) &
76 iter2->first->QueryMask().value_or(~0_ui32))
77 {
78 switch (iter->first->PreferredBoundingVolume())
79 {
80 case movable_object::PreferredBoundingVolumeType::BoundingSphere:
81 {
82 switch (iter2->first->PreferredBoundingVolume())
83 {
84 case movable_object::PreferredBoundingVolumeType::BoundingSphere:
85 {
86 if (sphere_sphere_hit(*iter, *iter2))
87 result.emplace_back(iter->first, iter2->first);
88 break;
89 }
90
91 case movable_object::PreferredBoundingVolumeType::BoundingBox:
92 default:
93 {
94 if (sphere_box_hit(*iter, *iter2))
95 result.emplace_back(iter->first, iter2->first);
96 break;
97 }
98 }
99
100 break;
101 }
102
103 case movable_object::PreferredBoundingVolumeType::BoundingBox:
104 default:
105 {
106 switch (iter2->first->PreferredBoundingVolume())
107 {
108 case movable_object::PreferredBoundingVolumeType::BoundingSphere:
109 {
110 if (sphere_box_hit(*iter2, *iter)) //Flipped
111 result.emplace_back(iter->first, iter2->first);
112 break;
113 }
114
115 case movable_object::PreferredBoundingVolumeType::BoundingBox:
116 default:
117 {
118 if (box_box_hit(*iter, *iter2))
119 result.emplace_back(iter->first, iter2->first);
120 break;
121 }
122 }
123
124 break;
125 }
126 }
127 }
128 }
129 }
130
131 return result;
132}

148ResultType IntersectionSceneQuery::Execute() const noexcept
149{
150 if (!scene_graph_)
151 return {};
152
153 auto objects =
154 scene_query::detail::get_eligible_objects(
155 scene_graph_->RootNode(), query_mask_.value_or(~0_ui32), query_type_mask_.value_or(~0_ui32), only_visible_objects_);
156 scene_query::detail::derive_bounding_volumes(objects);
157
158 if (query_region_)
159 scene_query::detail::remove_objects_outside_region(objects, *query_region_);
160
161 return detail::intersects(objects);
162}
