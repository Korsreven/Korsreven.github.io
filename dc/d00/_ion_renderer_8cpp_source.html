124NonOwningPtr<renderer::detail::render_batch> Renderer::InsertBatch(renderer::detail::render_batches::iterator where, const RenderPrimitive &primitive)
125{
126 auto offset =
127 where == std::begin(batches_) ? 0 :
128 (*(where - 1))->offset + (*(where - 1))->capacity;
129
130 where = batches_.insert(where,
131 make_owning<renderer::detail::render_batch>(
132 primitive.WorldZ(), offset, 0, primitive.MakeVertexBatch())
133 );
134
135 auto vertex_data_size = primitive.VertexDataSize();
136 auto initial_size = std::max(vertex_data_size, initial_batch_data_size_);
137 GrowBatch(where, initial_size);
138 return *where;
139}

141void Renderer::GroupWithBatch(RenderPrimitive &primitive, renderer::detail::render_batch &batch)
142{
143 auto vertex_data_size = primitive.VertexDataSize();
144 auto min_space_left = std::optional<int>();
145 auto where = std::make_optional(std::end(batch.slots));
146
147 for (auto iter = std::begin(batch.slots), end = *where; iter != end; ++iter)
148 {
149 //Check if slot is occupied
150 if (iter->primitive)
151 continue;
152
153 if (auto space_left = iter->capacity - vertex_data_size;
154 space_left >= 0) //Has space to store primitive
155 {
156 if (space_left < min_space_left.value_or(space_left + 1))
157 {
158 min_space_left = space_left;
159 *where = iter; //Record slot that fits the primitive best
160 }
161
162 //No space left, exact fit
163 if (space_left == 0)
164 break;
165 }
166 }
167
168 //Add primitive to slot (has space)
169 if (where != std::end(batch.slots))
170 {
171 (*where)->primitive = &primitive;
172 (*where)->need_update = true;
173
174 //Has space left
175 if (*min_space_left > 0)
176 {
177 (*where)->capacity -= *min_space_left;
178
179 //Add empty slot with the remaining space
180 batch.slots.emplace(*where + 1, *min_space_left, nullptr);
181 }
182 }
183 else //Could not be added to any existing slots (no slots with space available)
184 {
185 batch.slots.emplace_back(vertex_data_size, &primitive);
186 batch.used_capacity += batch.slots.back().capacity;
187 }
188}

190void Renderer::GrowBatch(renderer::detail::render_batches::iterator where, int size)
191{
192 auto geometric = (*where)->capacity + (*where)->capacity / 2;
193 auto grow_size = geometric < (*where)->capacity + size ? size : geometric - (*where)->capacity;
194
195 //Reallocate if not enough capacity
196 if (std::ssize(vertex_data_) < used_capacity_ + grow_size)
197 {
198 vertex_data_.insert(std::end(vertex_data_), grow_size, 0.0_r); //Reallocates!
199
200 //Fill up allocated space (using vectors growth strategy)
201 if (std::size(vertex_data_) < vertex_data_.capacity())
202 {
203 auto size_left = vertex_data_.capacity() - std::size(vertex_data_);
204 vertex_data_.insert(std::end(vertex_data_), size_left, 0.0_r); //Does not reallocate
205 }
206
207 need_update_ = true; //Vertex data has reallocated
208 }
209
210 //Batch is not last, copy data to the right
211 if (where + 1 != std::end(batches_))
212 {
213 auto iter = where + 1;
214 std::copy_backward(std::begin(vertex_data_) + (*iter)->offset,
215 std::begin(vertex_data_) + used_capacity_,
216 std::begin(vertex_data_) + used_capacity_ + grow_size);
217
218 //Update offset on all succeeding batches
219 for (auto end = std::end(batches_); iter != end; ++iter)
220 (*iter)->offset += grow_size;
221 }
222
223 used_capacity_ += grow_size;
224 (*where)->capacity += grow_size;
225 (*where)->need_update = detail::update_status::YesSuccessive;
226}

228detail::groupable_status Renderer::IsGroupable(const RenderPrimitive &primitive, const renderer::detail::render_batch &batch) const noexcept
229{
230 if (auto iter =
231 std::find_if(std::begin(batch.slots), std::end(batch.slots),
232 [&](auto &slot) noexcept
233 {
234 return !!slot.primitive;
235 }); iter != std::end(batch.slots))
236 //Batch not empty, check if groupable with first primitive
237 return primitive.IsGroupable(*iter->primitive) ?
238 detail::groupable_status::Yes : detail::groupable_status::No;
239 else //Batch empty
240 return detail::groupable_status::YesEmpty;
241}

244void Renderer::RefreshPrimitives()
245{
246 //Collect all primitives in batches
247 for (auto &batch : batches_)
248 {
249 for (auto &slot : batch->slots)
250 {
251 if (slot.primitive)
252 pending_primitives_.push_back(slot.primitive);
253 }
254 }
255
256 //Collect all primitives just added
257 for (auto &primitive : added_primitives_)
258 pending_primitives_.push_back(primitive);
259
260 //Collect all hidden primitives
261 for (auto &primitive : hidden_primitives_)
262 pending_primitives_.push_back(primitive);
263
264
265 //Refresh all primitives
266 for (auto &primitive : pending_primitives_)
267 primitive->Refresh();
268
269 pending_primitives_.clear();
270}

272void Renderer::GroupAddedPrimitives()
273{
274 //Sort added primitives before grouping
275 std::stable_sort(std::begin(added_primitives_), std::end(added_primitives_),
276 [](auto &primitive, auto &primitive2) noexcept
277 {
278 return primitive->WorldZ() < primitive2->WorldZ(); //Asc
279 });
280
281 for (auto &primitive : added_primitives_)
282 {
283 auto z = primitive->WorldZ();
284 auto where_to_group = std::optional<detail::render_batches::iterator>{};
285 auto where_to_insert = std::make_optional(std::begin(batches_));
286
287 //Search from right to left
288 for (auto iter = std::rbegin(batches_), end = std::rend(batches_); iter != end; ++iter)
289 {
290 if ((*iter)->z <= z)
291 {
292 auto status = detail::groupable_status::No;
293
294 //Check if primitive can be grouped with batch
295 if ((*iter)->z == z &&
296 (status = IsGroupable(*primitive, **iter)) != detail::groupable_status::No)
297 {
298 if (status == detail::groupable_status::Yes)
299 {
300 GroupWithBatch(*primitive, **iter);
301 where_to_group = {};
302 where_to_insert = {};
303 break;
304 }
305 else //YesEmpty
306 where_to_group = iter.base() - 1; //Record empty batch (not optimal)
307 }
308 else if (*where_to_insert == std::begin(batches_))
309 *where_to_insert = iter.base(); //Record batch insertion point
310
311 if ((*iter)->z < z)
312 break;
313 }
314 }
315
316 //Can only be grouped with empty batch
317 if (where_to_group)
318 {
319 (**where_to_group)->used_capacity = 0;
320 (**where_to_group)->last_used_capacity = 0;
321 (**where_to_group)->slots.clear();
322 (**where_to_group)->vertex_batch = primitive->MakeVertexBatch();
323 (**where_to_group)->need_update = detail::update_status::Yes;
324 GroupWithBatch(*primitive, ***where_to_group);
325 }
326 //Could not be grouped with any existing batches
327 else if (where_to_insert)
328 {
329 auto ptr = InsertBatch(*where_to_insert, *primitive);
330 GroupWithBatch(*primitive, *ptr);
331 }
332 }
333
334 added_primitives_.clear();
335}

337void Renderer::UpdateBatchSlots()
338{
339 for (auto &batch : batches_)
340 {
341 //Find slots that does not fit primitive any longer
342 for (auto iter = std::begin(batch->slots), end = std::end(batch->slots); iter != end; ++iter)
343 {
344 if (iter->primitive && //Primitive data size has changed
345 iter->primitive->VertexDataSize() != iter->capacity)
346 {
347 auto off = iter - std::begin(batch->slots);
348
349 auto primitive = iter->primitive;
350 iter->primitive = nullptr;
351 iter->need_update = true;
352 GroupWithBatch(*primitive, *batch);
353
354 //Restart from off
355 iter = std::begin(batch->slots) + off;
356 end = std::end(batch->slots);
357 }
358 }
359 }

362void Renderer::CompressBatches() noexcept
363{
364 for (auto &batch : batches_)
365 {
366 //Find first empty slot
367 auto iter =
368 std::find_if(std::begin(batch->slots), std::end(batch->slots),
369 [&](auto &slot) noexcept
370 {
371 return !slot.primitive;
372 });
373
374 if (iter == std::end(batch->slots))
375 continue; //Nothing to compress
376
377 //Find offset of first empty slot
378 auto slot_offset = batch->offset;
379 for (auto it = std::begin(batch->slots); it != iter; ++it)
380 slot_offset += it->capacity;
381
382 //Erase all empty slots
383 batch->slots.erase(
384 std::remove_if(iter, std::end(batch->slots),
385 [&](auto &slot) noexcept
386 {
387 auto empty_slot = !slot.primitive;
388 if (empty_slot)
389 {
390 //Empty slot, copy data to the left
391 std::copy(std::begin(vertex_data_) + slot_offset + slot.capacity,
392 std::begin(vertex_data_) + batch->offset + batch->last_used_capacity,
393 std::begin(vertex_data_) + slot_offset);
394
395 batch->used_capacity -= slot.capacity;
396 }
397 else
398 slot_offset += slot.capacity;
399
400 return empty_slot;
401 }), std::end(batch->slots));
402 }
403}

405void Renderer::UpdateBatches()
406{
407 for (auto iter = std::begin(batches_), end = std::end(batches_); iter != end; ++iter)
408 {
409 //Grow batch if not enough capacity
410 if ((*iter)->capacity < (*iter)->used_capacity)
411 GrowBatch(iter, (*iter)->used_capacity - (*iter)->capacity);
412 }

415void Renderer::PrepareVertexData() noexcept
416{
417 auto out_of_date = false;
418 for (auto &batch : batches_)
419 {
420 auto data_to_upload = std::optional<std::pair<int, int>>{};
421 //A minimal data chunk [offset, size] that must be uploaded to VRAM
422
423 //Prepare all primitives
424 for (auto slot_offset = batch->offset; auto &slot : batch->slots)
425 {
426 slot.need_update |= slot.primitive->PrepareVertexData();
427
428 //Copy data from primitive to vertex data
429 if (slot.need_update)
430 {
431 auto &vertex_data = slot.primitive->WorldVertexData();
432 std::copy(std::begin(vertex_data), std::end(vertex_data), std::begin(vertex_data_) + slot_offset);
433 slot.need_update = false;
434
435 if (!data_to_upload)
436 data_to_upload = std::pair{slot_offset, 0}; //Upload offset
437
438 data_to_upload->second = //Upload size
439 slot_offset + slot.capacity - data_to_upload->first;
440 }
441
442 slot_offset += slot.capacity;
443 }
444
445
446 if (batch->need_update == detail::update_status::YesSuccessive)
447 out_of_date = true; //This and all succeeding batches
448
449 auto batch_need_update = out_of_date ||
450 batch->need_update == detail::update_status::Yes;
451
452 if (batch_need_update || batch->used_capacity != batch->last_used_capacity)
453 {
454 //Vertex data has not reallocated
455 if (!need_update_)
456 {
457 //Has vertex buffer object
458 if (vbo_ && *vbo_ && batch_need_update)
459 batch->vertex_batch.VertexBuffer( //Update vertex buffer
460 vbo_->SubBuffer(batch->offset * sizeof(real), batch->capacity * sizeof(real))
461 );
462
463 //Update vertex data
464 batch->vertex_batch.VertexData(
465 {std::data(vertex_data_) + batch->offset, batch->used_capacity}
466 );
467 }
468
469 batch->need_update = detail::update_status::No;
470 }
471 else if (data_to_upload)
472 {
473 //Vertex data has not reallocated
474 if (!need_update_)
475 {
476 //Has vertex buffer object
477 if (vbo_ && *vbo_)
478 vbo_->Data( //Upload minimal data chunk from RAM to VRAM
479 {std::data(vertex_data_) + data_to_upload->first, data_to_upload->second},
480 data_to_upload->first * sizeof(real)
481 );
482 else //Update vertex data
483 batch->vertex_batch.VertexData(
484 {std::data(vertex_data_) + batch->offset, batch->used_capacity}
485 );
486 }
487 }
488
489 batch->last_used_capacity = batch->used_capacity;
490 }
491
492 //Vertex data has reallocated
493 if (need_update_)
494 {
495 if (!vbo_)
496 vbo_.emplace(render::vertex::vertex_buffer_object::VertexBufferUsage::Dynamic);
497
498 //Has vertex buffer object
499 if (vbo_ && *vbo_)
500 {
501 vbo_->Data(vertex_data_); //Upload data from RAM to VRAM
502
503 //Update all vertex buffers
504 for (auto &batch : batches_)
505 batch->vertex_batch.VertexBuffer(
506 vbo_->SubBuffer(batch->offset * sizeof(real), batch->capacity * sizeof(real)),
507 false //Data already uploaded
508 );
509 }
510
511 //Update all vertex data
512 for (auto &batch : batches_)
513 batch->vertex_batch.VertexData(
514 {std::data(vertex_data_) + batch->offset, batch->used_capacity},
515 false //Data already uploaded
516 );
517
518 need_update_ = false;
519 }

589bool Renderer::AddPrimitive(RenderPrimitive &primitive)
590{
591 if (!primitive.ParentRenderer())
592 {
593 primitive.ParentRenderer(this);
594
595 //Primitive is visible
596 if (primitive.WorldVisible() && primitive.VertexCount() > 0)
597 added_primitives_.push_back(&primitive);
598 else
599 hidden_primitives_.push_back(&primitive);
600
601 return true;
602 }
603 else
604 return false;
605}

607bool Renderer::RefreshPrimitive(RenderPrimitive &primitive)
608{
609 if (primitive.ParentRenderer() != this)
610 return false;
611
612 auto iter_hidden = std::find(std::begin(hidden_primitives_), std::end(hidden_primitives_), &primitive);
613 auto iter_added = iter_hidden == std::end(hidden_primitives_) ?
614 std::find(std::begin(added_primitives_), std::end(added_primitives_), &primitive) :
615 std::end(added_primitives_);
616
617 //Primitive is visible
618 if (primitive.WorldVisible() && primitive.VertexCount() > 0)
619 {
620 //Primitive has just been added, no-op
621 if (iter_added != std::end(added_primitives_))
622 return true;
623
624 //Move from hidden to added, fast
625 else if (iter_hidden != std::end(hidden_primitives_))
626 {
627 hidden_primitives_.erase(iter_hidden);
628 added_primitives_.push_back(&primitive);
629 return true;
630 }
631 }
632 else //Primitive is not visible
633 {
634 //Primitive is already hidden, no-op
635 if (iter_hidden != std::end(hidden_primitives_))
636 return true;
637
638 //Move from added to hidden, fast
639 else if (iter_added != std::end(added_primitives_))
640 {
641 added_primitives_.erase(iter_added);
642 hidden_primitives_.push_back(&primitive);
643 return true;
644 }
645 }
646
647 //Full refresh, slow
648 RemovePrimitive(primitive);
649 AddPrimitive(primitive);
650 return true;
651}

654bool Renderer::RemovePrimitive(RenderPrimitive &primitive) noexcept
655{
656 if (primitive.ParentRenderer() != this)
657 return false;
658
659 //Check if primitive is in a batch
660 for (auto &batch : batches_)
661 {
662 for (auto &slot : batch->slots)
663 {
664 if (slot.primitive == &primitive)
665 {
666 slot.primitive->ParentRenderer(nullptr);
667 slot.primitive = nullptr;
668 return true;
669 }
670 }
671 }
672
673 //Check if primitive has just been added
674 if (auto iter = std::find(std::begin(added_primitives_), std::end(added_primitives_), &primitive);
675 iter != std::end(added_primitives_))
676 {
677 (*iter)->ParentRenderer(nullptr);
678 added_primitives_.erase(iter);
679 return true;
680 }
681
682 //Check if primitive is hidden
683 if (auto iter = std::find(std::begin(hidden_primitives_), std::end(hidden_primitives_), &primitive);
684 iter != std::end(hidden_primitives_))
685 {
686 (*iter)->ParentRenderer(nullptr);
687 hidden_primitives_.erase(iter);
688 return true;
689 }
690
691 return false;
692}

771void Renderer::Draw() noexcept
772{
773 static NonOwningPtr<shaders::ShaderProgram> active_shader_program = nullptr;
774
775 //Draw all batches
776 for (auto &batch : batches_)
777 {
778 //Check if batch has vertex data to draw
779 if (batch->used_capacity == 0)
780 continue;
781
782 auto primitive = batch->slots.front().primitive;
783 //Use first primitive (could be any primitive)
784
785 //For each pass
786 for (auto &pass : primitive->RenderPasses())
787 {
788 auto active_program = active_shader_program.get();
789 auto shader_program = pass.RenderProgram().get();
790 auto use_shader = shader_program && shader_program->Owner() && shader_program->Handle();
791
792 //Switch shader program
793 if (shader_program != active_program)
794 {
795 if (use_shader) //Custom pipeline
796 shader_program->Owner()->ActivateShaderProgram(*shader_program);
797 else if (active_program) //Fixed-function pipeline
798 active_program->Owner()->DeactivateShaderProgram(*active_program);
799
800 active_shader_program = pass.RenderProgram();
801 }
802
803
804 pass.Blend();
805
806 if (primitive->PointSize() != 1.0_r)
807 detail::set_point_size(primitive->PointSize());
808
809 if (primitive->LineThickness() != 1.0_r)
810 detail::set_line_width(primitive->LineThickness());
811
812 if (primitive->WireFrame())
813 detail::enable_wire_frame();
814
815 if (primitive->PointSprite())
816 detail::enable_point_sprite();
817
818
819 for (auto iterations = pass.Iterations(); iterations > 0; --iterations)
820 batch->vertex_batch.Draw(shader_program); //Draw batch
821
822
823 if (primitive->PointSprite())
824 detail::disable_point_sprite();
825
826 if (primitive->WireFrame())
