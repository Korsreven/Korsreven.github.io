97NonOwningPtr<Viewport> RenderTarget::GetViewport(const Vector2 &position) noexcept
98{
99 for (auto &viewport : Viewports())

101 if (viewport.Bounds().Contains(position))
102 return Get(*viewport.Name());
103 }
104
105 return nullptr;
106}
