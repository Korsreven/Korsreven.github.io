194 inline auto make_color_name_map() noexcept
195 {
196 auto color_name_map = color_map;
197 std::stable_sort(std::begin(color_name_map), std::end(color_name_map),
198 [](const auto &x, const auto &y) noexcept
199 {
200 return *x.second < *y.second;
201 });
202 return color_name_map;
203 }

206 constexpr auto is_control_character(char c) noexcept
207 {
208 return c >= '\0' && c < ' ' ||
209 c == '\x7F'; //Delete character (127)

212 constexpr auto parse_as_floating_point(std::string_view str) noexcept
213 {
214 for (auto c : str)
215 {
216 switch (c)
217 {
218 //Exponent
219 case 'e':
220 case 'E':
221
222 //Decimal point
223 case '.':
224 return true;
225 }
226 }
227
228 return false;
229 }
230
231 constexpr auto parse_as_integer(std::string_view str) noexcept
232 {
233 return !parse_as_floating_point(str);
234 }
