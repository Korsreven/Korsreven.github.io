161NonOwningPtr<SoundChannel> Sound::Play(NonOwningPtr<SoundChannel> sound_channel, bool paused)
162{
163 if (!sound_channel)
164 return Play(paused);
165
166 if (handle_)
167 {
168 if (auto system = sound_manager::detail::get_system(*handle_); system)
169 sound_channel->Handle(
170 sound_manager::detail::play_sound(
171 *system, *handle_,
172 sound_channel->CurrentChannelGroup() ? sound_channel->CurrentChannelGroup()->Handle() : nullptr,
173 sound_channel->Handle(),
174 paused)
175 );
176 }
177
178 return sound_channel;
179}
